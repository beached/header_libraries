483 // _GLIBCXX_RESOLVE_LIB_DEFECTS
484 // 2063. Contradictory requirements for string move assignment
485 basic_string &operator=( basic_string &&_str ) noexcept(
486 Alloc_traits::_S_nothrow_move( ) ) {
487 if( !m_is_local( ) && Alloc_traits::_S_propagate_on_move_assign( ) &&
488 !Alloc_traits::_S_always_equal( ) &&
489 m_get_allocator( ) != _str.m_get_allocator( ) ) {
490 // Destroy existing storage before replacing allocator.
491 m_destroy( m_allocated_capacity );
492 m_data( m_local_data( ) );
493 m_set_length( 0 );
494 }
495 // Replace allocator if POCMA is true.
496 std::__alloc_on_move( m_get_allocator( ), _str.m_get_allocator( ) );
497
498 if( _str.m_is_local( ) ) {
499 // We've always got room for a short string, just copy it
500 // (unless this is a self-move, because that would violate the
501 // char_traits::copy precondition that the ranges don't overlap).
502 if( _builtin_expect( std::addressof( _str ) != this, true ) ) {
503 if( _str.size( ) ) {
504 this->S_copy( m_data( ), _str.m_data( ), _str.size( ) );
505 }
506 m_set_length( _str.size( ) );
507 }
508 } else if( Alloc_traits::_S_propagate_on_move_assign( ) ||
509 Alloc_traits::_S_always_equal( ) ||
510 m_get_allocator( ) == _str.m_get_allocator( ) ) {
511 // Just move the allocated pointer, our allocator can free it.
512 pointer _data = nullptr;
513 size_type _capacity;
514 if( !m_is_local( ) ) {
515 if( Alloc_traits::_S_always_equal( ) ) {
516 // _str can reuse our existing storage.
517 _data = m_data( );
518 _capacity = m_allocated_capacity;
519 } else { // _str can't use it, so free it.
520 m_destroy( m_allocated_capacity );
521 }
522 }
523
524 m_data( _str.m_data( ) );
525 m_length( _str.length( ) );
526 m_capacity( _str.m_allocated_capacity );
527 if( _data ) {
528 _str.m_data( _data );
529 _str.m_capacity( _capacity );
530 } else {
531 _str.m_data( _str.m_local_buf );
532 }
533 } else { // Need to do a deep copy
534 assign( _str );
535 }
536 _str.clear( );
537 return *this;
538 }

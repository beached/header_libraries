473  // _GLIBCXX_RESOLVE_LIB_DEFECTS
474  // 2063. Contradictory requirements for string move assignment
475  basic_string &operator=( basic_string &&__str ) noexcept(
476  Alloc_traits::_S_nothrow_move( ) ) {
477  if( !m_is_local( ) && Alloc_traits::_S_propagate_on_move_assign( ) &&
478  !Alloc_traits::_S_always_equal( ) &&
479  m_get_allocator( ) != __str.m_get_allocator( ) ) {
480  // Destroy existing storage before replacing allocator.
481  m_destroy( m_allocated_capacity );
482  m_data( m_local_data( ) );
483  m_set_length( 0 );
484  }
485  // Replace allocator if POCMA is true.
486  std::__alloc_on_move( m_get_allocator( ), __str.m_get_allocator( ) );
487 
488  if( __str.m_is_local( ) ) {
489  // We've always got room for a short string, just copy it
490  // (unless this is a self-move, because that would violate the
491  // char_traits::copy precondition that the ranges don't overlap).
492  if( __builtin_expect( std::__addressof( __str ) != this, true ) ) {
493  if( __str.size( ) ) {
494  this->_S_copy( m_data( ), __str.m_data( ), __str.size( ) );
495  }
496  m_set_length( __str.size( ) );
497  }
498  } else if( Alloc_traits::_S_propagate_on_move_assign( ) ||
499  Alloc_traits::_S_always_equal( ) ||
500  m_get_allocator( ) == __str.m_get_allocator( ) ) {
501  // Just move the allocated pointer, our allocator can free it.
502  pointer __data = nullptr;
503  size_type __capacity;
504  if( !m_is_local( ) ) {
505  if( Alloc_traits::_S_always_equal( ) ) {
506  // __str can reuse our existing storage.
507  __data = m_data( );
508  __capacity = m_allocated_capacity;
509  } else { // __str can't use it, so free it.
510  m_destroy( m_allocated_capacity );
511  }
512  }

514  m_data( __str.m_data( ) );
515  m_length( __str.length( ) );
516  m_capacity( __str.m_allocated_capacity );
517  if( __data ) {
518  __str.m_data( __data );
519  __str.m_capacity( __capacity );

193 template<typename U>
194 constexpr int compare( value_ptr<U> const &rhs ) const noexcept {
195 if( !m_value ) {
196 if( !rhs.m_value ) {
197 return 0;
198 }
199 return -1;
200 }
201 if( !rhs.m_value ) {
202 return 1;
203 }
204 if( Compare{ }( *m_value, *rhs.m_value ) ) {
205 return -1;
206 }
207 if( Compare{ }( *rhs.m_value, *m_value ) ) {
208 return 1;
209 }
210 return 0;
211 }

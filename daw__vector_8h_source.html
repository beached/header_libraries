78  static std::size_t const page_size = [] {
79  int res = ::getpagesize( );

81  res = 4096U;

314  static constexpr void overlapped_relocate( pointer source,
315  size_type where_idx,
316  size_type range_size,
317  size_type insert_size ) {
318  assert( where_idx <= range_size );
319  if( insert_size == 0 ) {
320  return;
321  }
322  size_type const reloc_size = range_size - insert_size;
323  auto dfirst =
324  static_cast<difference_type>( ( range_size + insert_size ) - 1U );
325 #if not defined( NDEBUG )
326  auto dlast = static_cast<difference_type>( ( range_size + insert_size ) -
327  reloc_size );
328 #endif
329  auto sfirst = static_cast<difference_type>( range_size - 1U );
330  auto slast = static_cast<difference_type>( range_size - reloc_size );
331 
332  while( sfirst >= slast ) {
333  if constexpr( std::is_nothrow_move_constructible_v<value_type> ) {
334  construct_at( source + dfirst, DAW_MOVE( source[sfirst] ) );
335  } else {
336  construct_at( source + dfirst, source[sfirst] );
337  }
338  --dfirst;
339  --sfirst;
340  }
341  assert( dfirst < dlast );
342  }

344  constexpr void resize_impl( size_type sz ) noexcept {
345  pointer const old_ptr = data( );
346  if( sz <= size( ) ) {
347  if constexpr( std::is_trivially_destructible_v<value_type> ) {
348  m_size = m_first + static_cast<difference_type>( sz );
349  return;
350  } else {
351  if( empty( ) ) {
352  return;
353  }
354  while( m_size > ( m_first + static_cast<difference_type>( sz ) ) ) {
355  std::destroy_at( --m_size );
356  }
357  }
358  } else {
359  auto const old_cap = capacity( );
360  if( sz >= old_cap ) {
361  size_type new_size = calc_new_size( sz );
362  pointer const new_ptr = [&] {
363  if constexpr( has_reallocate<allocator_type>::value ) {
364  return this->reallocate( old_ptr, size( ), new_size );
365  } else {
366  return this->allocate( new_size );
367  }
368  }( );
369  if( new_ptr != old_ptr ) {
370  auto const old_sz = size( );
371  relocate( old_ptr, old_sz, new_ptr );
372  clear( );
373  this->deallocate( m_first, capacity( ) );
374  m_first = new_ptr;
375  m_size = m_first + static_cast<difference_type>( old_sz );
376  }
377  m_capacity = m_first + new_size;
378  }
379  }

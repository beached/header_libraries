78 inline std::size_t round_to_page_size( std::size_t sz ) {
79 static std::size_t const page_size = [] {
80 int res = ::getpagesize( );
81 if( res <= 0 ) {
82 res = 4096U;
83 }
84 return static_cast<std::size_t>( res );
85 }( );
86 auto result =
87 static_cast<std::size_t>( std::round(
88 static_cast<double>( sz ) / static_cast<double>( page_size ) +
89 0.5 ) ) *
90 page_size;
91 return result;
92 }

315 static constexpr void overlapped_relocate( pointer source,
316 size_type where_idx,
317 size_type range_size,
318 size_type insert_size ) {
319 assert( where_idx <= range_size );
320 if( insert_size == 0 ) {
321 return;
322 }
323 size_type const reloc_size = range_size - insert_size;
324 auto dfirst =
325 static_cast<difference_type>( ( range_size + insert_size ) - 1U );
326#if not defined( NDEBUG )
327 auto dlast = static_cast<difference_type>( ( range_size + insert_size ) -
328 reloc_size );
329#endif
330 auto sfirst = static_cast<difference_type>( range_size - 1U );
331 auto slast = static_cast<difference_type>( range_size - reloc_size );
332
333 while( sfirst >= slast ) {
334 if constexpr( std::is_nothrow_move_constructible_v<value_type> ) {
335 construct_at( source + dfirst, std::move( source[sfirst] ) );
336 } else {
337 construct_at( source + dfirst, source[sfirst] );
338 }
339 --dfirst;
340 --sfirst;
341 }
342 assert( dfirst < dlast );
343 }

345 constexpr void resize_impl( size_type sz ) noexcept {
346 pointer const old_ptr = data( );
347 if( sz <= size( ) ) {
348 if constexpr( std::is_trivially_destructible_v<value_type> ) {
349 m_size = m_first + static_cast<difference_type>( sz );
350 return;
351 } else {
352 if( empty( ) ) {
353 return;
354 }
355 while( m_size > ( m_first + static_cast<difference_type>( sz ) ) ) {
356 std::destroy_at( --m_size );
357 }
358 }
359 } else {
360 auto const old_cap = capacity( );
361 if( sz >= old_cap ) {
362 size_type new_size = calc_new_size( sz );
363 pointer const new_ptr = [&] {
364 if constexpr( has_reallocate<allocator_type>::value ) {
365 return this->reallocate( old_ptr, size( ), new_size );
366 } else {
367 return this->allocate( new_size );
368 }
369 }( );
370 if( new_ptr != old_ptr ) {
371 auto const old_sz = size( );
372 relocate( old_ptr, old_sz, new_ptr );
373 clear( );
374 this->deallocate( m_first, capacity( ) );
375 m_first = new_ptr;
376 m_size = m_first + static_cast<difference_type>( old_sz );
377 }
378 m_capacity = m_first + new_size;
379 }
380 }
381 }

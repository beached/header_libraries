29 template<typename queue_type, typename value_type = uint8_t,
30 typename BitQueueLSB = bit_queue_source_native_endian>
31 class basic_bit_queue {
32 static_assert( std::is_unsigned_v<queue_type> and
33 std::is_unsigned_v<value_type>,
34 "Only unsigned integral types are supported" );
35 size_t m_size = 0;
36 queue_type m_queue = 0;
37
38 public:
39 explicit basic_bit_queue( ) = default;
40
41 constexpr explicit basic_bit_queue( queue_type v ) noexcept
42 : m_size( sizeof( m_queue ) * 8 )
43 , m_queue( std::move( v ) ) {}
44
45 [[nodiscard]] constexpr size_t size( ) const noexcept {
46 return m_size;
47 }
48
49 [[nodiscard]] constexpr bool can_pop( size_t const bits ) const noexcept {
50 return m_size >= bits;
51 }
52
53 [[nodiscard]] constexpr bool empty( ) const noexcept {
54 return 0 == m_size;
55 }
56
57 [[nodiscard]] constexpr size_t capacity( ) const noexcept {
58 return sizeof( m_queue ) * 8;
59 }
60
61 [[nodiscard]] constexpr bool full( ) const noexcept {
62 return size( ) == capacity( );
63 }
64
65 private:
66 template<typename T>
67 constexpr auto
68 source_to_native_endian( T value,
69 bit_queue_source_little_endian ) noexcept {
70 return daw::to_native_endian<daw::endian::little>( value );
71 }
72
73 template<typename T>
74 constexpr auto
75 source_to_native_endian( T value, bit_queue_source_big_endian ) noexcept {
76 return daw::to_native_endian<daw::endian::big>( value );
77 }
78
79 public:
80 constexpr void push_back( value_type value,
81 size_t const bits = sizeof( value_type ) * 8 ) {
82 daw::exception::dbg_throw_on_false(
83 ( capacity( ) - m_size ) >= bits,
84 "Not enough bits to hold value pushed" );
85
86 value &= mask_msb<value_type>(
87 sizeof( value_type ) * 8 -
88 bits ); // get_left_mask<value_type>( sizeof( value_type ) * 8 - bits );
89 m_queue <<= bits;
90 m_queue |= source_to_native_endian( value, BitQueueLSB{ } );
91 m_size += bits;
92 }
93
94 [[nodiscard]] constexpr value_type pop_front( size_t const bits ) noexcept {
95 daw::exception::dbg_throw_on_false( m_size >= bits,
96 "Not enough bits to pop request" );
97 auto const mask = mask_msb<queue_type>( capacity( ) - bits );
98 auto result =
99 static_cast<value_type>( static_cast<value_type>( m_queue & mask ) );
100 m_queue >>= bits;
101 m_size -= bits;
102 return result;
103 }
104
105 [[nodiscard]] constexpr value_type pop_back( size_t const bits ) {
106 daw::exception::dbg_throw_on_false( m_size >= bits,
107 "Not enough bits to pop request" );
108 auto mask = queue_type( ( 1 << bits ) - 1 );
109 mask <<= size( ) - bits;
110 auto result = static_cast<value_type>( m_queue & mask );
111 result >>= size( ) - bits;
112 m_queue = m_queue & ~mask;
113 m_size -= bits;
114 return result;
115 }
116
117 constexpr void clear( ) noexcept {
118 m_queue = 0;
119 m_size = 0;
120 }
121
122 constexpr value_type pop_all( ) noexcept {
123 auto result = static_cast<value_type>( m_queue );
124 clear( );
125 return result;
126 }
127
128 [[nodiscard]] constexpr queue_type const &value( ) const noexcept {
129 return m_queue;
130 }
131 }; // basic_bit_queue

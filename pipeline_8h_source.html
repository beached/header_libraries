116 DAW_ATTRIB_FLATTEN constexpr auto pipeline( Fn &&fn, Fns &&...fns ) {
117 // Store all passed functions as they must outlive the call to pipeline
118 return [tpfns = pimpl::make_tpfns<false>(
119 DAW_FWD( fn ), DAW_FWD( fns )... )]( Range auto &&r ) {

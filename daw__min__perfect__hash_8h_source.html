30 template<size_t Bits>
31 constexpr auto determine_opt_sz( ) {
32 constexpr size_t sz = [] {
33 std::pair<uint8_t, size_t> sizes[] = { { 8U, Bits % 8U },
34 { 16U, Bits % 16U },
35 { 32U, Bits % 32U },
36 { 64U, Bits % 64U } };
37 size_t min_pos = 0;
38 daw::algorithm::do_n_arg<3>( [&]( size_t n ) {
39 ++n;
40 if( sizes[n].second < sizes[min_pos].second ) {
41 min_pos = n;
42 } else if( sizes[n].second == sizes[min_pos].second and
43 sizes[n].first > sizes[min_pos].first ) {
44 min_pos = n;
45 }
46 } );
47 return sizes[min_pos].first;
48 }( );
49 if constexpr( sz == 8U ) {
50 return uint8_t{ };
51 } else if constexpr( sz == 16U ) {
52 return uint16_t{ };
53 } else if constexpr( sz == 32U ) {
54 return uint32_t{ };
55 } else if constexpr( sz == 64U ) {
56 return uint64_t{ };
57 }
58 }

75 constexpr void set_bit( size_t idx ) noexcept {
76 auto const bin = idx / m_bits_per_bin;

105 template<typename hash_result>
106 [[nodiscard]] static constexpr hash_result
107 hash_combine( hash_result seed, hash_result hash ) noexcept {
108
109 uintmax_t const s = seed + 1;
110 uintmax_t const h = hash;
111 return static_cast<hash_result>(
112 s ^ ( h + 0x9e3779b9ULL + ( s << 6ULL ) + ( s >> 2ULL ) ) );
113 }

117 // Minimal Perfect Hash Table
118 // Adapted from
119 // https://blog.demofox.org/2015/12/14/o1-data-lookups-with-minimal-perfect-hashing/
120 // and https://blog.gopheracademy.com/advent-2017/mphf/
121 template<size_t N, typename Key, typename Value,
122 typename Hasher = std::hash<Key>,
123 typename KeyEqual = std::equal_to<>>
124 struct perfect_hash_table {
125 static_assert( std::is_default_constructible_v<Key> );
126 static_assert( std::is_default_constructible_v<Value> );
127 static_assert( std::is_trivially_copyable_v<Key> );
128 static_assert( std::is_trivially_copyable_v<Value> );
129 using key_type = Key;
130 using mapped_type = Value;
131 using size_type = std::size_t;
132 using difference_type = std::ptrdiff_t;
133 using reference = std::pair<key_type const &, mapped_type &>;
134 using const_reference = std::pair<key_type const &, mapped_type const &>;
135 using key_equal = KeyEqual;
136 using key_type_pointer = key_type const *;
137 using mapped_type_pointer = mapped_type *;
138 using const_mapped_type_pointer = mapped_type const *;
139 using hash_result = daw::remove_cvref_t<std::invoke_result_t<Hasher, Key>>;
140 using salt_type = intmax_t;
141 static size_t constexpr m_data_size = mph_impl::next_pow2<N>( );
142 static_assert( m_data_size <= static_cast<size_t>( max_value<salt_type> ) );
143
144 /***
145 * Construct a perfect_hash_table from a range of pair like items that have
146 * the key in their first member and the value in their second member
147 *
148 * @tparam ForwardIterator points to a range of items with a std::pair like
149 * type
150 * @param first Start of range, inclusive
151 * @param last End of range, exclusive
152 */
153 template<typename ForwardIterator>
154 constexpr perfect_hash_table( ForwardIterator first,
155 ForwardIterator last ) {
156 daw::array<bucket_t<ForwardIterator>, m_num_buckets> buckets{ };
157
158 static_assert( std::is_default_constructible_v<ForwardIterator> );
159
160 daw::algorithm::iota( buckets.begin( ), buckets.end( ),
161 static_cast<size_type>( 0 ),
162 []( auto &i, size_type num ) {
163 i.bucket_index = num;
164 } );
165
166 for( auto it = first; it != last; ++it ) {
167 auto const bucket = scale_hash<m_num_buckets>( call_hash( it->first ) );
168 buckets[bucket].items.push_back( it );
169 }
170
171 daw::sort( buckets.begin( ), buckets.end( ),
172 []( auto const &lhs, auto const &rhs ) {
173 return lhs.items.size( ) > rhs.items.size( );
174 } );
175
176 daw::array<bool, m_data_size> slots_claimed{ };
177
178 daw::algorithm::find_if( buckets.cbegin( ), buckets.cend( ),
179 [&]( auto const &bucket ) {
180 if( bucket.items.empty( ) ) {
181 return true;
182 }
183 find_salt_for_bucket( bucket, slots_claimed );
184 return false;
185 } );
186 }
187
188 explicit constexpr perfect_hash_table(
189 std::pair<Key, Value> const ( &data )[N] )
190 : perfect_hash_table( data, data + static_cast<ptrdiff_t>( N ) ) {}

193 find( Key const &key ) const {

251 scale_hash( hash_result hash ) noexcept {
252 return static_cast<hash_result>( hash % Space );

255 [[nodiscard]] static constexpr hash_result call_hash( Key const &key ) {

265 template<typename ForwardIterator>
266 constexpr void
267 find_salt_for_bucket( bucket_t<ForwardIterator> const &bucket,
268 daw::array<bool, m_data_size> &slots_claimed ) {
269 if( bucket.items.size( ) == 1 ) {
270 auto const pos = daw::algorithm::find_index_of(
271 slots_claimed.cbegin( ), slots_claimed.cend( ), false );
272 if( pos == slots_claimed.size( ) ) {
273 // Should never happen, there should always be at least one unclaimed
274 // slot
275 std::abort( );
276 }
277 slots_claimed[pos] = true;
278 m_salts[bucket.bucket_index] = -( static_cast<salt_type>( pos ) + 1 );
279 m_keys[pos] = bucket.items[0]->first;
280 m_values[pos] = bucket.items[0]->second;
281 }
282
283 for( salt_type salt = 1;; ++salt ) {
284 daw::bounded_vector_t<hash_result, m_data_size> slots_this_bucket{ };
285
286 auto const success = daw::algorithm::all_of(
287 bucket.items.begin( ), bucket.items.end( ),
288 [&]( auto const *const item ) -> bool {
289 hash_result const slot_wanted =
290 scale_hash( call_hash( item->first, salt ) );
291 if( slots_claimed[slot_wanted] or
292 ( daw::algorithm::find(
293 slots_this_bucket.begin( ), slots_this_bucket.end( ),
294 slot_wanted ) != slots_this_bucket.end( ) ) ) {
295 return false;
296 }
297 slots_this_bucket.push_back( slot_wanted );
298 return true;
299 } );
300
301 if( success ) {
302 m_salts[bucket.bucket_index] = salt;
303 for( size_type i = 0; i < bucket.items.size( ); ++i ) {
304 auto const pos = static_cast<size_t>( slots_this_bucket[i] );
305 m_keys[pos] = bucket.items[i]->first;
306 m_values[pos] = bucket.items[i]->second;
307 slots_claimed[pos] = true;
308 }
309 return;
310 }
311 }
312 }

314 [[nodiscard]] constexpr size_type find_data_index( Key const &key ) const {
315 size_type const hash = call_hash( key );
316 auto const salt = m_salts[scale_hash<m_num_buckets>( hash )];
317
318 if( salt < 0 ) {
319 return static_cast<size_type>( ( -salt ) - 1 );
320 }
321 return scale_hash( mph_impl::hash_combine<hash_result>(
322 static_cast<hash_result>( salt ), hash ) );
323 }

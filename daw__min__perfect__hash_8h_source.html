28  template<size_t Bits>
29  constexpr auto determine_opt_sz( ) {
30  constexpr size_t sz = [] {
31  std::pair<uint8_t, size_t> sizes[] = { { 8U, Bits % 8U },
32  { 16U, Bits % 16U },
33  { 32U, Bits % 32U },
34  { 64U, Bits % 64U } };
35  size_t min_pos = 0;
36  daw::algorithm::do_n_arg<3>( [&]( size_t n ) {
37  ++n;
38  if( sizes[n].second < sizes[min_pos].second ) {
39  min_pos = n;
40  } else if( sizes[n].second == sizes[min_pos].second and
41  sizes[n].first > sizes[min_pos].first ) {
42  min_pos = n;
43  }
44  } );
45  return sizes[min_pos].first;
46  }( );
47  if constexpr( sz == 8U ) {
48  return uint8_t{ };
49  } else if constexpr( sz == 16U ) {
50  return uint16_t{ };
51  } else if constexpr( sz == 32U ) {
52  return uint32_t{ };
53  } else if constexpr( sz == 64U ) {
54  return uint64_t{ };
55  }
56  }

106  hash_combine( hash_result seed, hash_result hash ) noexcept {
107 
108  uintmax_t const s = seed + 1;
109  uintmax_t const h = hash;

116  // Minimal Perfect Hash Table
117  // Adapted from
118  // https://blog.demofox.org/2015/12/14/o1-data-lookups-with-minimal-perfect-hashing/
119  // and https://blog.gopheracademy.com/advent-2017/mphf/
120  template<size_t N, typename Key, typename Value,
121  typename Hasher = std::hash<Key>,
122  typename KeyEqual = std::equal_to<>>
123  struct perfect_hash_table {
124  static_assert( std::is_default_constructible_v<Key> );
125  static_assert( std::is_default_constructible_v<Value> );
126  static_assert( std::is_trivially_copyable_v<Key> );
127  static_assert( std::is_trivially_copyable_v<Value> );
128  using key_type = Key;
129  using mapped_type = Value;
130  using size_type = std::size_t;
131  using difference_type = std::ptrdiff_t;
132  using reference = std::pair<key_type const &, mapped_type &>;
133  using const_reference = std::pair<key_type const &, mapped_type const &>;
134  using key_equal = KeyEqual;
135  using key_type_pointer = key_type const *;
136  using mapped_type_pointer = mapped_type *;
137  using const_mapped_type_pointer = mapped_type const *;
138  using hash_result = daw::remove_cvref_t<std::invoke_result_t<Hasher, Key>>;
139  using salt_type = intmax_t;
140  static size_t constexpr m_data_size = mph_impl::next_pow2<N>( );
141  static_assert(
142  m_data_size <=
143  static_cast<size_t>( ( std::numeric_limits<salt_type>::max )( ) ) );
144 
145  /***
146  * Construct a perfect_hash_table from a range of pair like items that have
147  * the key in their first member and the value in their second member
148  *
149  * @tparam ForwardIterator points to a range of items with a std::pair like
150  * type
151  * @param first Start of range, inclusive
152  * @param last End of range, exclusive
153  */
154  template<typename ForwardIterator>
155  constexpr perfect_hash_table( ForwardIterator first,
156  ForwardIterator last ) {
157  daw::array<bucket_t<ForwardIterator>, m_num_buckets> buckets{ };
158 
159  static_assert( std::is_default_constructible_v<ForwardIterator> );
160 
161  daw::algorithm::iota( buckets.begin( ), buckets.end( ),
162  static_cast<size_type>( 0 ),
163  []( auto &i, size_type num ) {
164  i.bucket_index = num;
165  } );
166 
167  for( auto it = first; it != last; ++it ) {
168  auto const bucket = scale_hash<m_num_buckets>( call_hash( it->first ) );
169  buckets[bucket].items.push_back( it );
170  }
171 
172  daw::sort( buckets.begin( ), buckets.end( ),
173  []( auto const &lhs, auto const &rhs ) {
174  return lhs.items.size( ) > rhs.items.size( );
175  } );
176 
177  daw::array<bool, m_data_size> slots_claimed{ };
178 
179  daw::algorithm::find_if( buckets.cbegin( ), buckets.cend( ),
180  [&]( auto const &bucket ) {
181  if( bucket.items.empty( ) ) {
182  return true;
183  }
184  find_salt_for_bucket( bucket, slots_claimed );
185  return false;
186  } );
187  }
188

266  template<typename ForwardIterator>
267  constexpr void
268  find_salt_for_bucket( bucket_t<ForwardIterator> const &bucket,
269  daw::array<bool, m_data_size> &slots_claimed ) {
270  if( bucket.items.size( ) == 1 ) {
271  auto const pos = daw::algorithm::find_index_of(
272  slots_claimed.cbegin( ), slots_claimed.cend( ), false );
273  if( pos == slots_claimed.size( ) ) {
274  // Should never happen, there should always be at least one unclaimed
275  // slot
276  std::abort( );
277  }
278  slots_claimed[pos] = true;
279  m_salts[bucket.bucket_index] = -( static_cast<salt_type>( pos ) + 1 );
280  m_keys[pos] = bucket.items[0]->first;
281  m_values[pos] = bucket.items[0]->second;
282  }
283 
284  for( salt_type salt = 1;; ++salt ) {
285  daw::bounded_vector_t<hash_result, m_data_size> slots_this_bucket{ };
286 
287  auto const success = daw::algorithm::all_of(
288  bucket.items.begin( ), bucket.items.end( ),
289  [&]( auto const *const item ) -> bool {
290  hash_result const slot_wanted =
291  scale_hash( call_hash( item->first, salt ) );
292  if( slots_claimed[slot_wanted] or
293  ( daw::algorithm::find(
294  slots_this_bucket.begin( ), slots_this_bucket.end( ),
295  slot_wanted ) != slots_this_bucket.end( ) ) ) {
296  return false;
297  }
298  slots_this_bucket.push_back( slot_wanted );
299  return true;
300  } );
301 
302  if( success ) {
303  m_salts[bucket.bucket_index] = salt;
304  for( size_type i = 0; i < bucket.items.size( ); ++i ) {
305  auto const pos = static_cast<size_t>( slots_this_bucket[i] );
306  m_keys[pos] = bucket.items[i]->first;
307  m_values[pos] = bucket.items[i]->second;
308  slots_claimed[pos] = true;
309  }
310  return;
311  }
312  }
313  }

315  [[nodiscard]] constexpr size_type find_data_index( Key const &key ) const {
316  size_type const hash = call_hash( key );
317  auto const salt = m_salts[scale_hash<m_num_buckets>( hash )];
318 
319  if( salt < 0 ) {
320  return static_cast<size_type>( ( -salt ) - 1 );
321  }
322  return scale_hash( mph_impl::hash_combine<hash_result>(
323  static_cast<hash_result>( salt ), hash ) );
324  }

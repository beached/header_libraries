356 template<typename... Args>
357 node_id_t add_node( Args &&...args ) {
358 auto const id = cur_id;
359 ++cur_id;
360 m_nodes.emplace( std::make_pair(
361 id, graph_impl::node_impl_t<T>(
362 node_id_t{ id }, daw::construct_a<T>( DAW_FWD( args )... ) ) ) );
363
364 return node_id_t{ id };
365 }

368 return m_nodes.count( id.value( ) ) > 0;

405 void add_directed_edge( node_id_t from, node_id_t to ) {
406 daw::exception::dbg_precondition_check( has_node( from ) );
407 daw::exception::dbg_precondition_check( has_node( to ) );

429 void remove_node( node_id_t id ) {
430 daw::exception::dbg_precondition_check( has_node( id ) );
431
432 auto &node = get_raw_node( id );
433 for( auto n_id : node.incoming_edges( ) ) {
434 get_node( n_id ).outgoing_edges( ).erase( id );
435 }
436 for( auto n_id : node.outgoing_edges( ) ) {
437 get_node( n_id ).incoming_edges( ).erase( id );
438 }
439 m_nodes.erase( id.value( ) );
440 }

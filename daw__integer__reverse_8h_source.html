20 [[nodiscard]] DAW_ATTRIB_INLINE constexpr std::uint8_t
21 reverse_bits( std::uint8_t value ) {
22#if not DAW_HAS_CLANG_VER_LT( 10, 0 ) and __has_builtin( __builtin_bitreverse8 )
23 return __builtin_bitreverse8( value );
24#else
25 auto x = static_cast<unsigned>( value );
26 x = ( ( x & 0x1 ) << 7 ) | ( ( x & 0x2 ) << 5 ) | ( ( x & 0x4 ) << 3 ) |
27 ( ( x & 0x8 ) << 1 ) | ( ( x & 0x10 ) >> 1 ) | ( ( x & 0x20 ) >> 3 ) |
28 ( ( x & 0x40 ) >> 5 ) | ( ( x & 0x80 ) >> 7 );
29 return static_cast<std::uint8_t>( x );
30#endif
31 }

33 [[nodiscard]] DAW_ATTRIB_INLINE constexpr std::uint16_t

37 return __builtin_bitreverse16( value );

39 auto a = static_cast<unsigned>( value );

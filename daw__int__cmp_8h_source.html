22 template<typename T, typename U,
23 std::enable_if_t<daw::is_integral_v<T> and daw::is_integral_v<U>,
24 std::nullptr_t> = nullptr>
25 DAW_ATTRIB_INLINE constexpr bool cmp_equal( T t, U u ) noexcept {
26 using UT = daw::make_unsigned_t<T>;
27 using UU = daw::make_unsigned_t<U>;
28 if constexpr( daw::is_signed_v<T> == daw::is_signed_v<U> ) {
29 return t == u;
30 } else if constexpr( daw::is_signed_v<T> ) {
31 return t < 0 ? false : static_cast<UT>( t ) == u;
32 } else {
33 return u < 0 ? false : t == static_cast<UU>( u );
34 }
35 }

60 constexpr difference_type get_offset( difference_type n ) const noexcept {
61 auto const sz =
62 static_cast<difference_type>( impl::container_size( *m_container ) );
63 n += m_position;
64 if( n < 0 ) {
65 n *= -1;
66 n %= sz;
67 return sz - n;
68 } else if( n >= sz ) {
69 return n % sz;
70 }
71 return n;
72 }

74 constexpr decltype( auto ) get_iterator( difference_type n ) const {
75 auto result = std::next( std::begin( *m_container ), n );
76 return result;

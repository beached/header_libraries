25 template<std::size_t C, typename Needle, typename Projection,
26 typename... ProjectedValues>
27 [[nodiscard]] DAW_ATTRIB_INLINE constexpr auto
28 finder( Needle const &needle, Projection const &projection,
29 ForwardIterator auto it_current,
30 ProjectedValues &&...projected_values )
31 requires( not InputIterator<Needle> and not InputIterator<Projection> ) //
32 {
33
34 static_assert( C >= 0 );
35 if constexpr( C == 0 ) {
36 static_assert( std::invocable<Needle, ProjectedValues...> );
37 return std::invoke( needle, DAW_FWD( projected_values )... );
38 } else {
39 auto next = it_current;
40 static_assert( std::invocable<Projection, decltype( *next )>,
41 "Could not project iterators value_type" );
42 ++it_current;
43 return finder<C - 1>( needle, projection, std::move( it_current ),
44 DAW_FWD( projected_values )...,
45 std::invoke( projection, *std::move( next ) ) );
46 }
47 }

68 template<std::size_t FindWidth, typename Needle,
69 typename Projection = std::identity>
70 requires( FindWidth > 0 ) //
71 struct FindIf_t {
72 DAW_NO_UNIQUE_ADDRESS Needle needle;
73 DAW_NO_UNIQUE_ADDRESS Projection projection;
74
75 template<ForwardRange R>
76 [[nodiscard]] constexpr auto operator( )( R &&r ) const {
77 auto first = std::begin( r );
78 auto const last = std::end( r );
79 while( first != last ) {
80 ForwardIterator auto p0 = first;
81 Iterator auto plast = daw::safe_next( first, last, FindWidth );
82 if( static_cast<std::size_t>( std::distance( p0, plast ) ) <
83 FindWidth ) {
84 return last;
85 }
86 if( finder<FindWidth>( needle, projection, p0 ) ) {
87 return first;
88 }
89 ++first;
90 }
91 return first;
92 }
93 };

49 #if not defined( __cpp_lib_ranges )

52  } else if constexpr( requires {
53  { l - f } -> std::convertible_to<difference_type>;
54  } ) {
55  return static_cast<size_type>( l - f );
56  } else {
57  size_type count = 0;
58  while( f != l ) {
59  ++count;
60  std::advance( f, 1 );
61  }
62  return count;
63  }
64 #if __cpp_lib_ranges < 201911L

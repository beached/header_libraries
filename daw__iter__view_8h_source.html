60 static constexpr size_type distance( iterator f, last_iterator l ) {
61#if not defined( __cpp_lib_ranges )
62 if constexpr( std::is_same_v<iterator, last_iterator> ) {
63 return static_cast<size_type>( std::distance( f, l ) );
64 } else if constexpr( requires {
65 { l - f } -> std::convertible_to<difference_type>;
66 } ) {
67 return static_cast<size_type>( l - f );
68 } else {
69 size_type count = 0;
70 while( f != l ) {
71 ++count;
72 std::advance( f, 1 );
73 }
74 return count;
75 }
76#if __cpp_lib_ranges < 201911L
77#else
78 return static_cast<size_type>( std::ranges::distance( f, l ) );
79#endif
80#else
81 return static_cast<size_type>( std::ranges::distance( f, l ) );
82#endif
83 }

66 public:
67 using values_type = std::vector<Chunk>;
68 using value_type = typename values_type::value_type;
69 using reference = typename values_type::reference;

74 private:
75 // Must be mutable as we hide our size and expand the values < size to fit
76 // as needed
77 values_type mutable m_items;
78 size_t m_size;
79
80 [[nodiscard]] auto lfind( size_t pos ) {
81 if( pos >= size( ) ) {
82 return size( );
83 }
84 auto item_pos = daw::algorithm::find_last_of(
85 m_items.begin( ), m_items.end( ), [pos]( auto const &item ) {
86 return pos >= item.start( );
87 } );
88 if( item_pos < size( ) ) {
89 // We are within the max size of the container. Feel free to add it
90 auto offset = pos - item_pos->start( );
91 if( offset > item_pos->size( ) + 1 ) {
92 ++item_pos;
93 item_pos = m_items.emplace( item_pos );
94 item_pos->start( ) = pos;
95 item_pos->items( ).emplace_back( );
96 }
97 }
98 return item_pos;
99 }
100
101 public:
102 [[nodiscard]] size_t size( ) const {
103 return m_size;
104 }
105
106 [[nodiscard]] reference operator[]( size_t pos ) {
107 auto item = lfind( pos );
108 }

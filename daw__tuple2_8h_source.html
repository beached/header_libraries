30  template<std::size_t... Sizes, std::size_t... Is>
31  constexpr auto cartesian_product_impl2( std::index_sequence<Sizes...>,
32  std::index_sequence<Is...> ) {
33  static_assert( sizeof...( Sizes ) != 0 );
34  constexpr std::array x = [] {
35  constexpr std::size_t orig[]{ Sizes... };
36  std::array<std::size_t, sizeof...( Is )> result{ };
37  std::size_t idx = 0;
38  for( std::size_t n = 0; n < sizeof...( Sizes ); ++n ) {
39  for( std::size_t m = 0; m < orig[n]; ++m ) {
40  result[idx] = encode_location( n, m );
41  ++idx;
42  }
43  }
44  return result;
45  }( );
46  return std::index_sequence<x[Is]...>{ };
47  }

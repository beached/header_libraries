24 /***
25 * @brief Given types From and To and an expression E whose type and value
26 * category are the same as those of std::declval<From>(),
27 * convertible_to<From, To> requires E to be both implicitly and explicitly
28 * convertible to type To. The implicit and explicit conversions are required
29 * to produce equal results.
30 */
31 template<typename From, typename To>
32 concept convertible_to = std::is_convertible_v<From, To> and requires {
33 { static_cast<To>( std::declval<From>( ) ) };
34 };

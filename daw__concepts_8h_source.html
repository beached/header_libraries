23  /***
24  * @brief Given types From and To and an expression E whose type and value
25  * category are the same as those of std::declval<From>(),
26  * convertible_to<From, To> requires E to be both implicitly and explicitly
27  * convertible to type To. The implicit and explicit conversions are required
28  * to produce equal results.
29  */
30  template<typename From, typename To>
31  concept convertible_to = std::is_convertible_v<From, To> and
32  requires {
33  { static_cast<To>( std::declval<From>( ) ) };
34  };

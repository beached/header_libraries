31 struct Argument {
32 daw::string_view name{ };
33 daw::string_view value{ };

35 private:

38 constexpr Argument( daw::string_view args ) {
39 if( args.starts_with( "--" ) ) {
40 // We are a named argument, starting with '--'
41 args.remove_prefix( 2 );
42 name = args.pop_front_until( '=' );
43 if( args.empty( ) ) {
44 return;
45 }
46 }
47 // We have an assigned value
48 if( args.starts_with( '"' ) and args.ends_with( '"' ) ) {
49 args.remove_prefix( 1 );
50 args.remove_suffix( 1 );
51 }
52 value = args;
53 }
54 };

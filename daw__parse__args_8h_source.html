43  constexpr Argument( daw::string_view args ) {
44  if( args.size( ) >= 2 and nsc_and( args[0] == '-', args[1] == '-' ) ) {
45  // We are a named argument, starting with '--'
46  args.remove_prefix( 2 );

48  if( args.empty( ) ) {

392 template<typename... Args>

511 template<typename Predicate, typename Visitor>
512 constexpr void visit( Predicate &&pred, Visitor &&vis ) {
513 static_assert( std::is_invocable_v<Predicate, raw_node_t const &>,
514 "Predicate must accept a node as argument" );
515 static_assert( std::is_invocable_v<Visitor, raw_node_t &>,
516 "Visitor must accept a node as argument" );
517 for( auto &node_ref : m_nodes ) {
518 auto node = get_node( node_ref.value.id( ) );
519 if( daw::invoke( pred, node ) ) {
520 daw::invoke( vis, std::move( node ) );
521 }
522 }

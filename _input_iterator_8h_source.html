34 template<typename It>
35 static decltype( auto ) as_orig( std::byte *p ) {
36 return *reinterpret_cast<It *>( p );
37 }

39 using op_del_t = fn_t<void( std::byte * )>;
40 using op_cmp_t = fn_t<bool( std::byte *, std::byte * )>;
41 using op_inc_t = fn_t<void( std::byte * )>;
42 template<typename Value>
43 using op_arrow_t = fn_t<Value const *( std::byte * )>;
44 using op_clone_t = fn_t<std::byte *( std::byte * )>;

46 enum fn_ary_item { OpDel, OpEqual, OpNotEqual, OpInc, OpArrow, OpClone };
47
48 template<typename Value>
49 using vtable_t = std::tuple<op_del_t,
50 op_cmp_t,
51 op_cmp_t,
52 op_inc_t,
53 op_arrow_t<Value>,
54 op_clone_t>;
55
56 template<fn_ary_item Idx, typename Value>
57 decltype( auto ) get( vtable_t<Value> const *tbl ) {
58 return std::get<Idx>( *tbl );
59 }

61 template<typename Iterator, typename Value>
62 static vtable_t<Value> const *get_vtable( ) noexcept {
63 static constexpr vtable_t<Value> vtable{
64 +[]( std::byte *p ) -> void {
65 // op_del
66 delete reinterpret_cast<Iterator *>( p );
67 },
68 +[]( std::byte *l, std::byte *r ) -> bool {
69 // op_equal
70 return as_orig<Iterator>( l ) == as_orig<Iterator>( r );
71 },
72 +[]( std::byte *l, std::byte *r ) -> bool {
73 // op_not_equal
74 return as_orig<Iterator>( l ) != as_orig<Iterator>( r );
75 },
76 +[]( std::byte *p ) -> void {
77 // op_inc
78 ++as_orig<Iterator>( p );
79 },
80 +[]( std::byte *p ) -> Value const * {
81 // op_arrow
82 if constexpr( std::is_pointer_v<Iterator> ) {
83 return as_orig<Iterator>( p );
84 } else {
85 return as_orig<Iterator>( p ).operator->( );
86 }
87 },
88 +[]( std::byte *p ) -> std::byte * {
89 // op_clone
90 return reinterpret_cast<std::byte *>(
91 new Iterator( std::as_const( as_orig<Iterator>( p ) ) ) );
92 } };
93 return &vtable;
94 }

120 public:
121 template<typename Iterator,
122 std::enable_if_t<not std::is_same_v<InputIterator, Iterator>,
123 std::nullptr_t> = nullptr>
124 explicit InputIterator( Iterator const &it )
125 : m_iterator( create( it ) )
126 , m_vtable( input_impl::get_vtable<Iterator, value_type>( ) ) {
127 static_assert(
128 std::is_same_v<std::remove_cv_t<value_type>,
129 typename std::iterator_traits<Iterator>::value_type> );

132 InputIterator( InputIterator const &other )
133 : m_iterator( other.fn_op_clone( )( other.m_iterator ) )
134 , m_vtable( other.m_vtable ) {}
135
136 InputIterator &operator=( InputIterator const &rhs ) {
137 if( this != &rhs ) {
138 // copy first so that if it throws we are cool
139 auto tmp = rhs.fn_op_clone( )( rhs.m_iterator );
140 std::swap( m_iterator, tmp );
141 m_vtable = rhs.m_vtable;
142 input_impl::get<input_impl::OpDel, value_type>( m_vtable )( tmp );
143 }
144 return *this;
145 }

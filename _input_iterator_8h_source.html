60  static vtable_t<Value> const *get_vtable( ) noexcept {
61  static constexpr vtable_t<Value> vtable{
62  +[]( std::byte *p ) -> void {
63  // op_del
64  delete reinterpret_cast<Iterator *>( p );
65  },
66  +[]( std::byte *l, std::byte *r ) -> bool {

69  },

73  },

134 template<typename Fn, typename Projection = std::identity>
135 struct filter_t {
136 DAW_NO_UNIQUE_ADDRESS mutable Fn m_func;
137 DAW_NO_UNIQUE_ADDRESS Projection m_projection{ };
138
139 template<Range R>
140 [[nodiscard]] constexpr auto operator( )( R &&r ) const {
141 static_assert( std::invocable<Projection, range_value_t<R>>,
142 "Projection must be invocable with range_value_t<R>" );
143 using projected_t = std::invoke_result_t<Projection, range_value_t<R>>;
144 static_assert(
145 std::invocable<Fn, projected_t>,
146 "Filter requires an invokable function with the projected value" );
147 static_assert(
148 std::convertible_to<std::invoke_result_t<Fn, projected_t>, bool>,
149 "Filter requires an invokable function that returns a bool" );
150 return filter_view<iterator_t<R>, Fn>( std::begin( r ), std::end( r ),
151 m_func, m_projection );
152 }
153
154 template<typename Value>
155 [[nodiscard]] constexpr auto operator( )( Value &&v ) const {
156 static_assert( std::invocable<Projection, Value>,
157 "Projection must be invocable with range_value_t<R>" );
158 using projected_t = std::invoke_result_t<Projection, Value>;
159 static_assert(
160 std::is_invocable_r_v<bool, Fn, projected_t>,
161 "Filter requires an invokable function that returns a bool" );
162 using result_t = daw::remove_cvref_t<std::invoke_result_t<Fn, Value>>;
163 if( std::invoke( m_func, std::invoke( m_projection, v ) ) ) {
164 return std::optional<result_t>( std::invoke( m_func, DAW_FWD( v ) ) );
165 }
166 return std::optional<result_t>( );
167 }
168 };

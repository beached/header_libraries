55 template<typename Value, size_t N, size_t HashSize = sizeof( size_t )>
56 struct fixed_lookup {
57 static_assert( std::is_default_constructible_v<Value>,
58 "Value must be default constructible" );
59 static_assert( N > 0, "Must supply a positive initial_size larger than 0" );
60
61 using value_t = daw::traits::root_type_t<Value>;
62 using hash_value_t = typename daw::generic_hash_t<HashSize>::hash_value_t;
63 static_assert(
64 N <= max_value<hash_value_t>,
65 "Cannot allocate more values than can be addressed by hash value" );
66
67 using reference = value_t &;
68 using const_reference = value_t const &;
69
70 private:
71 std::array<hash_value_t, N> m_hashes;
72 std::array<value_t, N> m_values;
73
74 public:
75 static constexpr hash_value_t capacity( ) noexcept {
76 return N;
77 }
78
79 constexpr hash_value_t size( ) const noexcept {
80 hash_value_t count = 0;
81 for( auto hash : m_hashes ) {
82 count +=
83 hash >= impl::fixed_lookup_sentinals::fixed_lookup_sentinals_size ? 1
84 : 0;
85 }
86 return count;
87 }
88
89 private:
90 template<typename KeyType>
91 static constexpr hash_value_t hash_fn( KeyType &&key ) noexcept {
92 auto const hash = daw::generic_hash<HashSize>( DAW_FWD( key ) );
93 auto const divisor =
94 max_value<hash_value_t> -
95 impl::fixed_lookup_sentinals::fixed_lookup_sentinals_size;
96 return ( hash % divisor ) +
97 impl::fixed_lookup_sentinals::fixed_lookup_sentinals_size;
98 }
99
100 constexpr hash_value_t scale_hash( hash_value_t const hash ) const {
101 // Scale value to capacity using MAD(Multiply-Add-Divide) compression
102 // Use the two largest Prime's that fit in a hash_value_t
103 return ( ( hash * impl::largest_primes<HashSize>::a ) +
104 impl::largest_primes<HashSize>::b ) %
105 capacity( );
106 }
107
108 constexpr auto lookup( hash_value_t const hash ) const noexcept {
109 struct lookup_result_t {
110 hash_value_t position;
111 bool found;
112
113 constexpr lookup_result_t( hash_value_t pos, bool is_found ) noexcept
114 : position{ pos }
115 , found{ is_found } {}
116
117 constexpr operator bool( ) const noexcept {
118 return found;
119 }
120 };
121
122 auto const hash_pos = scale_hash( hash );
123 for( hash_value_t n = hash_pos; n != m_hashes.size( ); ++n ) {
124 if( hash == m_hashes[n] ) {
125 return lookup_result_t{ n, true };
126 } else if( impl::fixed_lookup_sentinals::empty == m_hashes[n] ) {
127 return lookup_result_t{ n, false };
128 }
129 }
130 for( hash_value_t n = 0; n != hash_pos; ++n ) {
131 if( m_hashes[n] == hash ) {
132 return lookup_result_t{ n, true };
133 } else if( m_hashes[n] == impl::fixed_lookup_sentinals::empty ) {
134 return lookup_result_t{ n, false };
135 }
136 }
137 return lookup_result_t{ m_hashes.size( ), false };
138 }

141 constexpr fixed_lookup( ) noexcept(
142 noexcept( std::is_nothrow_default_constructible_v<value_t> ) ) = default;

144 template<typename Key>
145 constexpr hash_value_t find_existing( Key &&key ) const {
146 auto const hash = hash_fn( DAW_FWD( key ) );
147 auto const is_found = lookup( hash );
148 daw::exception::daw_throw_on_false(
149 is_found, "Attempt to access an undefined key" );
150 return is_found.position;
151 }

153 template<typename Key>
154 constexpr hash_value_t insert( Key &&key, Value value ) {
155 auto const hash = hash_fn( DAW_FWD( key ) );
156 auto const is_found = lookup( hash );
157 daw::exception::daw_throw_on_true(
158 !is_found and is_found.position == m_hashes.size( ),
159 "Fixed hash table does not have enough space to allocate all entries" );
160 m_hashes[is_found.position] = hash;
161 m_values[is_found.position] = std::move( value );
162 return is_found.position;
163 }

165 constexpr reference get_existing( hash_value_t position ) {
166 return m_values[position];
167 }

169 constexpr const_reference get_existing( hash_value_t position ) const {
170 return m_values[position];
171 }

173 template<typename Key>
174 constexpr const_reference operator[]( Key &&key ) const {
175 return m_values[find_existing( DAW_FWD( key ) )];
176 }

178 template<typename Key>
179 constexpr reference operator[]( Key &&key ) {
180 auto const hash = hash_fn( DAW_FWD( key ) );
181 auto const is_found = lookup( hash );
182 daw::exception::daw_throw_on_true(
183 !is_found and is_found.position == m_hashes.size( ),
184 "Fixed hash table does not have enough space to allocate all entries" );
185 m_hashes[is_found.position] = hash;

191 auto const hash = hash_fn( DAW_FWD( key ) );

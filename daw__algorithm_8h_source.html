88 template<typename Iterator, typename Iterator2, typename Iterator3,
89 typename Distance,
90 daw::enable_when_t<
91 not traits::is_random_access_iterator<Iterator>> = nullptr>
92 constexpr void safe_advance_impl( Iterator2 const first, Iterator &it,
93 Iterator3 const last,
94 Distance dist ) noexcept {
95
96 // Move it forward towards last
97 for( ; it != last and dist > 0; --dist, ++it ) {}
98
99 // Move it backwards towards first
100 for( ; it != first and dist < 0; ++dist, --it ) {}
101 }
102
103 template<
104 typename Iterator, typename Iterator2, typename Iterator3,
105 typename Distance,
106 daw::enable_when_t<traits::is_random_access_iterator<Iterator>> = nullptr>
107 constexpr void safe_advance_impl( Iterator2 const first, Iterator &it,
108 Iterator3 const last,
109 Distance dist ) noexcept {
110
111 if( dist < 0 ) {
112 auto const dist_to_first = it - first;
113 if( dist_to_first < 0 ) {
114 it = first;
115 return;
116 }
117 dist = -dist;
118 it -= (daw::min)( static_cast<ptrdiff_t>( dist_to_first ),
119 static_cast<ptrdiff_t>( dist ) );
120 return;
121 }
122 auto const dist_to_last = last - it;
123 if( dist_to_last < 0 ) {
124 it = last;
125 return;
126 }
127 it += (daw::min)( static_cast<ptrdiff_t>( dist_to_last ),
128 static_cast<ptrdiff_t>( dist ) );
129 }

288 template<typename ForwardIterator, typename Value,
289 typename Compare = std::less<>>
290 constexpr ForwardIterator lower_bound( ForwardIterator first,
291 ForwardIterator last, Value &&value,
292 Compare cmp = { } ) {
293 // Precondition checks
294 traits::is_forward_access_iterator_test<ForwardIterator>( );
295 traits::is_input_iterator_test<ForwardIterator>( );
296 traits::is_predicate_test<Compare, decltype( *first ), Value>( );

298 while( first != last ) {
299 auto mid = std::next( first, std::distance( first, last ) / 2 );

301 first = std::next( mid );
302 } else {
303 last = mid;
304 }
305 }
306 return first;

338 template<typename ForwardIterator, typename Value,
339 typename Predicate = std::less<>>
340 constexpr ForwardIterator
341 binary_search( ForwardIterator first, ForwardIterator const last,
342 Value &&value, Predicate less_than = Predicate{ } ) {
343
344 traits::is_forward_access_iterator_test<ForwardIterator>( );
345 traits::is_input_iterator_test<ForwardIterator>( );
346 traits::is_predicate_test<Predicate, decltype( *first ), Value>( );
347
348 exception::precondition_check(
349 first < last, ": First position must be less than second" );
350
351 auto it_last = last;
352
353 while( first < it_last ) {
354 auto const mid = algorithm_details::midpoint( first, it_last );
355 if( less_than( *mid, value ) ) {
356 first = mid;
357 daw::advance( first, 1 );
358 } else if( less_than( value, *mid ) ) {
359 it_last = mid;
360 } else { // equal
361 return mid;
362 }
363 }
364 return last;
365 }

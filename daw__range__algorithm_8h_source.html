351 template<typename Container, typename UnaryPredicate>
352 auto where( Container &container, UnaryPredicate predicate ) {
353 using value_type = typename std::iterator_traits<decltype( std::begin(
354 container ) )>::value_type;
355 std::vector<std::reference_wrapper<value_type>> result;
356 for( auto &v : container ) {
357 if( predicate( v ) ) {
358 result.push_back( std::ref( v ) );
359 }
360 }
361 return result;
362 }

302  static constexpr size_type scale_hash( size_type hash,

304  size_type const prime_a = 18446744073709551557u;

309  constexpr std::optional<size_type>
310  find_index( size_type hash, Key const &key ) const noexcept {
311  size_type const scaled_hash = scale_hash( hash, m_data.size( ) );
312 
313  for( size_type n = scaled_hash; n < m_data.size( ); ++n ) {
314  if( !m_data[n].has_value ) {
315  return n;
316  }
317  if( key_equal{ }( m_data[n].key, key ) ) {
318  return n;
319  }
320  }
321  for( size_type n = 0; n < scaled_hash; ++n ) {
322  if( !m_data[n].has_value ) {
323  return n;
324  }
325  if( key_equal{ }( m_data[n].key, key ) ) {
326  return n;
327  }
328  }
329  return { };
330  }

298  static constexpr size_type scale_hash( size_type hash,
299  size_type range_size ) noexcept {
300  size_type const prime_a = 18446744073709551557u;
301  size_type const prime_b = 18446744073709551533u;
302  return ( hash * prime_a + prime_b ) % range_size;
303  }

305  constexpr std::optional<size_type>
306  find_index( size_type hash, Key const &key ) const noexcept {
307  size_type const scaled_hash = scale_hash( hash, m_data.size( ) );
308 
309  for( size_type n = scaled_hash; n < m_data.size( ); ++n ) {
310  if( !m_data[n].has_value ) {
311  return n;
312  }
313  if( key_equal{ }( m_data[n].key, key ) ) {
314  return n;
315  }
316  }
317  for( size_type n = 0; n < scaled_hash; ++n ) {
318  if( !m_data[n].has_value ) {
319  return n;
320  }
321  if( key_equal{ }( m_data[n].key, key ) ) {
322  return n;
323  }
324  }
325  return { };
326  }

332  constexpr iterator insert( Key const &key ) noexcept {
333  auto const index = find_index( hasher{ }( key ), key );
334  if( !index ) {

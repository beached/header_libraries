298 [[nodiscard]] constexpr char AsciiUpper( char chr ) noexcept {
299 if( 'a' <= chr and chr <= 'z' ) {
300 return static_cast<char>(
301 static_cast<unsigned>( static_cast<unsigned char>( chr ) ) & ~32U );
302 }
303 return chr;
304 }

306 [[nodiscard]] constexpr char AsciiLower( char chr ) noexcept {
307 if( 'A' <= chr and chr <= 'Z' ) {
308 return static_cast<char>( static_cast<unsigned char>(
309 static_cast<unsigned>( static_cast<unsigned char>( chr ) ) | 32U ) );
310 }
311 return chr;
312 }

794 "Must have integral number of To's in From" );

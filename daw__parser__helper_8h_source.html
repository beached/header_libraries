383 template<typename ForwardIterator, typename Divider, typename... Dividers>
384 [[nodiscard]] auto split_on( ForwardIterator first, ForwardIterator last,
385 Divider &&divider, Dividers &&...dividers ) {
386 std::vector<ForwardIterator> endings;
387 auto result =
388 until_value( first, last, DAW_FWD( divider ), DAW_FWD( dividers )... );

390 endings.push_back( result.last );

433 template<typename T>
434 [[nodiscard]] constexpr bool is_unicode_whitespace( T val ) noexcept {
435 switch( static_cast<uint32_t>( val ) ) {
436 case 0x00000009: // CHARACTER TABULATION
437 case 0x0000000A: // LINE FEED
438 case 0x0000000B: // LINE TABULATION
439 case 0x0000000C: // FORM FEED
440 case 0x0000000D: // CARRIAGE RETURN
441 case 0x00000020: // SPACE
442 case 0x00000085: // NEXT LINE
443 case 0x000000A0: // NO-BREAK SPACE
444 case 0x00001680: // OGHAM SPACE MARK
445 case 0x00002000: // EN QUAD
446 case 0x00002001: // EM QUAD
447 case 0x00002002: // EN SPACE
448 case 0x00002003: // EM SPACE
449 case 0x00002004: // THREE-PER-EM SPACE
450 case 0x00002005: // FOUR-PER-EM SPACE
451 case 0x00002006: // SIX-PER-EM SPACE
452 case 0x00002007: // FIGURE SPACE
453 case 0x00002008: // PUNCTUATION SPACE
454 case 0x00002009: // THIN SPACE
455 case 0x0000200A: // HAIR SPACE
456 case 0x00002028: // LINE SEPARATOR
457 case 0x00002029: // PARAGRAPH SEPARATOR
458 case 0x0000202F: // NARROW NO-BREAK SPACE
459 case 0x0000205F: // MEDIUM MATHEMATICAL SPACE
460 case 0x00003000: // IDEOGRAPHIC SPACE
461 return true;
462 default:
463 return false;
464 }

503 template<typename ForwardIterator>
504 [[nodiscard]] constexpr auto trim_right( ForwardIterator first,
505 ForwardIterator last ) {

507 // left to right. So that sucks for perf, try not to just right trim

538 template<typename T>
539 class is_crlf {
540 using value_type = daw::traits::root_type_t<T>;
541 std::list<value_type> m_last_values;
542 size_t m_count;
543
544 [[nodiscard]] inline bool match( ) const noexcept {
545 if( m_last_values.size( ) != 2 * m_count ) {
546 return false;
547 }
548 bool result = true;
549 for( auto it = m_last_values.begin( );
550 result and it != m_last_values.end( ); ++it ) {
551 result = result and is_a( *( it ), '\r' );
552 ++it;
553 result = result and is_a( *( it ), '\n' );
554 }
555 return result;
556 }
557
558 public:
559 is_crlf( size_t count = 1 )
560 : m_last_values{ }
561 , m_count{ count } {}
562
563 [[nodiscard]] inline bool operator( )( value_type const &v ) {
564 m_last_values.push_back( std::move( v ) );
565 while( m_last_values.size( ) > 4 ) {
566 m_last_values.pop_front( );
567 }
568 return match( );
569 }

665 template<typename ForwardIterator, typename Value>
666 [[nodiscard]] constexpr find_result_t<ForwardIterator>
667 until_last_of( ForwardIterator first, ForwardIterator last,
668 Value const &value ) {
669 auto result = until_value( first, last, value );
670 bool found = static_cast<bool>( result );
671 while( result and result.end( ) != last ) {
672 auto tmp = until_value( std::next( result.end( ) ), last, value );
673 result.found = static_cast<bool>( tmp );
674 if( tmp ) {
675 result.last = tmp.end( );
676 }
677 }
678 result.found = found;

682 template<typename ForwardIterator, typename Sequence>
683 [[nodiscard]] constexpr find_result_t<ForwardIterator>
684 find_first_of_any( ForwardIterator first, ForwardIterator last,
685 Sequence values ) {
686 auto first_val = *std::begin( values );
687 auto result = until_value( first, last, first_val );
688 if( not result ) {
689 return result;
690 }
691 auto last_first = result.first;
692 auto new_end = std::next( result.first );
693 while( new_end != last ) {
694 auto seq_pos = std::next( std::begin( values ) );
695 while( seq_pos != std::end( values ) and new_end != last and
696 *seq_pos == *new_end ) {
697 if( *new_end == first_val ) {
698 last_first = new_end;
699 }
700 ++seq_pos;
701 ++new_end;
702 }
703 if( seq_pos == std::end( values ) ) {
704 result.last = new_end;
705 return result;
706 }
707 result = until_value( first, last, first_val );
708 }
709 return make_find_result( first, last, false );
710 }

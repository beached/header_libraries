384 template<typename ForwardIterator, typename Divider, typename... Dividers>
385 [[nodiscard]] auto split_on( ForwardIterator first, ForwardIterator last,
386 Divider &&divider, Dividers &&...dividers ) {
387 std::vector<ForwardIterator> endings;
388 auto result =
389 until_value( first, last, DAW_FWD( divider ), DAW_FWD( dividers )... );
390 while( result ) {
391 endings.push_back( result.last );
392 result = until_value( result.last, last, DAW_FWD( divider ),

394 }
395 if( result.first != result.last ) {
396 endings.push_back( result.last );
397 }

434 template<typename T>
435 [[nodiscard]] constexpr bool is_unicode_whitespace( T val ) noexcept {
436 switch( static_cast<uint32_t>( val ) ) {
437 case 0x00000009: // CHARACTER TABULATION
438 case 0x0000000A: // LINE FEED
439 case 0x0000000B: // LINE TABULATION
440 case 0x0000000C: // FORM FEED
441 case 0x0000000D: // CARRIAGE RETURN
442 case 0x00000020: // SPACE
443 case 0x00000085: // NEXT LINE
444 case 0x000000A0: // NO-BREAK SPACE
445 case 0x00001680: // OGHAM SPACE MARK
446 case 0x00002000: // EN QUAD
447 case 0x00002001: // EM QUAD
448 case 0x00002002: // EN SPACE
449 case 0x00002003: // EM SPACE
450 case 0x00002004: // THREE-PER-EM SPACE
451 case 0x00002005: // FOUR-PER-EM SPACE
452 case 0x00002006: // SIX-PER-EM SPACE
453 case 0x00002007: // FIGURE SPACE
454 case 0x00002008: // PUNCTUATION SPACE
455 case 0x00002009: // THIN SPACE
456 case 0x0000200A: // HAIR SPACE
457 case 0x00002028: // LINE SEPARATOR
458 case 0x00002029: // PARAGRAPH SEPARATOR
459 case 0x0000202F: // NARROW NO-BREAK SPACE
460 case 0x0000205F: // MEDIUM MATHEMATICAL SPACE
461 case 0x00003000: // IDEOGRAPHIC SPACE
462 return true;
463 default:
464 return false;
465 }

504 template<typename ForwardIterator>
505 [[nodiscard]] constexpr auto trim_right( ForwardIterator first,
506 ForwardIterator last ) {

508 // left to right. So that sucks for perf, try not to just right trim

539 template<typename T>
540 class is_crlf {
541 using value_type = daw::traits::root_type_t<T>;
542 std::list<value_type> m_last_values;
543 size_t m_count;
544
545 [[nodiscard]] inline bool match( ) const noexcept {
546 if( m_last_values.size( ) != 2 * m_count ) {
547 return false;
548 }
549 bool result = true;
550 for( auto it = m_last_values.begin( );
551 result and it != m_last_values.end( ); ++it ) {
552 result = result and is_a( *( it ), '\r' );
553 ++it;
554 result = result and is_a( *( it ), '\n' );
555 }
556 return result;
557 }
558
559 public:
560 is_crlf( size_t count = 1 )
561 : m_last_values{ }
562 , m_count{ count } {}
563
564 [[nodiscard]] inline bool operator( )( value_type const &v ) {
565 m_last_values.push_back( std::move( v ) );
566 while( m_last_values.size( ) > 4 ) {
567 m_last_values.pop_front( );
568 }
569 return match( );
570 }

667 [[nodiscard]] constexpr find_result_t<ForwardIterator>
668 until_last_of( ForwardIterator first, ForwardIterator last,
669 Value const &value ) {
670 auto result = until_value( first, last, value );
671 bool found = static_cast<bool>( result );
672 while( result and result.end( ) != last ) {
673 auto tmp = until_value( std::next( result.end( ) ), last, value );
674 result.found = static_cast<bool>( tmp );
675 if( tmp ) {
676 result.last = tmp.end( );
677 }
678 }
679 result.found = found;

683 template<typename ForwardIterator, typename Sequence>
684 [[nodiscard]] constexpr find_result_t<ForwardIterator>
685 find_first_of_any( ForwardIterator first, ForwardIterator last,
686 Sequence values ) {
687 auto first_val = *std::begin( values );
688 auto result = until_value( first, last, first_val );
689 if( not result ) {
690 return result;
691 }
692 auto last_first = result.first;
693 auto new_end = std::next( result.first );
694 while( new_end != last ) {
695 auto seq_pos = std::next( std::begin( values ) );
696 while( seq_pos != std::end( values ) and new_end != last and
697 *seq_pos == *new_end ) {
698 if( *new_end == first_val ) {
699 last_first = new_end;
700 }
701 ++seq_pos;
702 ++new_end;
703 }
704 if( seq_pos == std::end( values ) ) {
705 result.last = new_end;
706 return result;
707 }
708 result = until_value( first, last, first_val );
709 }
710 return make_find_result( first, last, false );
711 }

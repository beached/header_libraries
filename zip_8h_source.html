257  template<Range... Ranges>
258  [[nodiscard]] constexpr auto ZipMore( Ranges &&...rs ) {
259  return [=]<Range R>( R &&r ) {
260  if constexpr( requires( R ) {
261  typename iterator_t<R>::i_am_a_daw_zip_iterator_class;
262  } ) {
263  // zip_view
264  auto tp_first = std::begin( DAW_FWD( r ) ).base( );
265  auto tp_last = std::end( DAW_FWD( r ) ).base( );
266  static_assert( std::tuple_size_v<decltype( tp_first )> ==
267  std::tuple_size_v<decltype( tp_last )>,
268  "There is a bug in zip_view. The begin( ) and end( ) "
269  "should have the same size" );
270  return [&]<std::size_t... Is>( std::index_sequence<Is...> ) {
271  return zip_view( rs...,
272  range_t{ std::get<Is>( std::move( tp_first ) ),
273  std::get<Is>( std::move( tp_last ) ) }... );
274  }( std::make_index_sequence<
275  std::tuple_size_v<decltype( tp_first )>>{ } );
276  } else if constexpr( daw::is_tuple_like_v<range_value_t<R>> ) {
277  // tuple like.
278  return [&]<std::size_t... Is>( std::index_sequence<Is...> ) {
279  return zip_view( rs..., std::get<Is>( DAW_FWD( r ) )... );
280  }( std::make_index_sequence<std::tuple_size_v<DAW_TYPEOF( r )>>{ } );
281  } else {
282  return zip_view( rs..., DAW_FWD( r ) );
283  }
284  };

275 template<Range... Ranges>
276 [[nodiscard]] constexpr auto ZipMore( Ranges &&...rs ) {
277 return [=]<Range R>( R &&r ) {
278 if constexpr( requires( R ) {
279 typename iterator_t<R>::i_am_a_daw_zip_iterator_class;
280 } ) {
281 // zip_view
282 auto tp_first = std::begin( DAW_FWD( r ) ).base( );
283 auto tp_last = std::end( DAW_FWD( r ) ).base( );
284 static_assert( std::tuple_size_v<decltype( tp_first )> ==
285 std::tuple_size_v<decltype( tp_last )>,
286 "There is a bug in zip_view. The begin( ) and end( ) "
287 "should have the same size" );
288 return [&]<std::size_t... Is>( std::index_sequence<Is...> ) {
289 return zip_view( rs...,
290 range_t{ std::get<Is>( std::move( tp_first ) ),
291 std::get<Is>( std::move( tp_last ) ) }... );
292 }( std::make_index_sequence<
293 std::tuple_size_v<decltype( tp_first )>>{ } );
294 } else if constexpr( daw::is_tuple_like_v<range_value_t<R>> ) {
295 // tuple like.
296 return [&]<std::size_t... Is>( std::index_sequence<Is...> ) {
297 return zip_view( rs..., std::get<Is>( DAW_FWD( r ) )... );
298 }( std::make_index_sequence<std::tuple_size_v<DAW_TYPEOF( r )>>{ } );
299 } else {
300 return zip_view( rs..., DAW_FWD( r ) );
301 }
302 };

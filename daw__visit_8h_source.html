53 template<std::size_t Idx, typename Variant>
54 DAW_ATTRIB_FLATINLINE constexpr decltype( auto ) get_nt( Variant &&var ) {
55 using variant_t = std::remove_reference_t<Variant>;
56 if constexpr( get_nt_details::has_adl_get_if_v<variant_t> ) {
57 using get_nt_details::get_if;
58 auto *ptr = get_if<Idx>( std::addressof( var ) );
59 DAW_ASSUME( ptr != nullptr );
60 if constexpr( std::is_rvalue_reference_v<Variant> ) {
61 return std::move( *ptr );
62 } else {
63 return *ptr;
64 }
65 } else if constexpr( get_nt_details::has_std_get_if_v<variant_t> ) {
66 auto *ptr = std::get_if<Idx>( std::addressof( var ) );
67 DAW_ASSUME( ptr != nullptr );
68 if constexpr( std::is_rvalue_reference_v<Variant> ) {
69 return std::move( *ptr );
70 } else {
71 return *ptr;
72 }
73 } else if constexpr( get_nt_details::has_adl_get_v<variant_t> ) {
74 using get_nt_details::get;
75 return get<Idx>( DAW_FWD( var ) );
76 } else {
77 static_assert( get_nt_details::has_std_get_v<variant_t> );
78 return std::get<Idx>( DAW_FWD( var ) );
79 }

120 template<typename F, typename... Fs>
121 DAW_ATTRIB_FLATINLINE [[nodiscard]] constexpr decltype( auto )
122 overload( F &&f, Fs &&...fs ) noexcept {
123 if constexpr( sizeof...( Fs ) > 0 ) {
124 return overload_t{ daw::traits::lift_func( DAW_FWD( f ) ),
125 daw::traits::lift_func( DAW_FWD( fs ) )... };
126 } else {
127 if constexpr( std::is_function_v<daw::remove_cvref_t<F>> ) {
128 static_assert(
129 daw::traits::func_traits<F>::arity == 1,
130 "Function pointers only valid for functions with 1 argument" );
131 }
132 return daw::traits::lift_func( DAW_FWD( f ) );
133 }

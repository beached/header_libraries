52  template<typename Value>
53  static constexpr hash_value_t
54  append_hash_block( hash_value_t current_hash, Value value,
55  size_t count = sizeof( Value ) ) noexcept {
56  hash_value_t const JODY_HASH_CONSTANT = 0x1F5B;
57  hash_value_t const JODY_HASH_SHIFT = 14;
58 
59  auto len = count / sizeof( hash_value_t );
60  for( ; len > 0; --len ) {
61  auto const element =
62  extract_value<hash_value_t>( value, len * sizeof( hash_value_t ) );
63  current_hash += element;
64  current_hash += JODY_HASH_CONSTANT;
65  current_hash =
66  static_cast<hash_value_t>( current_hash << JODY_HASH_SHIFT ) |
67  current_hash >> ( sizeof( hash_value_t ) * 8 - JODY_HASH_SHIFT );
68  current_hash ^= element;
69  current_hash =
70  static_cast<hash_value_t>( current_hash << JODY_HASH_SHIFT ) |
71  current_hash >> ( sizeof( hash_value_t ) * 8 - JODY_HASH_SHIFT );
72  current_hash ^= JODY_HASH_CONSTANT;
73  current_hash += element;
74  }
75  return current_hash;

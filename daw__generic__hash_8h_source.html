43 static_assert( sizeof( Result ) <= sizeof( T ),
44 "result size must be <= to source size" );
45 auto const mask = static_cast<T>( ~( Result{ } & 0 ) ) << rshft_bits;
46 value &= mask;
47 value >>= rshft_bits;
48 return static_cast<Result>( value );

51 template<typename Value>
52 static constexpr hash_value_t
53 append_hash_block( hash_value_t current_hash, Value value,
54 std::size_t count = sizeof( Value ) ) noexcept {
55 hash_value_t const JODY_HASH_CONSTANT = 0x1F5B;
56 hash_value_t const JODY_HASH_SHIFT = 14;
57
58 auto len = count / sizeof( hash_value_t );
59 for( ; len > 0; --len ) {
60 auto const element =
61 extract_value<hash_value_t>( value, len * sizeof( hash_value_t ) );
62 current_hash += element;
63 current_hash += JODY_HASH_CONSTANT;
64 current_hash =
65 static_cast<hash_value_t>( current_hash << JODY_HASH_SHIFT ) |
66 current_hash >> ( bit_count_v<hash_value_t> - JODY_HASH_SHIFT );
67 current_hash ^= element;
68 current_hash =
69 static_cast<hash_value_t>( current_hash << JODY_HASH_SHIFT ) |
70 current_hash >> ( bit_count_v<hash_value_t> - JODY_HASH_SHIFT );
71 current_hash ^= JODY_HASH_CONSTANT;
72 current_hash += element;
73 }
74 return current_hash;
75 }

203  template<typename Fn>
204  struct Map_t {
205  mutable Fn m_func;
206 
207  [[nodiscard]] constexpr auto operator( )( auto &&r ) const {
208  using R = DAW_TYPEOF( r );
209  if constexpr( Range<R> ) {
210  static_assert( std::is_invocable_v<Fn, range_reference_t<R>>,
211  "Map requires the function to be able to be called "
212  "with invoke and the range_reference_t" );
213  static_assert( traits::NoVoidResults<Fn, range_reference_t<R>>,
214  "Map requires the result to not be void" );
215  return map_view( std::begin( r ), std::end( r ), m_func );
216  } else {
217  static_assert( std::is_invocable_v<Fn, R>,
218  "Map requires the function to be able to be called "
219  "with invoke and passed value" );
220  static_assert( traits::NoVoidResults<Fn, R>,
221  "Map requires the result to not be void" );
222  return std::invoke( m_func, DAW_FWD( r ) );
223  }
224  }
225  };

253  [[nodiscard]] constexpr auto operator( )( auto &&r ) const {
254  using R = DAW_TYPEOF( r );

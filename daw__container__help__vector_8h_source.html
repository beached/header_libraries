27 template<typename T, typename Alloc>
28 struct std_vector_layout {
29 static_assert( not std::is_same_v<T, bool>, "bad instance" );
30 using alloc_type = Alloc;
31 using alloc_traits = std::allocator_traits<alloc_type>;
32 using pointer = typename std::allocator_traits<alloc_type>::pointer;
33
34 pointer m_first;
35 pointer m_last;
36 std::__compressed_pair<pointer, alloc_type> m_capacity;
37
38 constexpr std::size_t capacity( ) const {
39 return static_cast<std::size_t>( m_capacity.first( ) - m_first );
40 }
41
42 constexpr std::size_t size( ) const {
43 return static_cast<std::size_t>( m_last - m_first );
44 }
45
46 constexpr alloc_type take_allocator( ) {
47 static_assert(
48 alloc_traits::is_always_equal::value or
49 alloc_traits::propagate_on_container_move_assignment::value,
50 "Unsupported allocator" );
51 if constexpr( alloc_traits::is_always_equal::value ) {
52 return alloc_type( );
53 } else {
54 return std::move(
55 *static_cast<alloc_type *>( m_capacity.second( ) ) );
56 }
57 }
58
59 constexpr pointer take_buffer( ) {
60 auto ret = daw::exchange( m_first, pointer( ) );
61 m_last = pointer( );
62 m_capacity.first( ) = pointer( );
63 return ret;
64 }
65
66 constexpr void capacity( std::size_t sz ) {
67 m_capacity.first( ) = m_first + static_cast<std::ptrdiff_t>( sz );
68 }
69
70 constexpr void size( std::size_t sz ) {
71 m_last = m_first + static_cast<std::ptrdiff_t>( sz );
72 }
73
74 constexpr void give_allocator( alloc_type alloc ) {
75 assert( m_first == pointer( ) );
76 static_assert(
77 alloc_traits::is_always_equal::value or
78 alloc_traits::propagate_on_container_move_assignment::value,
79 "Unsupported allocator" );
80 if constexpr( alloc_traits::propagate_on_container_move_assignment::
81 value ) {
82 m_capacity.second( ) = std::move( alloc );
83 }
84 }
85
86 constexpr void give_buffer( pointer p ) {
87 // Assumes vector is empty
88 assert( m_first == pointer( ) );
89 m_first = p;
90 }

68 Args... args ) {
69 std::vector<string_details::string_t> result;
70 result.reserve( sizeof...( args ) + 1 );
71 using std::to_string;
72 if constexpr( std::is_same_v<Arg, std::string> ) {

86 template<typename String>
87 auto split( String const &sv ) {
88 using string_t = std::decay_t<String>;
89 std::vector<string_t> result;
90 auto pos = sv.begin( );
91 decltype( pos ) find_iter;
92 while( true ) {
93 find_iter = find_if( pos, sv.end( ), []( auto const &c ) {
94 return static_cast<bool>( isspace( c ) );
95 } );
96 result.emplace_back( pos, find_iter );
97 if( find_iter == sv.end( ) ) {
98 break;
99 }
100 pos = ++find_iter;
101 }
102 return result;
103 }

237 // Summary: takes a format string like "{0} {1} {2}" and in this case 3
238 // parameters
239 template<typename String, typename Arg, typename... Args>
240 string_details::string_t fmt( String &&format_, Arg arg, Args... args ) {
241 std::regex const reg( R"^(\{(\d+(:\d)*)\})^" );
242 std::stringstream ss;
243 auto const arguments = string_details::unknowns_to_string( arg, args... );
244 std::smatch sm;
245 std::string format{ format_ };
246 while( std::regex_search( format, sm, reg ) ) {
247 auto const &prefix = sm.prefix( ).str( );
248 ss << prefix;
249 if( ends_with( prefix, "{" ) ) {
250 ss << sm[0].str( );
251 } else {
252 auto delims = split( sm[1].str( ), ':' );
253 if( 1 >= delims.size( ) ) {
254 std::stringstream ss2;
255 ss2 << sm[1].str( );
256 size_t v;
257 ss2 >> v;
258 ss << arguments[v];
259 } else if( 2 == delims.size( ) ) {
260 // Assumes the argument at pos is a double. If not, will crash
261 size_t pos = string_details::lexical_cast<size_t>( delims[0] );
262 int precision = string_details::lexical_cast<int>( delims[1] );
263 ss << std::fixed << std::setprecision( precision )
264 << string_details::lexical_cast<double>( arguments[pos] );
265 } else {
266 daw::exception::daw_throw<std::out_of_range>(
267 fmt( "Unknown string format. Too many colons(",
268 delims.size( ),
269 "): ",
270 sm[1].str( ) ) );
271 }
272 }
273 format = sm.suffix( ).str( );
274 }
275 ss << format;
276 return ss.str( );
277 }

382 template<typename CharT = char,
383 typename traits = std::char_traits<CharT>,
384 typename Alloc = std::allocator<CharT>>
385 void search_replace( std::basic_string<CharT, traits, Alloc> &in_str,
386 CharT const *search_for,
387 CharT const *replace_with ) {
388 struct {
389 inline auto operator( )( wchar_t const *ptr ) const noexcept {
390 return wcslen( ptr );
391 }
392 inline auto operator( )( char const *ptr ) const noexcept {
393 return strlen( ptr );
394 }
395 } str_size;
396
397 size_t pos = 0u;
398 auto const search_for_len = str_size( search_for );
399 auto const replace_with_len = str_size( replace_with );
400
401 while( ( pos = in_str.find( search_for, pos ) ) !=
402 std::basic_string<CharT, traits, Alloc>::npos ) {
403 in_str.replace( pos, search_for_len, replace_with );
404 pos += replace_with_len;
405 }
406 }

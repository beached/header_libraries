84  template<typename String>
85  auto split( String const &sv ) {
86  using string_t = std::decay_t<String>;
87  std::vector<string_t> result;
88  auto pos = sv.begin( );
89  decltype( pos ) find_iter;
90  while( true ) {
91  find_iter = find_if( pos, sv.end( ), []( auto const &c ) {
92  return static_cast<bool>( isspace( c ) );
93  } );
94  result.emplace_back( pos, find_iter );
95  if( find_iter == sv.end( ) ) {
96  break;
97  }
98  pos = ++find_iter;
99  }
100  return result;
101  }

233  // Summary: takes a format string like "{0} {1} {2}" and in this case 3
234  // parameters
235  template<typename String, typename Arg, typename... Args>
236  string_details::string_t fmt( String &&format_, Arg arg, Args... args ) {
237  std::regex const reg( R"^(\{(\d+(:\d)*)\})^" );
238  std::stringstream ss;
239  auto const arguments = string_details::unknowns_to_string( arg, args... );
240  std::smatch sm;
241  std::string format{ format_ };
242  while( std::regex_search( format, sm, reg ) ) {
243  auto const &prefix = sm.prefix( ).str( );
244  ss << prefix;
245  if( ends_with( prefix, "{" ) ) {
246  ss << sm[0].str( );
247  } else {
248  auto delims = split( sm[1].str( ), ':' );
249  if( 1 >= delims.size( ) ) {
250  std::stringstream ss2;
251  ss2 << sm[1].str( );
252  size_t v;
253  ss2 >> v;
254  ss << arguments[v];
255  } else if( 2 == delims.size( ) ) {
256  // Assumes the argument at pos is a double. If not, will crash
257  size_t pos = string_details::lexical_cast<size_t>( delims[0] );
258  int precision = string_details::lexical_cast<int>( delims[1] );
259  ss << std::fixed << std::setprecision( precision )
260  << string_details::lexical_cast<double>( arguments[pos] );
261  } else {
262  daw::exception::daw_throw<std::out_of_range>(
263  fmt( "Unknown string format. Too many colons(", delims.size( ),
264  "): ", sm[1].str( ) ) );
265  }
266  }
267  format = sm.suffix( ).str( );
268  }
269  ss << format;
270  return ss.str( );
271  }

18namespace daw {
19 template<std::size_t BitCount>
20 class bitset {
21 static_assert( BitCount > 2, "BitCount must be greater than 2" );
22 using limb_t = std::uint32_t;
23 static constexpr std::size_t limb_size = sizeof( limb_t ) * 8;
24 static constexpr std::size_t num_limbs = ( BitCount - 1 ) / limb_size + 1;
25 limb_t m_bits[num_limbs]{ };
26
27 DAW_ATTRIB_INLINE static constexpr std::size_t
28 get_limb( std::size_t index ) {
29 assert( index < size( ) );
30 return index / limb_size;
31 }
32
33 DAW_ATTRIB_INLINE static constexpr limb_t get_mask( std::size_t index ) {
34 return limb_t{ 1 } << ( index % limb_size );
35 }
36
37 public:
38 bitset( ) = default;
39
40 static DAW_CONSTEVAL std::size_t size( ) {
41 return BitCount;
42 }
43
44 DAW_ATTRIB_INLINE constexpr void unset( std::size_t index ) {
45 assert( index < size( ) );
46 m_bits[get_limb( index )] &= ~( 1u << index % 4 );
47 auto const limb = get_limb( index );
48 auto const mask = get_mask( index );
49 m_bits[limb] &= ~mask;
50 }
51
52 DAW_ATTRIB_INLINE constexpr void set( std::size_t index ) {
53 assert( index < size( ) );
54 auto const limb = get_limb( index );
55 auto const mask = get_mask( index );
56 m_bits[limb] |= mask;
57 }
58
59 DAW_ATTRIB_INLINE constexpr void set( std::size_t index, bool value ) {
60 assert( index < size( ) );
61 if( value ) {
62 set( index );
63 } else {
64 unset( index );
65 }
66 }
67
68 DAW_ATTRIB_INLINE constexpr bool test( std::size_t index ) const {
69 assert( index < size( ) );
70 auto const limb = get_limb( index );
71 auto const mask = get_mask( index );
72 return m_bits[limb] & mask;
73 }
74 };
75} // namespace daw

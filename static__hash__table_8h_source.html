45 [[nodiscard]] static constexpr size_t scale_hash( size_t hash_value ) {
46 const size_t prime_a = 18446744073709551557u;

66 [[nodiscard]] constexpr size_t find_impl( size_t const hash ) const {
67 auto const scaled_hash = scale_hash( hash );
68 for( size_t n = scaled_hash; n < Capacity; ++n ) {
69 if( m_values[n].hash_value ==
70 static_cast<std::size_t>( hash_sentinals::empty ) ) {
71 return n;
72 } else if( m_values[n].hash_value == hash ) {
73 return n;
74 }
75 }
76 for( size_t n = 0; n < scaled_hash; ++n ) {
77 if( m_values[n].hash_value ==
78 static_cast<std::size_t>( hash_sentinals::empty ) ) {
79 return n;
80 } else if( m_values[n].hash_value == hash ) {
81 return n;
82 }
83 }
84 return Capacity;
85 }

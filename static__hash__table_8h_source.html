47 [[nodiscard]] static constexpr std::size_t
48 scale_hash( std::uint64_t hash_value ) {
49 constexpr auto prime_a = 18446744073709551557ULL;
50 constexpr auto prime_b = 18446744073709551533ULL;
51 return static_cast<std::size_t>( ( hash_value * prime_a + prime_b ) %

71 [[nodiscard]] constexpr std::size_t find_impl( std::uint64_t hash ) const {
72 auto const scaled_hash = scale_hash( hash );
73 for( std::size_t n = scaled_hash; n < Capacity; ++n ) {
74 if( m_values[n].hash_value ==
75 static_cast<std::size_t>( hash_sentinals::empty ) ) {
76 return n;
77 } else if( m_values[n].hash_value == hash ) {
78 return n;
79 }
80 }
81 for( std::size_t n = 0; n < scaled_hash; ++n ) {
82 if( m_values[n].hash_value ==
83 static_cast<std::size_t>( hash_sentinals::empty ) ) {
84 return n;
85 } else if( m_values[n].hash_value == hash ) {
86 return n;
87 }
88 }
89 return Capacity;
90 }

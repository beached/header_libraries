286 constexpr void resize( size_type const count ) {
287
288 daw::exception::precondition_check<std::out_of_range>(
289 count <= capacity( ), "Attempt to resize past capacity of fix_stack" );
290
291 if( count > size( ) ) {
292 if( can_move_front( count ) ) {
293 do_move_to_front( );
294 }
295 for( size_type n = size( ); n < count; ++n ) {
296 m_stack[n] = value_type{ };
297 }
298 }
299 m_index = count;
300 }

302 constexpr iterator erase( const_iterator pos ) {
303 auto first = std::next( begin( ), std::distance( cbegin( ), pos ) );
304 auto last = std::next( first, 1 );
305 while( last != end( ) ) {
306 daw::iter_swap( first, last );
307 std::advance( first, 1 );
308 std::advance( last, 1 );
309 }
310 resize( size( ) - 1 );
311 if( pos == cend( ) ) {
312 return end( );
313 }
314 return std::next( begin( ), std::distance( cbegin( ), pos ) + 1 );
315 }
316
317 constexpr iterator erase( const_iterator first, const_iterator last ) {
318 auto first1 = std::next( begin( ), std::distance( cbegin( ), first ) );
319 auto first2 = std::next( begin( ), std::distance( cbegin( ), last ) );
320 while( first2 != end( ) ) {
321 daw::iter_swap( first1, first2 );
322 std::advance( first1, 1 );
323 std::advance( first2, 1 );
324 }
325 auto const dist = std::distance( first, last );
326
327 daw::exception::dbg_precondition_check<std::out_of_range>(
328 static_cast<difference_type>( size( ) ) > dist and dist >= 0,
329 "Attempt to resize past capacity of fix_stack" );
330
331 resize( size( ) - static_cast<size_type>( dist ) );
332 if( last == cend( ) ) {
333 return end( );
334 }
335 return std::next( begin( ), std::distance( cbegin( ), last ) + 1 );
336 }

62 constexpr void deallocate( pointer &ptr, size_type n ) noexcept {
63 DAW_SAFE_POINTER_ENSURE( ptr );
64 DAW_SAFE_POINTER_ENSURE( not ptr.get_base( ) or
65 ( ptr.get( ) == ptr.get_base( ) ) );
66 value_type *p = ptr.destroy( );

68 }

82 }

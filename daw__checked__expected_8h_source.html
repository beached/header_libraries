31 template<typename ExpectedException>
32 void is_expected_exception( std::exception_ptr ptr ) {
33 try {
34 std::rethrow_exception( ptr );
35 } catch( ExpectedException const & ) { return; } catch( ... ) {
36 struct unexpected_exception_type {
37 }; // Cannot catch what you cannot name
38 daw::exception::daw_throw<unexpected_exception_type>( );
39 }
40 }
41
42 template<typename... ExpectedExceptions, typename T>
43 constexpr auto is_expected_exception( T && ) noexcept
44 -> std::enable_if_t<( sizeof...( ExpectedExceptions ) == 0 )> {}
45
46 template<typename ExpectedException, typename... ExpectedExceptions>
47 auto is_expected_exception( std::exception_ptr ptr )
48 -> std::enable_if_t<( sizeof...( ExpectedExceptions ) > 0 )> {
49
50 try {
51 std::rethrow_exception( ptr );
52 } catch( ExpectedException const & ) { return; } catch( ... ) {
53 is_expected_exception<ExpectedExceptions...>( ptr );
54 }
55 }

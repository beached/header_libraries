32 [[nodiscard]] auto get_child_nodes( Graph &&graph, Node &&node ) {

34 std::remove_reference_t<decltype( graph.get_node( node.id( ) ) )>;

36 std::vector<node_t> result{ };

43 template<typename Node, typename T, typename Graph, typename Function,
44 typename Compare>
45 void topological_sorted_walk( Graph &&graph, Function &&func,
46 Compare comp = Compare{ } ) {
47
48 // Find all nodes that do not have incoming entries
49 std::vector<Node> root_nodes = [&] {
50 auto root_node_ids = graph.find_roots( );
51 auto result = std::vector<Node>( );
52 result.reserve( root_node_ids.size( ) );
53 std::transform( std::begin( root_node_ids ), std::end( root_node_ids ),
54 std::back_inserter( result ), [&]( node_id_t id ) {
55 return graph.get_node( id );
56 } );
57 return result;
58 }( );
59
60 constexpr bool perform_sort_v =
61 not std::is_same_v<Compare, daw::graph_alg_impl::NoSort>;
62
63 if constexpr( perform_sort_v ) {
64 std::sort( std::begin( root_nodes ), std::end( root_nodes ), comp );
65 }
66
67 auto excluded_edges =
68 std::unordered_map<node_id_t, std::vector<node_id_t>>( );
69
70 auto const exclude_edge = [&]( node_id_t from, node_id_t to ) {
71 auto &child = excluded_edges[to];
72 auto pos = std::find( std::begin( child ), std::end( child ), from );
73 if( pos == std::end( child ) ) {
74 child.push_back( from );
75 }
76 };
77
78 auto const has_parent_nodes = [&]( node_id_t n_id ) {
79 auto incoming = graph.get_raw_node( n_id ).incoming_edges( );
80 if( incoming.empty( ) ) {
81 return false;
82 }
83 if( excluded_edges.count( n_id ) == 0 ) {
84 return true;
85 }
86 for( auto ex_id : excluded_edges[n_id] ) {
87 incoming.erase( ex_id );
88 }
89 return not incoming.empty( );
90 };
91
92 while( not root_nodes.empty( ) ) {
93 auto node = root_nodes.back( );
94 root_nodes.pop_back( );
95 func( node );
96 {
97 auto child_nodes = graph_alg_impl::get_child_nodes( graph, node );
98 if constexpr( perform_sort_v ) {
99 std::sort( std::begin( child_nodes ), std::end( child_nodes ),
100 [&]( auto &&...args ) {
101 return not comp( DAW_FWD( args )... );
102 } );
103 }
104 for( auto child : child_nodes ) {
105 exclude_edge( node.id( ), child.id( ) );
106 if( not has_parent_nodes( child.id( ) ) ) {
107 root_nodes.push_back( child );
108 }
109 }
110 }
111 if constexpr( perform_sort_v ) {
112 std::sort( std::begin( root_nodes ), std::end( root_nodes ), comp );
113 }
114 }
115 }

117 template<typename T, typename ChildOrder, typename Graph, typename Function>
118 void bfs_walk( Graph &&graph, daw::node_id_t start_node_id, Function &&func,
119 ChildOrder ord ) {
120 std::unordered_set<daw::node_id_t> visited{ };
121 std::deque<daw::node_id_t> path{ };
122 path.push_back( start_node_id );
123
124 while( not path.empty( ) ) {
125 auto current_node = graph.get_node( path.front( ) );
126 path.pop_front( );
127 func( current_node );
128 visited.insert( current_node.id( ) );
129 if constexpr( std::is_same_v<ChildOrder, UnorderedWalk> ) {
130 std::copy_if( std::begin( current_node.outgoing_edges( ) ),
131 std::end( current_node.outgoing_edges( ) ),
132 std::back_inserter( path ), [&]( auto const &n_id ) {
133 return visited.count( n_id ) == 0;
134 } );
135 } else {
136 auto children = std::vector<daw::node_id_t>( );
137
138 std::copy_if( std::begin( current_node.outgoing_edges( ) ),
139 std::end( current_node.outgoing_edges( ) ),
140 std::back_inserter( children ),
141 [&]( auto const &n_id ) {
142 return visited.count( n_id ) == 0;
143 } );
144
145 std::sort( children.begin( ), children.end( ),
146 [&]( daw::node_id_t left_id, daw::node_id_t right_id ) {
147 return ord( graph.get_node( left_id ).value( ),
148 graph.get_node( right_id ).value( ) );
149 } );
150 std::copy( children.begin( ), children.end( ),
151 std::back_inserter( path ) );
152 }
153 }
154 }

158 ChildOrder ord ) {

164 auto current_node = graph.get_node( path.back( ) );
165 path.pop_back( );
166 func( current_node );
167 visited.insert( current_node.id( ) );
168 if constexpr( std::is_same_v<ChildOrder, UnorderedWalk> ) {
169 std::copy_if( std::begin( current_node.outgoing_edges( ) ),
170 std::end( current_node.outgoing_edges( ) ),
171 std::back_inserter( path ), [&]( auto const &n_id ) {
172 return visited.count( n_id ) == 0;
173 } );
174 } else {
175 auto children = std::vector<daw::node_id_t>( );
176 std::copy_if( std::begin( current_node.outgoing_edges( ) ),
177 std::end( current_node.outgoing_edges( ) ),
178 std::back_inserter( children ),
179 [&]( auto const &n_id ) {
180 return visited.count( n_id ) == 0;
181 } );
182

197 template<typename T, typename Compare = daw::graph_alg_impl::NoSort>
198 void mst( daw::graph_t<T> &graph, Compare /*TODO comp*/ = Compare{ } ) {
199 auto root_ids = graph.find_roots( );
200 for( auto start_node_id : root_ids ) {
201 std::unordered_set<daw::node_id_t> visited{ };
202 std::vector<std::pair<std::optional<daw::node_id_t>, daw::node_id_t>>
203 path{ };
204 path.push_back( { std::nullopt, start_node_id } );
205
206 while( not path.empty( ) ) {
207 auto current_id = path.back( );
208 path.pop_back( );
209 if( visited.count( current_id.second ) > 0 ) {
210 daw::exception::precondition_check(
211 current_id.first.has_value( ),
212 "I assumed that the root would have been first visited" );
213 graph.remove_directed_edge( *current_id.first, current_id.second );
214 continue;
215 }
216 visited.insert( current_id.second );
217 auto current_node = graph.get_node( current_id.second );
218 std::transform(
219 std::begin( current_node.outgoing_edges( ) ),
220 std::end( current_node.outgoing_edges( ) ),
221 std::back_inserter( path ), [parent = current_id.second]( auto id ) {
222 return std::pair<std::optional<daw::node_id_t>, node_id_t>( parent,
223 id );
224 } );
225 }
226 }

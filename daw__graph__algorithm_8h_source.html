43  template<typename Node, typename T, typename Graph, typename Function,
44  typename Compare>
45  void topological_sorted_walk( Graph &&graph, Function &&func,
46  Compare comp = Compare{ } ) {
47 
48  // Find all nodes that do not have incoming entries
49  std::vector<Node> root_nodes = [&] {
50  auto root_node_ids = graph.find_roots( );
51  auto result = std::vector<Node>( );
52  result.reserve( root_node_ids.size( ) );
53  std::transform( std::begin( root_node_ids ), std::end( root_node_ids ),
54  std::back_inserter( result ), [&]( node_id_t id ) {
55  return graph.get_node( id );
56  } );
57  return result;
58  }( );
59 
60  constexpr bool perform_sort_v =
61  not std::is_same_v<Compare, daw::graph_alg_impl::NoSort>;
62 
63  if constexpr( perform_sort_v ) {
64  std::sort( std::begin( root_nodes ), std::end( root_nodes ), comp );
65  }
66 
67  auto excluded_edges =
68  std::unordered_map<node_id_t, std::vector<node_id_t>>( );
69 
70  auto const exclude_edge = [&]( node_id_t from, node_id_t to ) {
71  auto &child = excluded_edges[to];
72  auto pos = std::find( std::begin( child ), std::end( child ), from );
73  if( pos == std::end( child ) ) {
74  child.push_back( from );
75  }
76  };
77 
78  auto const has_parent_nodes = [&]( node_id_t n_id ) {
79  auto incoming = graph.get_raw_node( n_id ).incoming_edges( );
80  if( incoming.empty( ) ) {
81  return false;
82  }
83  if( excluded_edges.count( n_id ) == 0 ) {
84  return true;
85  }
86  for( auto ex_id : excluded_edges[n_id] ) {
87  incoming.erase( ex_id );
88  }
89  return not incoming.empty( );
90  };
91 
92  while( not root_nodes.empty( ) ) {
93  auto node = root_nodes.back( );
94  root_nodes.pop_back( );
95  func( node );
96  {
97  auto child_nodes = graph_alg_impl::get_child_nodes( graph, node );
98  if constexpr( perform_sort_v ) {
99  std::sort( std::begin( child_nodes ), std::end( child_nodes ),
100  [&]( auto &&...args ) {
101  return not comp(
102  DAW_FWD( args )... );
103  } );
104  }
105  for( auto child : child_nodes ) {
106  exclude_edge( node.id( ), child.id( ) );
107  if( not has_parent_nodes( child.id( ) ) ) {
108  root_nodes.push_back( child );
109  }
110  }
111  }
112  if constexpr( perform_sort_v ) {
113  std::sort( std::begin( root_nodes ), std::end( root_nodes ), comp );
114  }
115  }
116  }

118  template<typename T, typename ChildOrder, typename Graph, typename Function>
119  void bfs_walk( Graph &&graph, daw::node_id_t start_node_id, Function &&func,
120  ChildOrder ord ) {
121  std::unordered_set<daw::node_id_t> visited{ };
122  std::deque<daw::node_id_t> path{ };
123  path.push_back( start_node_id );
124 
125  while( not path.empty( ) ) {
126  auto current_node = graph.get_node( path.front( ) );
127  path.pop_front( );
128  func( current_node );
129  visited.insert( current_node.id( ) );
130  if constexpr( std::is_same_v<ChildOrder, UnorderedWalk> ) {
131  std::copy_if( std::begin( current_node.outgoing_edges( ) ),
132  std::end( current_node.outgoing_edges( ) ),
133  std::back_inserter( path ), [&]( auto const &n_id ) {
134  return visited.count( n_id ) == 0;
135  } );
136  } else {
137  auto children = std::vector<daw::node_id_t>( );
138 
139  std::copy_if( std::begin( current_node.outgoing_edges( ) ),
140  std::end( current_node.outgoing_edges( ) ),
141  std::back_inserter( children ),
142  [&]( auto const &n_id ) {
143  return visited.count( n_id ) == 0;
144  } );
145 
146  std::sort( children.begin( ), children.end( ),
147  [&]( daw::node_id_t left_id, daw::node_id_t right_id ) {
148  return ord( graph.get_node( left_id ).value( ),
149  graph.get_node( right_id ).value( ) );
150  } );
151  std::copy( children.begin( ), children.end( ),
152  std::back_inserter( path ) );
153  }
154  }
155  }

158  void dfs_walk( Graph &&graph, daw::node_id_t start_node_id, Function &&func,
159  ChildOrder ord ) {
160  std::unordered_set<daw::node_id_t> visited{ };
161  std::vector<daw::node_id_t> path{ };
162  path.push_back( start_node_id );
163 
164  while( not path.empty( ) ) {
165  auto current_node = graph.get_node( path.back( ) );
166  path.pop_back( );
167  func( current_node );
168  visited.insert( current_node.id( ) );
169  if constexpr( std::is_same_v<ChildOrder, UnorderedWalk> ) {
170  std::copy_if( std::begin( current_node.outgoing_edges( ) ),
171  std::end( current_node.outgoing_edges( ) ),
172  std::back_inserter( path ), [&]( auto const &n_id ) {
173  return visited.count( n_id ) == 0;
174  } );
175  } else {
176  auto children = std::vector<daw::node_id_t>( );
177  std::copy_if( std::begin( current_node.outgoing_edges( ) ),
178  std::end( current_node.outgoing_edges( ) ),
179  std::back_inserter( children ),
180  [&]( auto const &n_id ) {
181  return visited.count( n_id ) == 0;
182  } );
183 
184  std::sort( children.begin( ), children.end( ),
185  [&]( daw::node_id_t left_id, daw::node_id_t right_id ) {
186  return not ord( graph.get_node( left_id ).value( ),
187  graph.get_node( right_id ).value( ) );
188  } );

197  // Minimum Spanning Tree
198  template<typename T, typename Compare = daw::graph_alg_impl::NoSort>
199  void mst( daw::graph_t<T> &graph, Compare /*TODO comp*/ = Compare{ } ) {
200  auto root_ids = graph.find_roots( );
201  for( auto start_node_id : root_ids ) {
202  std::unordered_set<daw::node_id_t> visited{ };
203  std::vector<std::pair<std::optional<daw::node_id_t>, daw::node_id_t>>
204  path{ };
205  path.push_back( { std::nullopt, start_node_id } );
206 
207  while( not path.empty( ) ) {
208  auto current_id = path.back( );
209  path.pop_back( );
210  if( visited.count( current_id.second ) > 0 ) {
211  daw::exception::precondition_check(
212  current_id.first.has_value( ),
213  "I assumed that the root would have been first visited" );
214  graph.remove_directed_edge( *current_id.first, current_id.second );
215  continue;
216  }
217  visited.insert( current_id.second );
218  auto current_node = graph.get_node( current_id.second );
219  std::transform(
220  std::begin( current_node.outgoing_edges( ) ),
221  std::end( current_node.outgoing_edges( ) ),
222  std::back_inserter( path ), [parent = current_id.second]( auto id ) {
223  return std::pair<std::optional<daw::node_id_t>, node_id_t>( parent,
224  id );
225  } );
226  }
227  }

25  check_increment = 1,

38  constexpr auto get_flag_value( bool CheckIncrement,
39  bool CheckDecrement,
40  bool CheckDereference ) const noexcept {
41  auto result =
42  static_cast<std::uint8_t>( CheckIncrement ? check_increment : 0 );

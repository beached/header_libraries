22 template<typename IteratorFirst, typename IteratorLast>
23 class checked_iterator_proxy_t {
24 enum flag_t : std::uint8_t {
25 check_increment = 1,
26 check_decrement = 2,
27 check_dereference = 4
28 };
29 IteratorFirst current;
30 IteratorFirst first;
31 IteratorLast last;
32 std::uint8_t flags = 0;
33
34 constexpr bool is_flag_set( flag_t flag ) const noexcept {
35 return 0 != ( flags & flag );
36 }
37
38 constexpr auto get_flag_value( bool CheckIncrement, bool CheckDecrement,
39 bool CheckDereference ) const noexcept {
40 auto result =
41 static_cast<std::uint8_t>( CheckIncrement ? check_increment : 0 );
42 result = static_cast<std::uint8_t>(
43 daw::nsc_or( result, CheckDecrement ) ? check_decrement : 0 );
44 result = static_cast<std::uint8_t>(
45 daw::nsc_or( result, CheckDereference ) ? check_dereference : 0 );
46
47 return result;
48 }

50 public:
51 checked_iterator_proxy_t( IteratorFirst it_first, IteratorLast it_last,
52 bool CheckIncrement = true,
53 bool CheckDecrement = true,
54 bool CheckDereference = true )
55 : current{ it_first }
56 , first{ std::move( it_first ) }
57 , last{ std::move( it_last ) }
58 , flags{ get_flag_value( CheckIncrement, CheckDecrement,
59 CheckDereference ) } {}

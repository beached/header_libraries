36 template<typename ValueType>
37 struct hash_table_item {
38 enum Sentinals : size_t {
39 sentinal_empty, // empty, nothing there
40 sentinal_removed, // has been removed so when looking for a hash just
41 // skip and don't treat as unfound
42 SentinalsSize // must be last. Marker to get number of Sentinals
43 };
44 using value_type = typename daw::traits::root_type_t<ValueType>;
45 size_t hash; // 0 is the sentinel to mark unused
46 value_type value;
47
48 void clear( ) {
49 hash = Sentinals::sentinal_removed;
50 value = value_type{ };
51 }
52
53 bool empty( ) const noexcept {
54 return Sentinals::sentinal_empty == hash;
55 }
56
57 bool removed( ) const noexcept {
58 return Sentinals::sentinal_removed == hash;
59 }
60
61 bool good( ) const noexcept {
62 return hash >= Sentinals::SentinalsSize;
63 }
64
65 explicit operator bool( ) const {
66 return hash >= Sentinals::SentinalsSize;
67 }
68
69 hash_table_item( )
70 : hash{ Sentinals::sentinal_empty }
71 , value{ } {}
72
73 ~hash_table_item( ) = default;
74
75 hash_table_item( hash_table_item const &other )
76 : hash{ ( other.hash == Sentinals::sentinal_removed )
77 ? Sentinals::sentinal_empty
78 : other.hash }
79 , value{ other.value } {}
80
81 hash_table_item( hash_table_item &&other ) noexcept
82 : hash{ std::move( other.hash ) }
83 , value{ std::move( other.value ) } {}
84
85 hash_table_item &operator=( hash_table_item const &rhs ) {
86 if( this != &rhs ) {
87 hash = ( rhs.hash == Sentinals::sentinal_removed )
88 ? Sentinals::sentinal_empty
89 : rhs.hash;
90 hash = rhs.hash;
91 value = rhs.value;
92 }
93 return *this;
94 }
95
96 hash_table_item &operator=( hash_table_item &&rhs ) noexcept {
97 if( this != &rhs ) {
98 hash = daw::exchange( rhs.hash, Sentinals::sentinal_empty );
99 value = std::move( rhs.value );
100 }
101 return *this;
102 }

378 template<typename KeyType>
379 static size_t hash_fn( KeyType &&key ) {
380 static const auto s_hash = []( auto &&k ) {
381 size_t result =
382 ( daw::fnv1a_hash( DAW_FWD( k ) ) %
383 ( max_value<std::size_t> -
384 impl::hash_table_item<value_type>::SentinalsSize ) ) +
385 impl::hash_table_item<value_type>::SentinalsSize; // Guarantee we
386 // cannot be zero
387 return result;
388 };
389 return s_hash( DAW_FWD( key ) );
390 }

393 std::string value = c_str;

397 static size_t scale_hash( size_t hash, size_t table_size ) {
398 // Scale value to capacity using MAD(Multiply-Add-Divide) compression
399 // Use the two largest Prime's that fit in a 64bit unsigned integral
400 daw::exception::daw_throw_on_false(
401 table_size <
402 max_size( ) ); // Table size must be less than max of ptrdiff_t as we
403 // use the value 0 as a sentinel. This should be rare
404 daw::exception::daw_throw_on_false(
405 hash >=
406 impl::hash_table_item<
407 value_type>::SentinalsSize ); // zero is a sentinel for no value
408 static const size_t prime_a = 18446744073709551557u;
409 static const size_t prime_b = 18446744073709551533u;
410 return ( hash * prime_a + prime_b ) % table_size;
411 }

413 static size_t resize_table( values_type &old_table, size_t new_size ) {
414 values_type new_hash_table{ new_size };
415 size_t load = 0;
416 for( auto &&current_item : old_table ) {
417 if( current_item ) {
418 ++load;
419 auto pos = find_item_by_hash( current_item.hash, new_hash_table );
420 daw::exception::daw_throw_on_false( pos != new_hash_table.cend( ) );
421 daw::cswap( *pos, current_item );
422 }
423 }
424 daw::cswap( old_table, new_hash_table );
425 return load;
426 }

428 static auto find_item_by_hash( size_t hash, values_type const &tbl,
429 bool skip_removed = true ) {
430 auto const scaled_hash =
431 static_cast<ptrdiff_t>( scale_hash( hash, tbl.size( ) ) );
432 auto const start_it = std::next( tbl.begin( ), scaled_hash );
433 // loop through all values until an empty spot is found(at end start at
434 // beginning)
435 auto const is_here = [skip_removed, hash]( auto const &item ) {
436 // hash can be empty, removed or have a value.
437 if( hash == item.hash ) {
438 return true;
439 }
440 if( item.empty( ) ) {
441 return true;
442 }
443 // Only option left is that item is removed
444 return !skip_removed;
445 };
446 auto pos = std::find_if( start_it, tbl.cend( ), is_here );
447 if( tbl.cend( ) == pos ) {
448 pos = std::find_if( tbl.begin( ), start_it, is_here );
449 if( start_it == pos ) {
450 pos = tbl.cend( );
451 }
452 }
453 return const_cast<priv_iterator>( pos );
454 }

456 static auto find_item_by_hash( size_t hash, hash_table const &tbl ) {

460 static priv_iterator find_item_by_hash_or_create( size_t hash,
461 hash_table &tbl ) {
462 if( tbl.empty( ) or
463 ( ( tbl.m_load * 100 ) / tbl.m_values.size( ) ) > tbl.m_max_load ) {
464 tbl.grow_table( );
465 }
466 auto pos = find_item_by_hash( hash, tbl.m_values );
467 if( pos == tbl.priv_end( ) ) {
468 pos = find_item_by_hash( hash, tbl.m_values, false );
469 if( pos == tbl.priv_end( ) ) {
470 tbl.grow_table( );
471 pos = find_item_by_hash( hash, tbl );
472 daw::exception::daw_throw_on_true( pos == tbl.priv_end( ) );
473 // This is a WAG. It be wrong but I had to pick something
474 // without evidence to support it.
475 }
476 }
477 if( !pos ) {
478 ++tbl.m_load;
479 pos->hash = hash;
480 }
481 return pos;
482 }

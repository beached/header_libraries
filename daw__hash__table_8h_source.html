36  struct hash_table_item {
37  enum Sentinals : size_t {
38  sentinal_empty, // empty, nothing there
39  sentinal_removed, // has been removed so when looking for a hash just
40  // skip and don't treat as unfound
41  SentinalsSize // must be last. Marker to get number of Sentinals
42  };

44  size_t hash; // 0 is the sentinel to mark unused

405  static size_t scale_hash( size_t hash, size_t table_size ) {
406  // Scale value to capacity using MAD(Multiply-Add-Divide) compression
407  // Use the two largest Prime's that fit in a 64bit unsigned integral
408  daw::exception::daw_throw_on_false(
409  table_size <
410  max_size( ) ); // Table size must be less than max of ptrdiff_t as we
411  // use the value 0 as a sentinel. This should be rare
412  daw::exception::daw_throw_on_false(
413  hash >=
414  impl::hash_table_item<
415  value_type>::SentinalsSize ); // zero is a sentinel for no value
416  static const size_t prime_a = 18446744073709551557u;
417  static const size_t prime_b = 18446744073709551533u;
418  return ( hash * prime_a + prime_b ) % table_size;
419  }

421  static size_t resize_table( values_type &old_table, size_t new_size ) {
422  values_type new_hash_table{ new_size };
423  size_t load = 0;
424  for( auto &&current_item : old_table ) {
425  if( current_item ) {
426  ++load;
427  auto pos = find_item_by_hash( current_item.hash, new_hash_table );
428  daw::exception::daw_throw_on_false( pos != new_hash_table.cend( ) );
429  daw::cswap( *pos, current_item );
430  }
431  }
432  daw::cswap( old_table, new_hash_table );
433  return load;
434  }

436  static auto find_item_by_hash( size_t hash, values_type const &tbl,
437  bool skip_removed = true ) {
438  auto const scaled_hash =
439  static_cast<ptrdiff_t>( scale_hash( hash, tbl.size( ) ) );
440  auto const start_it = std::next( tbl.begin( ), scaled_hash );
441  // loop through all values until an empty spot is found(at end start at
442  // beginning)
443  auto const is_here = [skip_removed, hash]( auto const &item ) {
444  // hash can be empty, removed or have a value.
445  if( hash == item.hash ) {
446  return true;
447  }
448  if( item.empty( ) ) {
449  return true;
450  }
451  // Only option left is that item is removed
452  return !skip_removed;
453  };
454  auto pos = std::find_if( start_it, tbl.cend( ), is_here );
455  if( tbl.cend( ) == pos ) {
456  pos = std::find_if( tbl.begin( ), start_it, is_here );
457  if( start_it == pos ) {
458  pos = tbl.cend( );
459  }
460  }
461  return const_cast<priv_iterator>( pos );
462  }

468  static priv_iterator find_item_by_hash_or_create( size_t hash,
469  hash_table &tbl ) {
470  if( tbl.empty( ) or
471  ( ( tbl.m_load * 100 ) / tbl.m_values.size( ) ) > tbl.m_max_load ) {
472  tbl.grow_table( );
473  }
474  auto pos = find_item_by_hash( hash, tbl.m_values );
475  if( pos == tbl.priv_end( ) ) {
476  pos = find_item_by_hash( hash, tbl.m_values, false );
477  if( pos == tbl.priv_end( ) ) {
478  tbl.grow_table( );
479  pos = find_item_by_hash( hash, tbl );
480  daw::exception::daw_throw_on_true( pos == tbl.priv_end( ) );

482  // without evidence to support it.
483  }
484  }

155 template<typename CharT, typename BinaryPredicate>
156 [[nodiscard]] constexpr CharT const *
157 find_first_of( CharT const *haystack_first, std::size_t haystack_size,
158 CharT const *needle_first, std::size_t needle_size,
159 BinaryPredicate p ) {
160 static_assert(
161 traits::is_binary_predicate_v<
162 BinaryPredicate,
163 typename std::iterator_traits<CharT const *>::value_type>,
164 "BinaryPredicate p does not fullfill the requires of a binary "
165 "predicate concept. See "
166 "http://en.cppreference.com/w/cpp/concept/BinaryPredicate" );
167 if constexpr( sizeof( CharT ) == 1 ) {
168 // On strings of char lets use a bitset to minimize memory and get
169 // O(N+M) instead of O(N*M)
170 auto const needle_array = [&] {
171 auto r = daw::bitset<256>{ };
172 for( std::size_t n = 0; n < needle_size; ++n ) {
173 r.set( static_cast<unsigned char>( needle_first[n] ) );
174 }
175 return r;
176 }( );
177 for( std::size_t hs_pos = 0; hs_pos < haystack_size; ++hs_pos ) {
178 if( needle_array.test(
179 static_cast<unsigned char>( haystack_first[hs_pos] ) ) ) {
180 return std::next( haystack_first,
181 static_cast<std::ptrdiff_t>( hs_pos ) );
182 }
183 }
184 } else {
185 for( std::size_t haystack_pos = 0; haystack_pos < haystack_size;
186 ++haystack_pos ) {
187 for( std::size_t needle_pos = 0; needle_pos < needle_size;
188 ++needle_pos ) {
189 if( p( haystack_first[haystack_pos],
190 needle_first[needle_pos] ) ) {
191 return std::next( haystack_first,
192 static_cast<std::ptrdiff_t>( haystack_pos ) );
193 }
194 }
195 }
196 }
197 return std::next( haystack_first,
198 static_cast<std::ptrdiff_t>( haystack_size ) );
199 }

297 template<typename CharT>
298 [[nodiscard]] constexpr CharT const *
299 search( CharT const *haystack, std::size_t haystack_sz,
300 CharT const *needle, std::size_t needle_sz ) {
301#if not defined( DAW_NO_MEMMEM ) and \
302 ( ( defined( _GNU_SOURCE ) or not defined( _WIN32 ) ) and \
303 defined( DAW_HAS_IF_CONSTEVAL_COMPAT ) )
304 if constexpr( sizeof( CharT ) == 1 ) {
305 DAW_IF_NOT_CONSTEVAL {
306 void const *result =
307 memmem( reinterpret_cast<char const *>( haystack ), haystack_sz,
308 reinterpret_cast<char const *>( needle ), needle_sz );
309 return static_cast<CharT const *>( result );
310 }
311 }
312#endif
313 auto const haystack_last =
314 std::next( haystack, static_cast<std::ptrdiff_t>( haystack_sz ) );
315 auto const needle_last =
316 std::next( needle, static_cast<std::ptrdiff_t>( needle_sz ) );
317 while( true ) {
318 auto haystack_it = haystack;
319 for( auto needle_it = needle;; ++haystack_it, ++needle_it ) {
320 if( needle_it == needle_last ) {
321 return haystack;
322 }
323 if( haystack_it == haystack_last ) {
324 return nullptr;
325 }
326 if( *haystack_it != *needle_it ) {
327 break;
328 }
329 }
330 ++haystack;
331 }
332 }

336 search( basic_string_view<CharT> haystack,
337 std::initializer_list<basic_string_view<CharT>> needles ) {
338 auto const last = daw::data_end( haystack );
339 for( ; not haystack.empty( ); haystack.remove_prefix( ) ) {
340 auto it = haystack.data( );
341
342 for( auto needle : needles ) {
343 if( needle.size( ) > haystack.size( ) ) {}
344 auto s_first = std::data( needle );
345 auto const s_last = daw::data_end( needle );
346 for( auto s_it = s_first;; ++it, ++s_it ) {
347 if( s_it == s_last ) {
348 return haystack.data( );
349 }
350 if( it == last ) {
351 return last;
352 }
353 if( not( *it == *s_it ) ) {
354 break;
355 }
356 }
357 }
358 }
359 return data_end( haystack );

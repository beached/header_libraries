40  template<typename Result>
41  constexpr Result parse_int( daw::sv1::string_view &str ) {
42  intmax_t count = std::numeric_limits<Result>::digits;
43  Result result = 0;
44  bool is_neg = false;
45  if( '-' == str.front( ) ) {
46 
47  daw::exception::precondition_check<invalid_input_exception>(
48  std::numeric_limits<Result>::is_signed );
49 
50  is_neg = true;
51  str.remove_prefix( );
52  }
53  while( count > 0 and !str.empty( ) and is_number( str.front( ) ) ) {
54  result *= static_cast<Result>( 10 );
55  result += static_cast<Result>( str.front( ) - '0' );
56  --count;
57  str.remove_prefix( );
58  }
59 
60  daw::exception::precondition_check<numeric_overflow_exception>(
61  str.empty( ) );
62  if( is_neg ) {
63  result *= static_cast<Result>( -1 );
64  }
65  return result;
66  }

222  template<bool skip_multiple>
223  struct basic_whitespace_splitter {
224  constexpr auto operator( )( daw::sv1::string_view str ) const {
225  struct result_t {
226  typename daw::sv1::string_view::size_type first;
227  typename daw::sv1::string_view::size_type last;
228  };
229 
230  using sv_size_t = typename daw::sv1::string_view::size_type;
231  sv_size_t n = 0;
232  size_t const sz = str.size( );
233 
234  while( n < sz and not_unicode_whitespace( str[n] ) ) {
235  ++n;
236  }
237  if( n == sz ) {
238  return result_t{ daw::sv1::string_view::npos,
239  daw::sv1::string_view::npos };
240  }
241  if( !skip_multiple ) {
242  return result_t{ n, n + 1 };
243  }
244  auto const f = n;
245  while( n < sz and is_unicode_whitespace( str[n] ) ) {
246  ++n;
247  }
248  return result_t{ f, n };
249  }
250  };

41  template<typename Result>
42  constexpr Result parse_int( daw::string_view &str ) {
43  intmax_t count = std::numeric_limits<Result>::digits;
44  Result result = 0;
45  bool is_neg = false;
46  if( '-' == str.front( ) ) {
47 
48  daw::exception::precondition_check<invalid_input_exception>(
49  std::numeric_limits<Result>::is_signed );
50 
51  is_neg = true;
52  str.remove_prefix( );
53  }
54  while( count > 0 and !str.empty( ) and is_number( str.front( ) ) ) {
55  result *= static_cast<Result>( 10 );
56  result += static_cast<Result>( str.front( ) - '0' );
57  --count;
58  str.remove_prefix( );
59  }
60 
61  daw::exception::precondition_check<numeric_overflow_exception>(
62  str.empty( ) );
63  if( is_neg ) {
64  result *= static_cast<Result>( -1 );
65  }
66  return result;
67  }

117  constexpr daw::string_view parse_to_value( daw::string_view str,
118  tag_t<daw::string_view> ) {
119  daw::exception::precondition_check<empty_input_exception>(
120  not str.empty( ) );
121 
122  struct input_too_small_exception {};
123  struct missing_expected_quotes_exception {};
124 
125  daw::exception::precondition_check<input_too_small_exception>(
126  str.size( ) > 2 );
127  daw::exception::precondition_check<missing_expected_quotes_exception>(
128  str.front( ) == '"' );

132  char last_char = str.pop_front( );

224  template<bool skip_multiple>
225  struct basic_whitespace_splitter {
226  constexpr auto operator( )( daw::string_view str ) const {
227  struct result_t {
228  typename daw::string_view::size_type first;
229  typename daw::string_view::size_type last;
230  };
231 
232  using sv_size_t = typename daw::string_view::size_type;
233  sv_size_t n = 0;
234  size_t const sz = str.size( );
235 
236  while( n < sz and not_unicode_whitespace( str[n] ) ) {
237  ++n;
238  }
239  if( n == sz ) {
240  return result_t{ daw::string_view::npos, daw::string_view::npos };
241  }
242  if( !skip_multiple ) {
243  return result_t{ n, n + 1 };
244  }
245  auto const f = n;
246  while( n < sz and is_unicode_whitespace( str[n] ) ) {
247  ++n;
248  }
249  return result_t{ f, n };
250  }
251  };

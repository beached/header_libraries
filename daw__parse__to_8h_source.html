43 template<typename Result>
44 constexpr Result parse_int( daw::string_view &str ) {
45 intmax_t count = std::numeric_limits<Result>::digits;
46 Result result = 0;
47 bool is_neg = false;
48 if( '-' == str.front( ) ) {
49
50 daw::exception::precondition_check<invalid_input_exception>(
51 std::numeric_limits<Result>::is_signed );
52
53 is_neg = true;
54 str.remove_prefix( );
55 }
56 while( count > 0 and !str.empty( ) and is_number( str.front( ) ) ) {
57 result *= static_cast<Result>( 10 );
58 result += static_cast<Result>( str.front( ) - '0' );
59 --count;
60 str.remove_prefix( );
61 }
62
63 daw::exception::precondition_check<numeric_overflow_exception>(
64 str.empty( ) );
65 if( is_neg ) {
66 result *= static_cast<Result>( -1 );
67 }
68 return result;
69 }

119 constexpr daw::string_view parse_to_value( daw::string_view str,
120 tag_t<daw::string_view> ) {
121 daw::exception::precondition_check<empty_input_exception>(
122 not str.empty( ) );
123
124 struct input_too_small_exception {};
125 struct missing_expected_quotes_exception {};
126
127 daw::exception::precondition_check<input_too_small_exception>(
128 str.size( ) > 2 );
129 daw::exception::precondition_check<missing_expected_quotes_exception>(
130 str.front( ) == '"' );
131
132 str.remove_prefix( );
133 auto const first = str.data( );
134 char last_char = str.pop_front( );
135 while( not str.empty( ) and
136 not impl::is_quote( last_char, str.front( ) ) ) {
137 last_char = str.pop_front( );
138 }
139 daw::exception::precondition_check<missing_expected_quotes_exception>(
140 str.data( ) != nullptr and str.front( ) == '"' );
141
142 return daw::string_view( first, str.data( ) );
143 }

203 struct default_splitter {
204 daw::string_view m_delemiter;
205
206 constexpr auto operator( )( daw::string_view str ) const noexcept {
207 using sv_size_t = typename daw::string_view::size_type;
208 struct result_t {
209 sv_size_t first;
210 sv_size_t last;
211 };
212 auto const pos = str.find( m_delemiter );
213 if( pos < str.size( ) ) {
214 return result_t{ pos, pos + m_delemiter.size( ) };
215 }
216 return result_t{ daw::string_view::npos, daw::string_view::npos };
217 }
218 };

223 template<bool skip_multiple>
224 struct basic_whitespace_splitter {
225 constexpr auto operator( )( daw::string_view str ) const {
226 struct result_t {
227 typename daw::string_view::size_type first;
228 typename daw::string_view::size_type last;
229 };
230
231 using sv_size_t = typename daw::string_view::size_type;
232 sv_size_t n = 0;
233 size_t const sz = str.size( );
234
235 while( n < sz and not_unicode_whitespace( str[n] ) ) {
236 ++n;
237 }
238 if( n == sz ) {
239 return result_t{ daw::string_view::npos, daw::string_view::npos };
240 }
241 if( !skip_multiple ) {
242 return result_t{ n, n + 1 };
243 }
244 auto const f = n;
245 while( n < sz and is_unicode_whitespace( str[n] ) ) {
246 ++n;
247 }
248 return result_t{ f, n };
249 }
250 };

22 template<std::size_t Size = sizeof( void * ),
23 std::size_t Align = alignof( std::max_align_t )>
24 struct sbo_storage {
25 static constexpr std::size_t storage_size =
26 Size >= sizeof( void * ) ? Size : sizeof( void * );
27
28 daw::aligned_storage_t<storage_size, Align> data{ };
29
30 enum class engaged_types : std::uint8_t { none, local, allocated };
31
32 engaged_types engaged = engaged_types::none;
33
34 template<typename T>
35 [[nodiscard]] T *get( ) {
36 if constexpr( sizeof( T ) <= storage_size ) {
37 assert( engaged == engaged_types::local );
38 return reinterpret_cast<T *>( &data );
39 } else {
40 assert( engaged == engaged_types::allocated );
41 auto ptr = reinterpret_cast<T **>( &data );
42 return *ptr;
43 }
44 }
45
46 template<typename T, typename AllocatedType = T>
47 [[nodiscard]] T const *get( ) const {
48 if constexpr( sizeof( AllocatedType ) <= storage_size ) {
49 assert( engaged == engaged_types::allocated );
50 return reinterpret_cast<T const *>( &data );
51 } else {
52 assert( engaged == engaged_types::allocated );
53 return *reinterpret_cast<T const **>( &data );
54 }
55 }

57 template<typename T,
58 typename Allocator = std::allocator<std::remove_cv_t<T>>>
59 void deallocate( Allocator const &alloc = Allocator{ } ) {
60 static_assert( not std::is_array_v<T>, "Arrays unsupported" );
61 if constexpr( sizeof( T ) <= storage_size ) {
62 assert( engaged == engaged_types::local );
63 get<T>( )->~T( );
64 } else {
65 assert( engaged == engaged_types::allocated );
66 Allocator allocator = Allocator{ alloc };
67 auto ptr = get<T>( );
68 std::allocator_traits<Allocator>::destroy( allocator, ptr );
69 std::allocator_traits<Allocator>::deallocate( allocator, ptr, 1 );
70 }
71 engaged = engaged_types::none;
72 }

74 template<typename T, typename Allocator = std::allocator<
75 std::remove_cv_t<std::remove_cv_t<T>>>>
76 std::remove_cv_t<T> *allocate( T &&value,
77 Allocator const &alloc = Allocator{ } ) {
78 using base_type = std::remove_cv_t<T>;
79 static_assert( not std::is_array_v<base_type>, "Arrays unsupported" );
80 assert( engaged == engaged_types::none );
81 if constexpr( sizeof( base_type ) <= storage_size ) {
82 if constexpr( std::is_aggregate_v<base_type> ) {
83 auto *result = new( &data ) base_type{ DAW_FWD( value ) };
84 engaged = engaged_types::local;
85 return result;
86 } else {
87 auto *result = new( &data ) base_type( DAW_FWD( value ) );
88 engaged = engaged_types::local;
89 return result;
90 }
91 } else {
92 // Allocate a value via allocator
93 using base_type_ptr = base_type *;
94 Allocator allocator = Allocator{ alloc };
95 base_type_ptr ptr =
96 std::allocator_traits<Allocator>::allocate( allocator, 1 );
97 // Store pointer from allocator in buffer
98 new( &data ) base_type_ptr{ ptr };
99 // Construct new value from passed value at location from allocator
100 std::allocator_traits<Allocator>::construct( allocator, ptr,
101 DAW_FWD( value ) );
102 engaged = engaged_types::allocated;
103 return ptr;
104 }
105 }
106 };

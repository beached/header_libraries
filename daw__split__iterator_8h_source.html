117 constexpr Iterator find_prev( ) noexcept {
118 static_assert( impl::can_decrement<Iterator>,
119 "Supplied Iterator is not Bidirectional" );
120 auto pos = daw::algorithm::find_last_of(
121 m_data.cbegin( ), m_position.cend( ), m_splitter );

131 "Supplied Iterator is not Bidirectional" );

133 return;

137 m_data.cbegin( ),
138 impl::splitter_size{ }( m_splitter ) );
139 m_position.begin( ) = find_prev( );

144 return m_position.end( );

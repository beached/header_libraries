176 template<typename U>
177 [[nodiscard]] constexpr difference_type
178 compare( indexed_iterator<U> const &rhs ) const noexcept {
179 daw::exception::dbg_precondition_check(
180 std::equal_to<>{ }( m_pointer, rhs.m_pointer ) );
181 daw::exception::dbg_precondition_check( m_position >= 0 );
182 daw::exception::dbg_precondition_check( rhs.m_position >= 0 );
183
184 auto const result = m_position - rhs.m_position;
185 if( result < 0 ) {
186 return -1;
187 }
188 if( result > 0 ) {
189 return 1;
190 }
191 return result;
192 }

200 daw::exception::dbg_precondition_check( lhs.m_position >=
201 rhs.m_position );

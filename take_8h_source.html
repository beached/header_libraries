25 struct Take_t {
26 std::size_t how_many = 0;
27
28 template<Range R>
29 [[nodiscard]] constexpr auto operator( )( R &&r ) const {
30 using iter_t = iterator_t<decltype( r )>;
31 auto first = std::begin( r );
32 auto last = std::end( r );
33
34 if constexpr( ForwardIterator<iter_t> ) {
35 auto const range_size = std::distance( first, last );
36 auto take_size =
37 std::min( { range_size, static_cast<std::ptrdiff_t>( how_many ) } );
38 return range_t{ sized_iterator<iter_t>{
39 first, static_cast<std::size_t>( take_size ) },
40 sized_iterator<iter_t>{ last, 0 } };
41 } else {
42 return range_t{ sized_iterator<iter_t>{ first, how_many },
43 sized_iterator<iter_t>{ last, 0 } };
44 }
45 }
46 };

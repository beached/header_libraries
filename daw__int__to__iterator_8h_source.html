20 template<typename T, typename Result = daw::remove_cvref_t<T>>
21 constexpr Result num_digits( T value ) noexcept {
22 static_assert( std::is_integral_v<Result>,
23 "Result must be an integer type" );
24 if( value == 0 ) {
25 return static_cast<Result>( 1 );
26 }
27 Result result = 0;
28 while( value != 0 ) {
29 value /= 10;
30 ++result;

179 template<typename CharT,
180 typename OutputIterator,
181 typename Integer,
182 typename Traits = daw::impl::char_traits<CharT>,
183 std::enable_if_t<
184 daw::all_true_v<
185 not std::is_integral_v<daw::remove_cvref_t<Integer>>,
186 not std::is_same_v<bool, daw::remove_cvref_t<Integer>>,
187 not std::is_floating_point_v<daw::remove_cvref_t<Integer>>,
188 not daw::traits::is_character_v<Integer>>,
189 std::nullptr_t> = nullptr>
190 constexpr OutputIterator
191 to_os_string( OutputIterator it, Integer value, daw::tag_t<int> ) {
192
193 if( value < 0 ) {
194 if( value == ( std::numeric_limits<Integer>::min )( ) ) {
195 constexpr auto m = min_strings::get(
196 CharT{ }, std::integral_constant<size_t, sizeof( Integer )>{ } );
197 return daw::algorithm::copy( m.begin( ), m.end( ), it );
198 }
199 *it = static_cast<CharT>( '-' );
200 ++it;
201 value = -value;
202 }
203 for( auto p10 = pow10<Integer>( whole_log10<uint16_t>( value ) );
204 p10 >= 1;
205 p10 /= static_cast<Integer>( 10 ) ) {
206
207 auto const tmp = value / p10;
208
209 daw::exception::dbg_precondition_check(
210 tmp >= 0 && tmp < 10,
211 "There should only ever be a single digit positive number" );
212
213 *it = Traits::get_char_digit( tmp );
214 ++it;
215
216 value -= tmp * p10;
217 }
218 return it;
219 }

186  constexpr OutputIterator to_os_string( OutputIterator it, Integer value,
187  daw::tag_t<int> ) {
188 
189  if( value < 0 ) {
190  if( value == ( std::numeric_limits<Integer>::min )( ) ) {
191  constexpr auto m = min_strings::get(
192  CharT{ }, std::integral_constant<size_t, sizeof( Integer )>{ } );
193  return daw::algorithm::copy( m.begin( ), m.end( ), it );
194  }

196  ++it;
197  value = -value;
198  }
199  for( auto p10 = pow10<Integer>( whole_log10<uint16_t>( value ) );

209  ++it;

212  }

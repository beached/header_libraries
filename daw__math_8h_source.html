199 // Based on the triple-angle formula: sin 3x = 3 sin x - 4 sin ^3 x
200 template<typename R>
201 [[nodiscard]] constexpr R sin_helper( R x ) noexcept {
202 return x < constants::tol
203 ? x
204 : static_cast<R>( 3.0 ) *
205 ( sin_helper( x / static_cast<R>( 3.0 ) ) ) -
206 static_cast<R>( 4.0 ) *
207 cube( sin_helper( x / static_cast<R>( 3.0 ) ) );
208 }

253 template<typename T, std::enable_if_t<std::is_floating_point_v<T>,
254 std::nullptr_t> = nullptr>
255 [[nodiscard]] constexpr bool
256 nearly_equal( T a, T b,
257 T epsilon = std::numeric_limits<T>::epsilon( ) ) noexcept {
258 if( a != a ) {
259 if( b != b ) {
260 return true; // both NaN
261 }
262 return false;
263 }
264 if( b != b ) {
265 return false;
266 }
267 // Code from http://floating-point-gui.de/errors/comparison/
268 auto absA = daw::math::abs( a );
269 auto absB = daw::math::abs( b );
270 auto diff = daw::math::abs( a - b );
271 if( diff != diff ) {
272 // both numbers where either max or min
273 return true;
274 }
275
276 if( a == b ) { // shortcut, handles infinities
277 return true;
278 }
279 if( a == 0 or b == 0 or diff < std::numeric_limits<T>::min_exponent ) {
280 // a or b is zero or both are extremely close to it
281 // // relative error is less meaningful here
282 return diff < ( epsilon * std::numeric_limits<T>::min_exponent );
283 }
284 // use relative error
285 return diff / ( daw::min )( ( absA + absB ),
286 ( std::numeric_limits<T>::max )( ) ) <
287 epsilon;
288 }

248  template<typename T, std::enable_if_t<std::is_floating_point_v<T>,
249  std::nullptr_t> = nullptr>
250  constexpr bool
251  nearly_equal( T a, T b,
252  T epsilon = std::numeric_limits<T>::epsilon( ) ) noexcept {
253  if( a != a ) {
254  if( b != b ) {
255  return true; // both NaN
256  }
257  return false;
258  }
259  if( b != b ) {
260  return false;
261  }
262  // Code from http://floating-point-gui.de/errors/comparison/
263  auto absA = daw::math::abs( a );
264  auto absB = daw::math::abs( b );
265  auto diff = daw::math::abs( a - b );
266  if( diff != diff ) {
267  // both numbers where either max or min
268  return true;
269  }
270 
271  if( a == b ) { // shortcut, handles infinities
272  return true;
273  }
274  if( a == 0 or b == 0 or diff < std::numeric_limits<T>::min_exponent ) {
275  // a or b is zero or both are extremely close to it
276  // // relative error is less meaningful here
277  return diff < ( epsilon * std::numeric_limits<T>::min_exponent );
278  }
279  // use relative error
280  return diff / ( daw::min )( ( absA + absB ),
281  ( std::numeric_limits<T>::max )( ) ) <
282  epsilon;
283  }

50 value_type pop_bits( std::size_t num_bits = bit_count_v<value_type> ) {
51 daw::exception::dbg_throw_on_true<std::overflow_error>(
52 num_bits > bit_count_v<value_type>,
53 "Attempt to pop more bits than can fit into value" );
54 daw::exception::dbg_throw_on_true<std::runtime_error>(
55 num_bits == 0, "Attempt to pop 0 bits" );
56
57 value_type result = 0;
58
59 if( num_bits > m_left_overs.size( ) ) {
60 if( !m_left_overs.empty( ) ) {
61 num_bits -= m_left_overs.size( );
62 result = m_left_overs.pop_all( );
63 result <<= num_bits;
64 }
65 daw::exception::dbg_throw_on_true<std::out_of_range>(
66 m_first == m_last, "Attempt to iterate past last item" );
67
68 m_left_overs.push_back( *m_first );
69 ++m_first;
70 }
71 result |= m_left_overs.pop_back( num_bits );
72 return result;
73 }
74
75 constexpr void clear_left_overs( ) noexcept {
76 m_left_overs.clear( );
77 }

80 template<typename T, typename BitStream>
81 auto pop_value( BitStream &bs, size_t bits_needed ) {
82 daw::exception::dbg_throw_on_false(
83 bits_needed <= bit_count_t<T>,
84 "Attempt to extra more bits than can fit" );
85
86 using value_type = typename BitStream::value_type;
87 T result = 0;
88 while( bits_needed >= bit_count_v<value_type> ) {
89 result <<= bit_count_v<value_type>;
90 result |= bs.pop_bits( );
91 bits_needed -= bit_count_v<value_type>;
92 }
93 if( bits_needed > 0 ) {
94 result <<= bits_needed;
95 result |= bs.pop_bits( bits_needed );
96 }
97 return result;
98 }

120 template<typename BitStream, typename TestValue>
121 void skip_until( BitStream &bs, TestValue const &v, size_t bit_count ) {
122 daw::exception::dbg_throw_on_true(
123 bit_count > bit_count_v<TestValue>,
124 "Attempt to use more bits than can be put into TestValue" );
125 daw::exception::dbg_throw_on_true( bit_count < 1,
126 "Attempt to pop less than 1 bits" );
127
128 auto cur_val = pop_value<TestValue>( bs );
129 while( bs and cur_val != v ) {
130 cur_val = pop_value<TestValue>( bs );
131 }
132 daw::exception::dbg_throw_on_true( cur_val != v,
133 "Not enough data in stream" );
134 }

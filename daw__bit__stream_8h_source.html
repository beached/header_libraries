93 value_type pop_bits( size_t num_bits = sizeof( value_type ) * 8 ) {
94 daw::exception::dbg_throw_on_true<std::overflow_error>(
95 num_bits > sizeof( value_type ) * 8,
96 "Attempt to pop more bits than can fit into value" );
97 daw::exception::dbg_throw_on_true<std::runtime_error>(
98 num_bits == 0, "Attempt to pop 0 bits" );
99
100 value_type result = 0;
101
102 if( num_bits > m_left_overs.size( ) ) {
103 if( !m_left_overs.empty( ) ) {
104 num_bits -= m_left_overs.size( );
105 result = m_left_overs.pop_all( );
106 result <<= num_bits;
107 }
108 daw::exception::dbg_throw_on_true<std::out_of_range>(
109 m_first == m_last, "Attempt to iterate past last item" );
110
111 m_left_overs.push_back( *m_first );
112 ++m_first;
113 }
114 result |= m_left_overs.pop_back( num_bits );
115 return result;
116 }
117
118 constexpr void clear_left_overs( ) noexcept {
119 m_left_overs.clear( );
120 }

123 template<typename T, typename BitStream>
124 auto pop_value( BitStream &bs, size_t bits_needed ) {
125 daw::exception::dbg_throw_on_false(
126 bits_needed <= sizeof( T ) * 8,
127 "Attempt to extra more bits than can fit" );
128
129 using value_type = typename BitStream::value_type;
130 T result = 0;
131 while( bits_needed >= sizeof( value_type ) * 8 ) {
132 result <<= sizeof( value_type ) * 8;
133 result |= bs.pop_bits( );
134 bits_needed -= sizeof( value_type ) * 8;
135 }
136 if( bits_needed > 0 ) {
137 result <<= bits_needed;
138 result |= bs.pop_bits( bits_needed );
139 }
140 return result;
141 }

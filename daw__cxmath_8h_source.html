244  constexpr fp_classes fp_classify_impl( daw::UInt32 fint ) {
245  fint &= 0x7FFF'FFFFUL;
246  if( fint == 0 ) {
247  return fp_classes::zero;
248  }
249  if( fint < 0x80'0000UL ) {
250  return fp_classes::subnormal;
251  }
252  if( fint > 0x7F80'0000UL ) {
253  return fp_classes::nan;
254  }
255  if( fint == 0x7F80'0000UL ) {
256  return fp_classes::infinity;
257  }
258  return fp_classes::normal;
259  }

261  constexpr fp_classes fp_classify_impl( daw::UInt64 dint ) {
262  auto lx = static_cast<std::uint32_t>( dint & 0x0000'0000'FFFF'FFFFULL );
263  auto hx = static_cast<std::uint32_t>( dint >> 32U );
264  lx |= hx & 0x000F'FFFFUL;
265  hx &= 0x7FF0'0000UL;
266  if( ( hx | lx ) == 0 ) {
267  return fp_classes::zero;
268  }
269  if( hx == 0 ) {
270  return fp_classes::subnormal;
271  }
272  if( hx == 0x7FF0'0000UL ) {
273  if( lx != 0 ) {
274  return fp_classes::nan;
275  }
276  return fp_classes::infinity;
277  }
278  return fp_classes::normal;
279  }

292  [[nodiscard]] constexpr Float pow2_impl2( intmax_t exp ) noexcept {
293  bool is_neg = exp < 0;
294  exp = is_neg ? -exp : exp;
295  auto const max_shft = ( daw::min )(
296  static_cast<size_t>( daw::numeric_limits<Float>::max_exponent10 ),
297  ( sizeof( size_t ) * 8ULL ) );
298  Float result = 1.0;
299 
300  while( static_cast<size_t>( exp ) >= max_shft ) {
301  result *= static_cast<Float>( ( daw::numeric_limits<size_t>::max )( ) );
302  exp -= max_shft;
303  }
304  if( exp > 0 ) {
305  result *= static_cast<Float>( 1ULL << static_cast<size_t>( exp ) );
306  }
307  if( is_neg && result != 0.0 ) {
308  result = static_cast<Float>( 1.0 ) / result;
309  }
310  return result;
311  }

444  // Adapted from Hackers delight 11-3. Do a binary decomposition of n
445  template<typename Arithmetic, typename Unsigned>
446  constexpr Arithmetic pow_n( Arithmetic x, Unsigned n ) {
447  static_assert( daw::is_arithmetic_v<Arithmetic> );
448  static_assert( daw::is_unsigned_v<Unsigned> );
449  auto result = Arithmetic{ 1 };
450  auto p = x;
451  while( true ) {
452  if( n & 1 ) {
453  result = p * result;
454  }
455  n >>= 1;
456  if( n == 0 ) {
457  return result;
458  }
459  p *= p;
460  }
461  }

555  [[nodiscard]] DAW_ATTRIB_INLINE constexpr std::uint32_t
556  count_trailing_zeros_cx32( std::uint32_t v ) noexcept {
557  if( v == 0 ) {
558  return 32U;
559  }
560  std::uint32_t c = 32U;
561  v &= static_cast<std::uint32_t>( -static_cast<std::int32_t>( v ) );
562  if( v ) {
563  c--;
564  }
565  if( v & 0x0000'FFFFU ) {
566  c -= 16;
567  }
568  if( v & 0x00FF'00FFU ) {
569  c -= 8;
570  }
571  if( v & 0x0F0F'0F0FU ) {
572  c -= 4;
573  }
574  if( v & 0x3333'3333U ) {
575  c -= 2;
576  }
577  if( v & 0x5555'5555U ) {
578  c -= 1;
579  }
580  return c;
581  }

888  [[nodiscard]] constexpr float_parts_t bits( float const f ) noexcept {
889 #if defined( DAW_CX_BIT_CAST )
890  static_assert( sizeof( float ) == 4U );
891  return float_parts_t( DAW_BIT_CAST( std::uint32_t, f ), f );
892 #else
893  // Once c++20 use bit_cast
894  if( f == 0.0f ) {
895  return { 0, f }; // also matches -0.0f and gives wrong result
896  } else if( f > ( daw::numeric_limits<float>::max )( ) ) {
897  // infinity
898  return { 0x7f80'0000, f };
899  } else if( f < -( daw::numeric_limits<float>::max )( ) ) {
900  // negative infinity
901  return { 0xff800000, f };
902  } else if( f != f ) {
903  // NaN
904  return { 0x7fc0'0000, f };
905  }
906  bool sign = f < 0.0f;
907  float abs_f = sign ? -f : f;
908  int32_t exponent = 254;
909 
910  while( abs_f < 0x1p87f ) {
911  abs_f *= 0x1p41f;
912  exponent -= 41;
913  }
914 
915  auto const a = static_cast<std::uint64_t>( abs_f * 0x1p-64f );
916  auto lz = static_cast<int32_t>( count_leading_zeroes( a ) );
917  exponent -= lz;
918 
919  if( exponent <= 0 ) {
920  exponent = 0;
921  lz = 8 - 1;
922  }
923 
924  auto significand =
925  static_cast<std::uint32_t>( ( a << ( lz + 1 ) ) >> ( 64 - 23 ) ); // [3]
926  return { ( static_cast<std::uint32_t>( sign ? 1U : 0U ) << 31U ) |
927  ( static_cast<std::uint32_t>( exponent ) << 23U ) |
928  significand,
929  f };
930 #endif

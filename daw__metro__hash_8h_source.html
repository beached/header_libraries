20  constexpr Unsigned as_le_uint( char const *ptr ) noexcept {
21  static_assert( std::is_integral_v<Unsigned> );
22  static_assert( std::is_unsigned_v<Unsigned> );
23  Unsigned result = 0;
24  daw::algorithm::do_n_arg<sizeof( Unsigned )>( [&]( size_t n ) {
25  // This is needed, the compiler must have it's unsigned char
26  auto const tmp = static_cast<unsigned char>( ptr[n] );
27  result = static_cast<Unsigned>( result | static_cast<Unsigned>( tmp )
28  << ( n * 8U ) );
29  } );

31  }

33  template<size_t BitCount, typename Unsigned>
34  constexpr Unsigned rotr( Unsigned value ) noexcept {
35  static_assert( std::is_integral_v<Unsigned> );
36  static_assert( std::is_unsigned_v<Unsigned> );
37  static_assert( BitCount < sizeof( Unsigned ) * 8U );
38  static_assert( BitCount > 0 );
39  constexpr size_t size_bits = sizeof( Unsigned ) * 8U;
40  return static_cast<Unsigned>(
41  static_cast<Unsigned>( value >> BitCount ) |
42  static_cast<Unsigned>( value << ( size_bits - BitCount ) ) );
43  }

50  constexpr uint64_t hash64( daw::view<char const *> buff, uint64_t seed ) {
51  constexpr uint64_t k0 = 0xd6d0'18f5;
52  constexpr uint64_t k1 = 0xa2aa'033b;
53  constexpr uint64_t k2 = 0x6299'2fc1;
54  constexpr uint64_t k3 = 0x30bc'5b29;
55 
56  uint64_t hash = ( seed + k2 ) * k0;
57  if( buff.size( ) >= 32U ) {
58  uint64_t v[4]{ hash, hash, hash, hash };
59

61  v[0] += metro_impl::as_le_uint<uint64_t>( buff.data( ) ) * k0;
62  v[0] = metro_impl::rotr<29U>( v[0] ) + v[2];
63  v[1] += metro_impl::as_le_uint<uint64_t>( buff.data( ) + 8 ) * k1;
64  v[1] = metro_impl::rotr<29U>( v[1] ) + v[3];
65  v[2] += metro_impl::as_le_uint<uint64_t>( buff.data( ) + 16 ) * k2;
66  v[2] = metro_impl::rotr<29U>( v[2] ) + v[0];
67  v[3] += metro_impl::as_le_uint<uint64_t>( buff.data( ) + 24 ) * k3;
68  v[3] = metro_impl::rotr<29U>( v[3] ) + v[1];
69  buff.remove_prefix( 32U );
70  } while( buff.size( ) >= 32 );
71 
72  v[2] ^= metro_impl::rotr<37U>( ( ( v[0] + v[3] ) * k0 ) + v[1] ) * k1;
73  v[3] ^= metro_impl::rotr<37U>( ( ( v[1] + v[2] ) * k1 ) + v[0] ) * k0;
74  v[0] ^= metro_impl::rotr<37U>( ( ( v[0] + v[2] ) * k0 ) + v[3] ) * k1;
75  v[1] ^= metro_impl::rotr<37U>( ( ( v[1] + v[3] ) * k1 ) + v[2] ) * k0;
76  hash += v[0] ^ v[1];
77  }
78 
79  if( buff.size( ) >= 16 ) {

81  hash + ( metro_impl::as_le_uint<uint64_t>( buff.data( ) ) * k2 );
82  v0 = metro_impl::rotr<29U>( v0 ) * k3;
83  uint64_t v1 =
84  hash + ( metro_impl::as_le_uint<uint64_t>( buff.data( ) + 8 ) * k2 );
85  v1 = metro_impl::rotr<29U>( v1 ) * k3;

87  v1 ^= metro_impl::rotr<21U>( v1 * k3 ) + v0;
88  hash += v1;
89  buff.remove_prefix( 16U );
90  }
91

93  hash += metro_impl::as_le_uint<uint64_t>( buff.data( ) ) * k3;
94  hash ^= metro_impl::rotr<55U>( hash ) * k1;
95  buff.remove_prefix( 8U );
96  }

100  metro_impl::as_le_uint<uint32_t>( buff.data( ) ) ) ) *
101  k3;
102  hash ^= metro_impl::rotr<26U>( hash ) * k1;
103  buff.remove_prefix( 4U );
104  }
105 
106  if( buff.size( ) >= 2 ) {
107  hash += static_cast<uint64_t>(
108  metro_impl::as_le_uint<uint16_t>( buff.data( ) ) ) *
109  k3;
110  hash ^= metro_impl::rotr<48U>( hash ) * k1;
111  buff.remove_prefix( 2U );
112  }
113

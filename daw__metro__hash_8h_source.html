21 template<typename Unsigned>
22 constexpr Unsigned as_le_uint( char const *ptr ) noexcept {
23 static_assert( std::is_integral_v<Unsigned> );
24 static_assert( std::is_unsigned_v<Unsigned> );
25 Unsigned result = 0;
26 daw::algorithm::do_n_arg<sizeof( Unsigned )>( [&]( size_t n ) {
27 // This is needed, the compiler must have it's unsigned char
28 DAW_UNSAFE_BUFFER_FUNC_START
29 auto const tmp = static_cast<unsigned char>( ptr[n] );
30 DAW_UNSAFE_BUFFER_FUNC_STOP
31 result = static_cast<Unsigned>( result | static_cast<Unsigned>( tmp )
32 << (n * bit_count_v<char>));
33 } );
34 return result;
35 }
36
37 template<size_t BitCount, typename Unsigned>
38 constexpr Unsigned rotr( Unsigned value ) noexcept {
39 static_assert( std::is_integral_v<Unsigned> );
40 static_assert( std::is_unsigned_v<Unsigned> );
41 static_assert( BitCount < bit_count_v<Unsigned> );
42 static_assert( BitCount > 0 );
43 constexpr size_t size_bits = bit_count_v<Unsigned>;
44 return static_cast<Unsigned>(
45 static_cast<Unsigned>( value >> BitCount ) |
46 static_cast<Unsigned>( value << ( size_bits - BitCount ) ) );
47 }

52 // An implementation of MetroHash64
53 // https://github.com/jandrewrogers/MetroHash
54 constexpr uint64_t hash64( daw::view<char const *> buff, uint64_t seed ) {
55 constexpr uint64_t k0 = 0xd6d0'18f5;
56 constexpr uint64_t k1 = 0xa2aa'033b;
57 constexpr uint64_t k2 = 0x6299'2fc1;
58 constexpr uint64_t k3 = 0x30bc'5b29;
59
60 uint64_t hash = ( seed + k2 ) * k0;
61 if( buff.size( ) >= 32U ) {
62 uint64_t v[4]{ hash, hash, hash, hash };
63
64 do {
65 v[0] += metro_impl::as_le_uint<uint64_t>( buff.data( ) ) * k0;
66 v[0] = metro_impl::rotr<29U>( v[0] ) + v[2];
67 v[1] += metro_impl::as_le_uint<uint64_t>( buff.data( ) + 8 ) * k1;
68 v[1] = metro_impl::rotr<29U>( v[1] ) + v[3];
69 v[2] += metro_impl::as_le_uint<uint64_t>( buff.data( ) + 16 ) * k2;
70 v[2] = metro_impl::rotr<29U>( v[2] ) + v[0];
71 v[3] += metro_impl::as_le_uint<uint64_t>( buff.data( ) + 24 ) * k3;
72 v[3] = metro_impl::rotr<29U>( v[3] ) + v[1];
73 buff.remove_prefix( 32U );
74 } while( buff.size( ) >= 32 );
75
76 v[2] ^= metro_impl::rotr<37U>( ( ( v[0] + v[3] ) * k0 ) + v[1] ) * k1;
77 v[3] ^= metro_impl::rotr<37U>( ( ( v[1] + v[2] ) * k1 ) + v[0] ) * k0;
78 v[0] ^= metro_impl::rotr<37U>( ( ( v[0] + v[2] ) * k0 ) + v[3] ) * k1;
79 v[1] ^= metro_impl::rotr<37U>( ( ( v[1] + v[3] ) * k1 ) + v[2] ) * k0;
80 hash += v[0] ^ v[1];
81 }
82
83 if( buff.size( ) >= 16 ) {
84 uint64_t v0 =
85 hash + ( metro_impl::as_le_uint<uint64_t>( buff.data( ) ) * k2 );
86 v0 = metro_impl::rotr<29U>( v0 ) * k3;
87 uint64_t v1 =
88 hash + ( metro_impl::as_le_uint<uint64_t>( buff.data( ) + 8 ) * k2 );
89 v1 = metro_impl::rotr<29U>( v1 ) * k3;
90 v0 ^= metro_impl::rotr<21U>( v0 * k0 ) + v1;
91 v1 ^= metro_impl::rotr<21U>( v1 * k3 ) + v0;
92 hash += v1;
93 buff.remove_prefix( 16U );
94 }
95
96 if( buff.size( ) >= 8 ) {
97 hash += metro_impl::as_le_uint<std::uint64_t>( buff.data( ) ) * k3;
98 hash ^= metro_impl::rotr<55U>( hash ) * k1;
99 buff.remove_prefix( 8U );
100 }
101
102 if( buff.size( ) >= 4 ) {
103 hash += static_cast<uint64_t>( static_cast<uint64_t>(
104 metro_impl::as_le_uint<uint32_t>( buff.data( ) ) ) ) *
105 k3;
106 hash ^= metro_impl::rotr<26U>( hash ) * k1;
107 buff.remove_prefix( 4U );
108 }
109
110 if( buff.size( ) >= 2 ) {
111 hash += static_cast<uint64_t>(
112 metro_impl::as_le_uint<uint16_t>( buff.data( ) ) ) *
113 k3;
114 hash ^= metro_impl::rotr<48U>( hash ) * k1;
115 buff.remove_prefix( 2U );
116 }
117
118 if( not buff.empty( ) ) {
119 hash += static_cast<uint64_t>( buff.front( ) ) * k3;
120 hash ^= metro_impl::rotr<37U>( hash ) * k1;
121 buff.remove_prefix( );
122 }
123
124 hash ^= metro_impl::rotr<28U>( hash );
125 hash *= k0;
126 hash ^= metro_impl::rotr<29U>( hash );
127
128 return hash;
129 }

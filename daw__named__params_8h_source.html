54 template<Options Option, typename... Opts>
55 requires( Options<Opts...> ) constexpr auto get_opt( Option &&def_value,
56 Opts &&...vals ) {
57 if constexpr( sizeof...( Opts ) == 0 ) {
58 // No replacements found
59 return DAW_FWD( def_value ).value;
60 } else if constexpr( std::is_same_v<std::remove_cvref_t<Option>,
61 std::remove_cvref_t<
62 daw::traits::first_type<Opts...>>> ) {
63 // Found a type match, use it for return value
64 constexpr auto first_val = []( auto &&v, auto &&... ) {
65 return DAW_FWD( v );
66 };
67 return first_val( DAW_FWD( vals )... ).value;
68 } else {
69 // Pop off first option and try next
70 constexpr auto pop_front = []( auto &&def, auto &&, auto &&...vs ) {
71 return get_opt<Option>( DAW_FWD( def ), DAW_FWD( vs )... );
72 };
73 return pop_front( DAW_FWD( def_value ), DAW_FWD( vals )... );

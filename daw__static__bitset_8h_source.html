88  template<size_t BitWidth, typename value_t>
89  constexpr value_t get_diff_mask( ) noexcept {
90  size_t bw = BitWidth;
91  if( bw > bsizeof<value_t> ) {
92  bw %= bsizeof<value_t>;

210  explicit constexpr static_bitset( daw::string_view binary_sv ) {
211  daw::exception::precondition_check<std::overflow_error>(
212  binary_sv.size( ) <= BitWidth );
213 
214  size_t cur_idx = 0U;
215  while( !binary_sv.empty( ) and cur_idx < m_element_capacity ) {
216  value_t tmp = 0U;
217  value_t cur_val = 1U;
218  for( size_t n = 0U; n < daw::bsizeof<value_t> and !binary_sv.empty( );
219  ++n ) {
220  switch( binary_sv.pop_back( ) ) {
221  case '1':
222  tmp |= cur_val;
223  break;
224  case '0':
225  break;
226  default:
227  DAW_THROW_OR_TERMINATE( std::invalid_argument, "binary_sv" );
228  }
229  cur_val *= 2ULL;
230  }
231  m_data[cur_idx] = tmp;
232  ++cur_idx;
233  }
234  }

284  std::string to_string( bitset_impl::fmt_binary_t = fmt_binary ) const {
285  std::string result( BitWidth, '0' );
286  size_t pos = m_element_capacity - 1;
287  for( size_t index = 0; index < m_data.size( ); ++index ) {
288  auto value = m_data[index];
289  if( !value ) {
290  pos -= daw::bsizeof<value_t>;
291  continue;
292  }
293  while( value ) {
294  auto tmp = value & 1U;
295  if( tmp == 1 ) {

463  constexpr static_bitset &operator<<=( size_t bits ) noexcept {
464  if( bits >= BitWidth ) {
465  clear( );
466  return *this;
467  }
468  if( bits < bsizeof<value_t> ) {
469  value_t carry = 0U;
470  for( size_t n = 0; n < m_data.size( ); ++n ) {
471  value_t next_carry = ( m_data[n] & daw::mask_lsb<value_t>( bits ) ) >>
472  ( bsizeof<value_t> - bits );
473  m_data[n] <<= bits;
474  m_data[n] |= carry;
475  carry = next_carry;
476  }
477  return *this;
478  }
479  if( bits == bsizeof<value_t> ) {
480  for( size_t n = m_data.size( ); n > 0; --n ) {
481  auto pos = n - 1;
482  m_data[pos] = m_data[pos - 1];
483  }
484  m_data[0] = 0U;
485  return *this;
486  }
487  while( bits >= bsizeof<value_t> ) {
488  operator<<=( bsizeof<value_t> );
489  bits -= bsizeof<value_t>;
490  }
491  return operator<<=( bits );
492  }

494  constexpr static_bitset &operator>>=( size_t bits ) noexcept {
495  if( bits >= BitWidth ) {
496  clear( );
497  return *this;
498  }
499  if( bits == 0U ) {
500  return *this;
501  }
502  if( bits < bsizeof<value_t> ) {
503  value_t carry = 0U;
504  for( size_t n = m_data.size( ); n > 0; --n ) {
505  auto const pos = n - 1;
506  value_t next_carry = ( m_data[pos] & daw::mask_msb<value_t>( bits ) )
507  << ( bsizeof<value_t> - bits );
508  m_data[pos] >>= bits;
509  m_data[pos] |= carry;
510  carry = next_carry;
511  }

515  for( size_t n = 0; n < m_data.size( ); ++n ) {
516  m_data[n] = m_data[n + 1];
517  }

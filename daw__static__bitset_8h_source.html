35 constexpr auto low_part( uintmax_t value ) noexcept {
36 using value_t = half_max_t<uintmax_t>;
37 auto const msb_mask =
38 daw::mask_msb<uintmax_t>( bsizeof<uintmax_t> - bsizeof<value_t> );
39
40 return static_cast<value_t>( value bitand msb_mask );
41 }

43 constexpr auto high_part( uintmax_t value ) noexcept {
44 using value_t = half_max_t<uintmax_t>;
45 return static_cast<value_t>( value >> bsizeof<value_t> );

48 constexpr auto overflow( uintmax_t &carry ) noexcept {
49 using value_t = half_max_t<uintmax_t>;
50 auto result = low_part( carry );

76 template<size_t BitWidth, typename value_t>
77 constexpr size_t get_elements_needed( ) noexcept {
78 static_assert( BitWidth > 0 );
79 if( BitWidth <= bsizeof<value_t> ) {
80 return 1U;
81 }
82 size_t result = BitWidth / bsizeof<value_t>;
83 if( BitWidth % bsizeof<value_t> != 0 ) {
84 ++result;
85 }
86 return result;
87 }

89 template<size_t BitWidth, typename value_t>
90 constexpr value_t get_diff_mask( ) noexcept {
91 size_t bw = BitWidth;
92 if( bw > bsizeof<value_t> ) {
93 bw %= bsizeof<value_t>;
94 } else if( bw < bsizeof<value_t> ) {
95 bw = bsizeof<value_t> - bw;
96 }
97 if( bw == 0 ) {
98 return max_value<value_t>;
99 }
100 return daw::mask_msb<value_t>( bw );
101 }

211 explicit constexpr static_bitset( daw::string_view binary_sv ) {
212 daw::exception::precondition_check<std::overflow_error>(
213 binary_sv.size( ) <= BitWidth );
214
215 size_t cur_idx = 0U;
216 while( !binary_sv.empty( ) and cur_idx < m_element_capacity ) {
217 value_t tmp = 0U;
218 value_t cur_val = 1U;
219 for( size_t n = 0U; n < daw::bsizeof<value_t> and !binary_sv.empty( );
220 ++n ) {
221 switch( binary_sv.pop_back( ) ) {
222 case '1':
223 tmp |= cur_val;
224 break;
225 case '0':
226 break;
227 default:
228 DAW_THROW_OR_TERMINATE( std::invalid_argument, "binary_sv" );
229 }
230 cur_val *= 2ULL;
231 }
232 m_data[cur_idx] = tmp;
233 ++cur_idx;
234 }
235 }

285 std::string to_string( bitset_impl::fmt_binary_t = fmt_binary ) const {
286 std::string result( BitWidth, '0' );
287 size_t pos = m_element_capacity - 1;
288 for( size_t index = 0; index < m_data.size( ); ++index ) {
289 auto value = m_data[index];
290 if( !value ) {
291 pos -= daw::bsizeof<value_t>;
292 continue;
293 }
294 while( value ) {
295 auto tmp = value & 1U;
296 if( tmp == 1 ) {
297 result[pos] = '1';
298 }
299 value >>= 1U;
300 --pos;
301 }
302 }
303 return result;
304 }
305
306 std::string to_string( bitset_impl::fmt_hex_t ) const {
307 std::string result( m_data.size( ) * sizeof( value_t ) * 2U, '0' );
308 size_t pos = m_data.size( ) * sizeof( value_t ) * 2U - 1U;
309 for( size_t index = 0; index < m_data.size( ); ++index ) {
310 auto value = m_data[index];
311 if( !value ) {
312 pos -= sizeof( value_t ) * 2U;
313 continue;
314 }
315 for( size_t bpos = 0; bpos < sizeof( value_t ); ++bpos ) {
316 uint8_t const tmp = static_cast<uint8_t>( value & 0xFF );
317 value >>= 8U;
318 result[pos--] = bitset_impl::get_low_nibble( tmp );
319 result[pos--] = bitset_impl::get_high_nibble( tmp );
320 }
321 }
322 return result;

464 constexpr static_bitset &operator<<=( size_t bits ) noexcept {
465 if( bits >= BitWidth ) {
466 clear( );
467 return *this;
468 }
469 if( bits < bsizeof<value_t> ) {
470 value_t carry = 0U;
471 for( size_t n = 0; n < m_data.size( ); ++n ) {
472 value_t next_carry = ( m_data[n] & daw::mask_lsb<value_t>( bits ) ) >>
473 ( bsizeof<value_t> - bits );
474 m_data[n] <<= bits;
475 m_data[n] |= carry;
476 carry = next_carry;
477 }
478 return *this;
479 }
480 if( bits == bsizeof<value_t> ) {
481 for( size_t n = m_data.size( ); n > 0; --n ) {
482 auto pos = n - 1;
483 m_data[pos] = m_data[pos - 1];
484 }
485 m_data[0] = 0U;
486 return *this;
487 }
488 while( bits >= bsizeof<value_t> ) {
489 operator<<=( bsizeof<value_t> );
490 bits -= bsizeof<value_t>;
491 }
492 return operator<<=( bits );
493 }
494
495 constexpr static_bitset &operator>>=( size_t bits ) noexcept {
496 if( bits >= BitWidth ) {
497 clear( );
498 return *this;
499 }
500 if( bits == 0U ) {
501 return *this;
502 }
503 if( bits < bsizeof<value_t> ) {
504 value_t carry = 0U;
505 for( size_t n = m_data.size( ); n > 0; --n ) {
506 auto const pos = n - 1;
507 value_t next_carry = ( m_data[pos] & daw::mask_msb<value_t>( bits ) )
508 << ( bsizeof<value_t> - bits );
509 m_data[pos] >>= bits;
510 m_data[pos] |= carry;
511 carry = next_carry;
512 }
513 return *this;
514 }
515 if( bits == bsizeof<value_t> ) {
516 for( size_t n = 0; n < m_data.size( ); ++n ) {
517 m_data[n] = m_data[n + 1];
518 }
519 m_data.back( ) = 0U;
520 return *this;
521 }
522 while( bits >= bsizeof<value_t> ) {
523 operator>>=( bsizeof<value_t> );
524 bits -= bsizeof<value_t>;
525 }
526 return operator>>=( bits );
527 }

441 template<typename ExceptionType = AssertException, typename Predicate,
442 typename Message, typename Container>
443 constexpr void assert_all_false( Container &&container,
444 Message &&assert_message,
445 Predicate &&predicate ) {
446 auto first = std::begin( container );
447 auto last = std::end( container );
448 while( first != last ) {
449 if( predicate( *first ) ) {
450 daw_throw<ExceptionType>( DAW_FWD( assert_message ) );
451 }
452 ++first;
453 }

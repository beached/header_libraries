24namespace daw {
25 template<typename CharT = char>
26 DAW_ATTRIB_NOINLINE std::optional<std::basic_string<CharT>>
27 read_file( std::string const &path ) noexcept {
28 auto const fsize = std::filesystem::file_size( path );
29 if( fsize == 0 ) {
30 return std::basic_string<CharT>{ };
31 }
32 if( static_cast<std::uintmax_t>( std::basic_string<CharT>{ }.max_size( ) ) <
33 fsize ) {
34 // File is too big to fit into string(WIN32)
35 return std::nullopt;
36 }
37 auto result =
38 std::basic_string<CharT>( static_cast<std::size_t>( fsize ), CharT{ } );
39 auto *f = fopen( path.c_str( ), "rb" );
40 if( not f ) {
41 return std::nullopt;
42 }
43 auto num_read = fread( result.data( ), sizeof( CharT ), result.size( ), f );
44 if( num_read != ( result.size( ) / sizeof( CharT ) ) ) {
45 return std::nullopt;
46 }
47 return result;
48 }

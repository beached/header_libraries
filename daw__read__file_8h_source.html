25  DAW_ATTRIB_NOINLINE std::optional<std::basic_string<CharT>>
26  read_file( std::string const &path ) noexcept {
27  auto const fsize = std::filesystem::file_size( path );
28  if( fsize == 0 ) {
29  return std::basic_string<CharT>{ };
30  }
31  auto result = std::basic_string<CharT>( fsize, CharT{ } );
32  auto *f = fopen( path.c_str( ), "rb" );
33  if( not f ) {
34  return std::nullopt;

36  auto num_read = fread( result.data( ), sizeof( CharT ), result.size( ), f );
37  if( num_read != ( result.size( ) / sizeof( CharT ) ) ) {

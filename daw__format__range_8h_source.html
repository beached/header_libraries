66 template<class ParseContext>
67 constexpr typename ParseContext::iterator
68 parse( ParseContext &ctx DAW_LIFETIME_BOUND ) {
69 auto ctx_b = std::begin( ctx );
70 auto const ctx_sz = static_cast<std::size_t>( std::distance( std::begin( ctx ), std::end( ctx ) ) );
71 auto f = daw::basic_string_view( ctx_b, ctx_sz );
72 if( ctx.begin( ) == ctx.end( ) or *ctx.begin( ) == '}' ) {
73 flags = "{}";
74 return ctx.begin( );
75 }
76 int brace_count = 1;
77 bool in_quote = false;
78 auto const *first = f.data( );
79 auto const *last = first;
80 while( not f.empty( ) ) {
81 switch( f.front( ) ) {
82 case '"':
83 in_quote = not in_quote;
84 break;
85 case '{':
86 if( not in_quote ) {
87 ++brace_count;
88 }
89 break;
90 case '}':
91 if( not in_quote ) {
92 --brace_count;
93 assert( brace_count >= 0 );
94 if( brace_count == 0 ) {
95 last = f.data( ) + 1;
96 f.remove_prefix( f.size( ) - 1 );
97 }
98 }
99 break;
100 default:
101 break;
102 }
103 f.remove_prefix( );
104 }
105 auto const sz = std::distance( first, last );
106 flags = daw::string_view( first, static_cast<std::size_t>( sz ) );
107 flags.expand_prefix( 2 );
108 assert( flags.front( ) == '{' );
109 return std::next( ctx.begin( ), sz - 1 );
110 // return ctx.end( ) - 1;
111 }

113 template<typename Ctx>
114 constexpr typename Ctx::iterator format( daw::fmt_range<R, CharT> c,
115 Ctx &ctx ) const {
116 using value_t = daw::range_value_t<R>;
117 if constexpr( std::is_same_v<daw::formatter_impl::DefaultCharT,
118 value_t> ) {
119 // format as string
120 auto out = std::copy(
121 std::begin( c.container ), std::end( c.container ), ctx.out( ) );
122 return out;
123 } else {
124 auto out = ctx.out( );
125 out = std::copy( std::data( c.Left ), daw::data_end( c.Left ), out );
126 bool is_first = true;
127 for( auto const &v : c.container ) {
128 if( is_first ) {
129 is_first = false;
130 } else {
131 out = std::copy(
132 std::data( c.Separator ), daw::data_end( c.Separator ), out );
133 }
134 if constexpr( daw::Range<daw::range_value_t<R>> ) {
135 auto t = daw::fmt_range( v );
136 out = std::vformat_to(
137 out, std::string_view( flags ), std::make_format_args( t ) );
138 } else if constexpr( daw::is_tuple_like_v<daw::range_value_t<R>> ) {
139 auto t = daw::fmt_tuple( v );
140 out = std::vformat_to(
141 out, std::string_view( flags ), std::make_format_args( t ) );
142 } else {
143 out = std::vformat_to(
144 out, std::string_view( flags ), std::make_format_args( v ) );
145 }
146 }
147 out = std::copy( std::data( c.Right ), daw::data_end( c.Right ), out );
148 return out;
149 }
150 }

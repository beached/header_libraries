62  template<class ParseContext>
63  constexpr typename ParseContext::iterator parse( ParseContext &ctx ) {
64  auto f = daw::string_view( ctx.begin( ), ctx.end( ) );
65  if( ctx.begin( ) == ctx.end( ) or *ctx.begin( ) == '}' ) {
66  flags = "{}";
67  return ctx.begin( );
68  }
69  int brace_count = 1;
70  bool in_quote = false;
71  auto const *first = f.data( );
72  auto const *last = first;
73  while( not f.empty( ) ) {
74  switch( f.front( ) ) {
75  case '"':
76  in_quote = not in_quote;
77  break;
78  case '{':
79  if( not in_quote ) {
80  ++brace_count;
81  }
82  break;
83  case '}':
84  if( not in_quote ) {
85  --brace_count;
86  assert( brace_count >= 0 );
87  if( brace_count == 0 ) {
88  last = f.data( ) + 1;
89  f.remove_prefix( f.size( ) - 1 );
90  }
91  }
92  break;
93  default:
94  break;
95  }
96  f.remove_prefix( );
97  }
98  auto const sz = std::distance( first, last );
99  flags = daw::string_view( first, static_cast<std::size_t>( sz ) );
100  flags.expand_prefix( 2 );
101  assert( flags.front( ) == '{' );
102  return std::next( ctx.begin( ), sz - 1 );

104  }

108  Ctx &ctx ) const {

110  if constexpr( std::is_same_v<daw::formatter_impl::DefaultCharT,
111  value_t> ) {
112  // format as string
113  auto out = std::copy( std::begin( c.container ),
114  std::end( c.container ), ctx.out( ) );
115  return out;
116  } else {

118  out = std::copy( std::data( c.Left ), daw::data_end( c.Left ), out );
119  bool is_first = true;
120  for( auto const &v : c.container ) {
121  if( is_first ) {
122  is_first = false;
123  } else {
124  out = std::copy( std::data( c.Separator ),
125  daw::data_end( c.Separator ), out );
126  }
127  if constexpr( daw::Range<daw::range_value_t<R>> ) {
128  auto t = daw::fmt_range( v );

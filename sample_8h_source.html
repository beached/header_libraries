19namespace daw::pipelines::pimpl {
20 template<typename RandomEngine,
21 typename Distribution = std::uniform_int_distribution<std::size_t>>
22 struct Sample_t {
23 std::size_t m_number_to_keep;
24 std::size_t m_sample_size;
25 RandomEngine m_engine = RandomEngine{ };
26
27 template<Range R>
28 [[nodiscard]] constexpr auto operator( )( R &&r ) const {
29 using result_type = typename Distribution::result_type;
30 static_assert( std::is_integral_v<result_type> );
31 daw_ensure( m_number_to_keep < m_sample_size );
32 daw_ensure(
33 std::cmp_less_equal( m_sample_size, max_value<result_type> ) );
34 auto distribution = Distribution{
35 result_type{ 1 }, static_cast<result_type>( m_sample_size ) };
36
37 return filter_view{ std::begin( r ), std::end( r ),
38 [engine = m_engine, number_to_keep = m_number_to_keep,
39 distribution]( auto const & ) mutable {
40 auto n = distribution( engine );
41 auto result = n <= number_to_keep;
42 return result;
43 } };
44 }
45 };
46} // namespace daw::pipelines::pimpl

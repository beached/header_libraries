24 template<
25 std::size_t StartN, typename... Args, std::size_t... Is,
26 std::enable_if_t<( sizeof...( Args ) >= ( sizeof...( Is ) + StartN ) ),
27 std::nullptr_t> = nullptr>
28 constexpr auto split_args_impl( std::tuple<Args...> &&args,
29 std::index_sequence<Is...> ) {
30 if constexpr( sizeof...( Args ) == 0 ) {
31 return std::tuple<>{ };
32 } else {
33 return std::tuple<std::decay_t<decltype( std::get<StartN + Is>( std::move(
34 args ) ) )>...>( std::get<StartN + Is>( std::move( args ) )... );
35 }
36 }

38 template<
39 std::size_t StartN, std::size_t EndN, typename... Args,
40 std::enable_if_t<( sizeof...( Args ) >= EndN ), std::nullptr_t> = nullptr>
41 constexpr auto split_args( std::tuple<Args...> args ) {
42 return split_args_impl<StartN>(
43 std::move( args ), std::make_index_sequence<EndN - StartN>{ } );
44 }

107 ( sizeof...( args ) * 4 ) ); // WAG for size
108 daw::string_view sv{ m_format_str };
109 result += sv.pop_front_until( '{' );
110 while( !sv.empty( ) ) {
111 auto const idx_str = sv.pop_front_until( '}' );
112 auto const idx =
113 daw::parser::parse_unsigned_int<uint8_t>( idx_str );
114 result += string_fmt_details::get_arg( idx, DAW_FWD( args )... );
115 result += sv.pop_front_until( '{' ).to_string( );
116 }
117 return result;
118 }

150 template<typename OutputIterator, typename... Args>
151 constexpr OutputIterator operator( )( OutputIterator out,
152 Args &&...args ) const {
153 return daw::visit_nt(
154 m_data,
155 [&out]( daw::basic_string_view<CharT> sv ) {
156 return daw::algorithm::copy( sv.begin( ), sv.end( ), out );
157 },
158 [&]( CharT c ) {
159 *out = c;
160 ++out;
161 return out;
162 },
163 [&]( size_t pos ) {
164 daw::pack_apply(
165 pos,
166 [&]( auto &&val ) {
167 using val_t = decltype( val );
168 if constexpr( std::is_convertible_v<
169 val_t,
170 daw::basic_string_view<CharT>> ) {
171 auto sv = daw::basic_string_view<CharT>( val );
172 out = daw::algorithm::copy( sv.begin( ), sv.end( ), out );
173 } else if constexpr( std::is_convertible_v<
174 val_t,
175 std::basic_string_view<CharT>> ) {
176 auto sv = std::basic_string_view<CharT>( val );
177 out = daw::algorithm::copy( sv.begin( ), sv.end( ), out );
178 } else if constexpr( std::is_same_v<
179 std::basic_string<CharT>,
180 daw::remove_cvref_t<val_t>> ) {
181 out =
182 daw::algorithm::copy( val.begin( ), val.end( ), out );
183 } else if constexpr( daw::can_to_os_string_int_v<
184 daw::remove_cvref_t<val_t>> ) {
185 out = daw::to_os_string_int<CharT>( out, val );
186 } else if constexpr( has_to_string_v<val_t> ) {
187 using daw::string_fmt::v1::string_fmt_details::to_string;
188 using std::to_string;
189 auto const str = to_string( DAW_FWD( val ) );
190 out =
191 daw::algorithm::copy( str.begin( ), str.end( ), out );
192 } else if constexpr( traits::is_streamable_v<std::istream &,
193 val_t> ) {
194 std::basic_stringstream<CharT> ss{ };
195 ss << val;
196 auto const str = ss.str( );
197 out =
198 daw::algorithm::copy( str.begin( ), str.end( ), out );
199 }
200 },
201 DAW_FWD( args )... );
202 return out;
203 } );

222 template<typename CharT, size_t N>
223 class fmt_t {
224 daw::bounded_vector_t<string_fmt_details::parse_token<CharT>, N / 2>
225 m_tokens;
226
227 constexpr static daw::
228 bounded_vector_t<string_fmt_details::parse_token<CharT>, N / 2>
229 parse_tokens( daw::basic_string_view<CharT> msg ) {
230 daw::bounded_vector_t<string_fmt_details::parse_token<CharT>, N / 2>
231 result{ };
232 size_t sz = 0;
233 while( !msg.empty( ) and sz < msg.size( ) ) {
234 assert( result.size( ) < result.capacity( ) );
235 switch( msg[sz] ) {
236 case static_cast<CharT>( '\\' ):
237 if( sz > 0 ) {
238 result.emplace_back( string_fmt_details::parse_token<CharT>(
239 msg.pop_front( sz ) ) );
240 sz = 0;
241 }
242 assert( !msg.empty( ) );
243 result.emplace_back(
244 string_fmt_details::parse_token<CharT>( msg.pop_front( ) ) );
245 continue;
246 case static_cast<CharT>( '{' ): {
247 if( sz > 0 ) {
248 result.emplace_back( string_fmt_details::parse_token<CharT>(
249 msg.pop_front( sz ) ) );
250 sz = 0;
251 }
252 msg.remove_prefix( );
253 assert( !msg.empty( ) );
254 auto const digit = daw::parser::parse_unsigned_int<size_t>(
255 msg.pop_front_until( '}' ) );
256 result.emplace_back(
257 string_fmt_details::parse_token<CharT>( digit ) );
258 continue;
259 }
260 }
261 ++sz;
262 }
263 if( sz > 0 ) {
264 assert( result.size( ) < result.capacity( ) );
265 result.emplace_back(
266 string_fmt_details::parse_token<CharT>( msg ) );
267 }
268 return result;
269 }

271 public:
272 explicit constexpr fmt_t( CharT const ( &fmt_string )[N] )
273 : m_tokens( parse_tokens( fmt_string ) ) {}
274
275 constexpr fmt_t( string_fmt_details::private_ctor,
276 CharT const *fmt_string )
277 : m_tokens(
278 parse_tokens( daw::basic_string_view<CharT>( fmt_string, N ) ) ) {
279 }

282 constexpr Result operator( )( Args &&...args ) const {
283 Result result{ };
284 if constexpr( string_fmt_details::has_reserve_v<Result> ) {
285 result.reserve( N * 2U );

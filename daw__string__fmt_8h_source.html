107  template<typename... Args>
108  std::string operator( )( Args &&...args ) const {
109  std::string result{ };
110  result.reserve( m_format_str.size( ) +
111  ( sizeof...( args ) * 4 ) ); // WAG for size
112  daw::string_view sv{ m_format_str };
113  result += sv.pop_front_until( '{' );
114  while( !sv.empty( ) ) {
115  auto const idx_str = sv.pop_front_until( '}' );
116  auto const idx =
117  daw::parser::parse_unsigned_int<uint8_t>( idx_str );
118  result +=
119  string_fmt_details::get_arg( idx, std::forward<Args>( args )... );
120  result += sv.pop_front_until( '{' ).to_string( );
121  }
122  return result;
123  }

246  constexpr static daw::bounded_vector_t<
247  string_fmt_details::parse_token<CharT>, N / 2>
248  parse_tokens( daw::basic_string_view<CharT> msg ) {
249  daw::bounded_vector_t<string_fmt_details::parse_token<CharT>, N / 2>
250  result{ };
251  size_t sz = 0;
252  while( !msg.empty( ) and sz < msg.size( ) ) {
253  assert( result.size( ) < result.capacity( ) );
254  switch( msg[sz] ) {
255  case static_cast<CharT>( '\\' ):
256  if( sz > 0 ) {
257  result.emplace_back( string_fmt_details::parse_token<CharT>(
258  msg.pop_front( sz ) ) );
259  sz = 0;
260  }
261  assert( !msg.empty( ) );
262  result.emplace_back(
263  string_fmt_details::parse_token<CharT>( msg.pop_front( ) ) );
264  continue;
265  case static_cast<CharT>( '{' ): {
266  if( sz > 0 ) {
267  result.emplace_back( string_fmt_details::parse_token<CharT>(
268  msg.pop_front( sz ) ) );
269  sz = 0;
270  }
271  msg.remove_prefix( );
272  assert( !msg.empty( ) );
273  auto const digit = daw::parser::parse_unsigned_int<size_t>(
274  msg.pop_front_until( '}' ) );
275  result.emplace_back(
276  string_fmt_details::parse_token<CharT>( digit ) );
277  continue;
278  }
279  }
280  ++sz;
281  }
282  if( sz > 0 ) {
283  assert( result.size( ) < result.capacity( ) );
284  result.emplace_back(
285  string_fmt_details::parse_token<CharT>( msg ) );
286  }
287  return result;
288  }

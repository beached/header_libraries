23 template<typename T, typename Allocator>
24 struct container_deleter : Allocator {
25 using alloc_traits = std::allocator_traits<Allocator>;
26 using pointer = typename alloc_traits::pointer;
27 std::size_t capacity = 0;
28
29 explicit container_deleter( ) = default;
30 explicit constexpr container_deleter(
31 Allocator alloc,
32 std::size_t
33 sz ) noexcept( std::is_nothrow_move_constructible_v<Allocator> )
34 : Allocator( std::move( alloc ) )
35 , capacity( sz ) {}
36
37 constexpr std::pair<Allocator, std::size_t> extract_allocator( ) {
38 return std::pair<Allocator, std::size_t>{
39 std::move( *static_cast<Allocator *>( this ) ),
40 daw::exchange( capacity, 0U ) };
41 }
42
43 constexpr void operator( )( pointer p ) {
44 auto count = capacity;
45 if constexpr( not std::is_trivially_destructible_v<
46 typename std::pointer_traits<pointer>::element_type> ) {
47 while( count-- > 0 ) {
48 alloc_traits::destroy( *this, std::addressof( p[count] ) );
49 }
50 }
51 alloc_traits::deallocate( *this, p, capacity );
52 capacity = 0;
53 }
54 };

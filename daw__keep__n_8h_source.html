20 enum class keep_n_order { ascending, descending };
21 namespace keep_n_impl {
22 template<keep_n_order Order, typename Function>
23 struct keep_n_pred {
24 Function func = { };
25
26 constexpr keep_n_pred( ) noexcept(
27 std::is_nothrow_constructible_v<Function> ) = default;
28 constexpr keep_n_pred( Function const &f ) noexcept(
29 std::is_nothrow_copy_constructible_v<Function> )
30 : func( f ) {}
31 constexpr keep_n_pred( Function &&f ) noexcept(
32 std::is_nothrow_move_constructible_v<Function> )
33 : func( std::move( f ) ) {}
34
35 template<typename... Args>
36 constexpr decltype( auto ) operator( )( Args &&...args ) {
37 if constexpr( Order == keep_n_order::ascending ) {

50 }
51 }

83 constexpr void insert( value_type const &v ) {
84 for( size_t n = 0; n < MaxItems; ++n ) {

83 [[nodiscard]] bool open( daw::string_view file,
84 open_mode mode = open_mode::read ) noexcept {
85
86 m_fdata.emplace( );
87 m_fdata->file =
88 ::open( file.data( ), mode == open_mode::read ? O_RDONLY : O_RDWR );
89 if( m_fdata->file < 0 ) {
90 return false;
91 }
92 {
93 auto const fsz = lseek( m_fdata->file, 0, SEEK_END );
94 lseek( m_fdata->file, 0, SEEK_SET );
95 if( fsz <= 0 ) {
96 m_fdata.reset( );
97 return false;
98 }
99 m_fdata->size = static_cast<size_type>( fsz );
100 }
101 m_fdata->ptr = static_cast<pointer>(
102 mmap( nullptr, m_fdata->size,
103 mode == open_mode::read ? PROT_READ : PROT_READ | PROT_WRITE,
104 MAP_SHARED, m_fdata->file, 0 ) );
105
106 if( m_fdata->ptr == MAP_FAILED ) {
107 m_fdata.reset( );
108 return false;
109 }
110 return true;
111 }

248 [[nodiscard]] bool open( daw::string_view file,
249 open_mode mode = open_mode::read ) noexcept {
250
251 m_fdata.emplace( );
252 {
253 HANDLE file_handle = ::CreateFileA(
254 file.data( ), mapfile_impl::CreateFileMode( mode ), 0, nullptr,
255 OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, nullptr );
256 if( file_handle == INVALID_HANDLE_VALUE ) {
257 return false;
258 }
259 LARGE_INTEGER fsz;
260 if( not::GetFileSizeEx( file_handle, &fsz ) or fsz.QuadPart <= 0 ) {
261 m_fdata.reset( );
262 return false;
263 }
264 m_fdata->size = static_cast<size_t>( fsz.QuadPart );
265 m_fdata->handle = ::CreateFileMapping(
266 file_handle, nullptr, mapfile_impl::PageMode( mode ), fsz.u.HighPart,
267 fsz.u.LowPart, nullptr );
268 if( m_fdata->handle == nullptr ) {
269 m_fdata.reset( );
270 return false;
271 }
272 CloseHandle( file_handle );
273 }
274 auto ptr = MapViewOfFile( m_fdata->handle, mapfile_impl::MapMode( mode ),
275 0, 0, 0 );
276 if( ptr == nullptr ) {
277 m_fdata.reset( );
278 return false;
279 }
280 m_fdata->ptr = static_cast<pointer>( ptr );
281 return true;
282 }

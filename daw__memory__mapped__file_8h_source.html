98  [[nodiscard]] bool open( std::string_view file,
99  open_mode mode = open_mode::read ) noexcept {
100 
101  m_file =
102  ::open( file.data( ), mode == open_mode::read ? O_RDONLY : O_RDWR );
103  if( m_file < 0 ) {
104  return false;
105  }
106  {
107  auto const fsz = lseek( m_file, 0, SEEK_END );
108  lseek( m_file, 0, SEEK_SET );
109  if( fsz <= 0 ) {
110  cleanup( );
111  return false;
112  }
113  m_size = static_cast<size_type>( fsz );
114  }
115  m_ptr = static_cast<pointer>(
116  mmap( nullptr, m_size,
117  mode == open_mode::read ? PROT_READ : PROT_READ | PROT_WRITE,
118  MAP_SHARED, m_file, 0 ) );
119 
120  if( m_ptr == MAP_FAILED ) {
121  m_ptr = nullptr;
122  cleanup( );
123  return false;
124  }
125  return true;
126  }

301  /***
302  * Open memory mapped file with file path supplied
303  * Caller is responsible for ensuring that the StringView is zero terminated
304  */
305  [[nodiscard]] bool open( std::string_view file,
306  open_mode mode = open_mode::read ) noexcept {
307 
308  {
309  HANDLE file_handle = ::CreateFileA(
310  file.data( ), mapfile_impl::CreateFileMode( mode ), 0, nullptr,
311  OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, nullptr );
312  if( file_handle == INVALID_HANDLE_VALUE ) {
313  return false;
314  }
315  LARGE_INTEGER fsz;
316  if( not ::GetFileSizeEx( file_handle, &fsz ) or fsz.QuadPart <= 0 ) {
317  cleanup( );
318  return false;
319  }
320  m_size = static_cast<size_t>( fsz.QuadPart );
321  m_handle = ::CreateFileMapping(
322  file_handle, nullptr, mapfile_impl::PageMode( mode ), fsz.u.HighPart,
323  fsz.u.LowPart, nullptr );
324  if( m_handle == nullptr ) {
325  cleanup( );
326  return false;
327  }
328  CloseHandle( file_handle );
329  }
330  auto ptr =
331  MapViewOfFile( m_handle, mapfile_impl::MapMode( mode ), 0, 0, 0 );
332  if( ptr == nullptr ) {
333  cleanup( );
334  return false;
335  }
336  m_ptr = static_cast<pointer>( ptr );
337  return true;
338  }

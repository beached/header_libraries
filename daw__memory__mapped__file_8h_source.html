88  if( m_fdata->file < 0 ) {
89  return false;
90  }
91  {

93  lseek( m_fdata->file, 0, SEEK_SET );
94  if( fsz <= 0 ) {
95  m_fdata.reset( );
96  return false;

247  [[nodiscard]] bool open( daw::string_view file,
248  open_mode mode = open_mode::read ) noexcept {
249 
250  m_fdata.emplace( );
251  {
252  HANDLE file_handle = ::CreateFileA(
253  file.data( ), mapfile_impl::CreateFileMode( mode ), 0, nullptr,
254  OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, nullptr );
255  if( file_handle == INVALID_HANDLE_VALUE ) {
256  return false;
257  }
258  LARGE_INTEGER fsz;
259  if( not ::GetFileSizeEx( file_handle, &fsz ) or fsz.QuadPart <= 0 ) {
260  m_fdata.reset( );

264  m_fdata->handle = ::CreateFileMapping(

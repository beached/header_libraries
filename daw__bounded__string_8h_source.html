366 template<size_t CapL, size_t CapR>
367 static constexpr int
368 compare( basic_bounded_string<CharT, CapL> const &lhs,
369 basic_bounded_string<CharT, CapR> const &rhs ) noexcept {
370 auto const str_compare = []( CharT const *p0, CharT const *p1,
371 size_t len ) {
372 auto const last = p0 + len;
373 while( p0 != last ) {
374 if( *p0 != *p1 ) {
375 if( *p0 < *p1 ) {
376 return -1;
377 }
378 return 1;
379 }
380 ++p0;
381 ++p1;
382 }
383 return 0;
384 };
385
386 auto cmp = str_compare( lhs.data( ), rhs.data( ),
387 (daw::min)( lhs.size( ), rhs.size( ) ) );
388 if( cmp == 0 ) {
389 if( lhs.size( ) < rhs.size( ) ) {
390 return -1;
391 }
392 if( rhs.size( ) < lhs.size( ) ) {
393 return 1;
394 }
395 }
396 return cmp;
397 }

401 CharT const *rhs_ptr,
402 size_t rhs_size ) noexcept {
403 auto const str_compare = []( CharT const *p0, CharT const *p1,
404 size_t len ) {
405 auto const last = p0 + len;
406 while( p0 != last ) {
407 if( *p0 != *p1 ) {
408 if( *p0 < *p1 ) {
409 return -1;
410 }
411 return 1;
412 }
413 ++p0;
414 ++p1;
415 }
416 return 0;
417 };
418

420 (daw::min)( lhs.size( ), rhs_size ) );
421 if( cmp == 0 ) {

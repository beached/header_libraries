364  template<size_t CapL, size_t CapR>
365  static constexpr int
366  compare( basic_bounded_string<CharT, CapL> const &lhs,
367  basic_bounded_string<CharT, CapR> const &rhs ) noexcept {
368  auto const str_compare = []( CharT const *p0, CharT const *p1,
369  size_t len ) {
370  auto const last = p0 + len;
371  while( p0 != last ) {
372  if( *p0 != *p1 ) {
373  if( *p0 < *p1 ) {
374  return -1;
375  }
376  return 1;
377  }
378  ++p0;
379  ++p1;
380  }
381  return 0;
382  };
383 
384  auto cmp = str_compare( lhs.data( ), rhs.data( ),
385  ( daw::min )( lhs.size( ), rhs.size( ) ) );
386  if( cmp == 0 ) {
387  if( lhs.size( ) < rhs.size( ) ) {
388  return -1;
389  }
390  if( rhs.size( ) < lhs.size( ) ) {
391  return 1;
392  }
393  }
394  return cmp;
395  }

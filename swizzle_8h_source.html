32 } };
33 } else if constexpr( is_tuple_like_v<range_value_t<R>> ) {
34 using std::get;

40 get<Indices>( DAW_FWD( v ) )... );
41 } };
42 } else {
43 // Not a Range
44 static_assert(
45 std::max( { Indices... } ) == 0,
46 "Swizzle requires only index 0 is valid for none tuple like types" );
47 return zip_view{ ( (void)Indices, DAW_FWD( r ) )... };
48 }
49 }

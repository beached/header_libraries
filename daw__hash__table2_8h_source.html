83  static constexpr size_t scale_hash( size_t hash, size_t table_size ) {
84  // Scale value to capacity using MAD(Multiply-Add-Divide) compression
85  // Use the two largest Prime's that fit in a 64bit unsigned integral
86  daw::exception::daw_throw_on_false( table_size > 0 );
87  daw::exception::daw_throw_on_false(
88  table_size <
89  max_size( ) ); // Table size must be less than max of ptrdiff_t as we
90  // use the value 0 as a sentinel. This should be rare
91  daw::exception::daw_throw_on_false( hash >=
92  impl::sentinals::sentinals_size );
93 
94  size_t const prime_a = 18446744073709551557u;
95  size_t const prime_b = 18446744073709551533u;
96  return ( hash * prime_a + prime_b ) % table_size;
97  }

99  constexpr auto lookup( size_t const hash ) const {
100  struct lookup_result_t {
101  size_t position;
102  size_t lookup_cost;
103  bool found;
104 
105  lookup_result_t( )
106  : position{ 0 }
107  , lookup_cost{ 0 }
108  , found{ false } {}
109  operator bool( ) const noexcept {
110  return found;
111  }
112  };
113  lookup_result_t result;
114  auto const s_hash = scale_hash( hash, m_hashes.size( ) );
115  result.position = s_hash;
116  size_t removed_found =
117  ( std::numeric_limits<std::size_t>::max )( ); // Need a check and this
118  // will be rare
119  for( ; result.position != m_hashes.size( ); ++result.position ) {
120  if( m_hashes[result.position] == hash ) {
121  result.found = true;
122  result.lookup_cost = result.position - s_hash;
123  return result;
124  } else if( m_hashes[result.position] == impl::sentinals::empty ) {
125  result.lookup_cost = result.position - s_hash;
126  return result;
127  } else if( m_hashes[result.position] == impl::sentinals::removed &&
128  result.position < removed_found ) {
129  removed_found = result.position;
130  }
131  }
132  result.position = 0;
133  for( ; result.position != s_hash; ++result.position ) {
134  if( m_hashes[result.position] == hash ) {
135  result.found = true;
136  result.lookup_cost =
137  ( m_hashes.size( ) - s_hash ) + result.position - s_hash;
138  return result;
139  } else if( m_hashes[result.position] == impl::sentinals::empty ) {
140  result.lookup_cost =
141  ( m_hashes.size( ) - s_hash ) + result.position - s_hash;
142  return result;
143  } else if( m_hashes[result.position] == impl::sentinals::removed &&
144  result.position < removed_found ) {
145  removed_found = result.position;
146  }
147  }
148  if( removed_found < ( std::numeric_limits<std::size_t>::max )( ) ) {
149  result.lookup_cost = m_hashes.size( );
150  result.position = removed_found;
151  return result;
152  }
153  result.position = m_hashes.size( ); // Indicate that there are no empty or
154  // removed cells, table is full
155  return result;
156  }

352 constructible_from<value_type,
353 iter_reference_type<ForwardIterator>> ) //
354 constexpr void assign( ForwardIterator first, ForwardIterator last ) {
355 auto const new_size =
356 static_cast<size_type>( std::distance( first, last ) );
357 if( new_size <= capacity( ) ) {
358 ForwardIterator mid = last;
359 bool growing = false;
360 if( new_size > size( ) ) {
361 growing = true;
362 mid = first;
363 std::advance( mid, size( ) );
364 }
365 pointer m = std::copy( first, mid, m_begin );
366 if( growing )
367 construct_at_end( mid, last, new_size - size( ) );
368 else
369 destruct_at_end( m );
370 } else {
371 vdeallocate( );
372 vallocate( recommend( new_size ) );
373 construct_at_end( first, last, new_size );
374 }

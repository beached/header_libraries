48  struct fnv1a_hash_t {
49  // TODO: check for UB if values are signed
50  template<typename Value, std::enable_if_t<std::is_integral_v<Value>,
51  std::nullptr_t> = nullptr>
52  [[nodiscard]] static constexpr fnv1a_uint_t
53  append_hash( fnv1a_uint_t current_hash, Value const &value ) noexcept {
54  for( fnv1a_uint_t n = 0; n < sizeof( Value ); ++n ) {
55  current_hash ^= static_cast<fnv1a_uint_t>(
56  ( static_cast<fnv1a_uint_t>( value ) &
57  ( static_cast<fnv1a_uint_t>( 0xFF ) << ( n * 8u ) ) ) >>
58  ( n * 8u ) );
59  current_hash *= impl::fnv_prime( );
60  }
61  return current_hash;
62  }

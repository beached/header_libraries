64  struct fnv1a_hash_t {
65  // TODO: check for UB if values are signed
66  template<typename Value, std::enable_if_t<std::is_integral_v<Value>,
67  std::nullptr_t> = nullptr>
68  [[nodiscard]] static constexpr fnv1a_uint_t
69  append_hash( fnv1a_uint_t current_hash, Value const &value ) noexcept {
70  for( fnv1a_uint_t n = 0; n < sizeof( Value ); ++n ) {
71  current_hash ^= static_cast<fnv1a_uint_t>(
72  ( static_cast<fnv1a_uint_t>( value ) &
73  ( fnv1a_uint_t{ 0xFFU } << ( n * 8U ) ) ) >>
74  ( n * 8U ) );
75  current_hash *= fnv1a_impl::fnv_prime;
76  }
77  return current_hash;
78  }

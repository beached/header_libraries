66 // TODO: check for UB if values are signed
67 template<typename Value, std::enable_if_t<std::is_integral_v<Value>,
68 std::nullptr_t> = nullptr>
69 [[nodiscard]] static constexpr fnv1a_uint_t
70 append_hash( fnv1a_uint_t current_hash, Value const &value ) noexcept {
71 for( fnv1a_uint_t n = 0; n < sizeof( Value ); ++n ) {
72 current_hash ^= static_cast<fnv1a_uint_t>(
73 ( static_cast<fnv1a_uint_t>( value ) &
74 ( fnv1a_uint_t{ 0xFFU } << (n * bit_count_v<char>)) ) >>
75 (n * bit_count_v<char>));
76 current_hash *= fnv1a_impl::fnv_prime;
77 }
78 return current_hash;
79 }

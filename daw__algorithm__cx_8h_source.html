25 template<typename ForwardIterator, typename LastType>
26 constexpr void rotate(
27 ForwardIterator first, ForwardIterator middle,
28 LastType last ) noexcept( noexcept( daw::cswap( *first, *middle ) ) ) {
29

31 traits::is_inout_iterator_test<ForwardIterator>( );

33 ForwardIterator tmp = middle;

35 daw::cswap( *first, *tmp );

37 ++tmp;

95 template<typename ForwardIterator, typename UnaryPredicate>
96 constexpr ForwardIterator partition( ForwardIterator first,
97 ForwardIterator last,
98 UnaryPredicate &&unary_predicate ) {
99 // Precondition checks
100 traits::is_forward_access_iterator_test<ForwardIterator>( );
101 traits::is_input_iterator_test<ForwardIterator>( );
102 traits::is_unary_predicate_test<UnaryPredicate, decltype( *first )>( );
103
104 first = find_if_not( first, last, unary_predicate );
105 if( first == last ) {
106 return first;
107 }
108
109 for( auto it = std::next( first ); it != last; ++it ) {
110 if( unary_predicate( *it ) ) {
111 daw::iter_swap( it, first );
112 ++first;
113 }
114 }
115 return first;
116 }

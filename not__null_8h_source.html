20  // not_null
21  //
22  // Restricts a pointer or smart pointer to only hold non-null values.
23  //
24  // Has zero size overhead over T.
25  //
26  // If T is a pointer ( i.e. T == U* ) then
27  // - allow construction from U*
28  // - disallow construction from null_ptr
29  // - disallow default construction
30  // - ensure construction from null U* fails
31  // - allow implicit conversion to U*
32  //
33  template<class T>
34  class [[deprecated( "use daw::not_null from daw_not_null.h" )]] not_null {
35  T m_ptr;
36 
37  public:
38  static_assert( std::is_assignable_v<T &, std::nullptr_t>,
39  "T cannot be assigned nullptr." );
40 
41  template<typename U, std::enable_if_t<std::is_convertible_v<U, T>,
42  std::nullptr_t> = nullptr>
43  constexpr not_null( U ptr )
44  : m_ptr( static_cast<T>( ptr ) ) {
45 
46  daw::exception::daw_throw_on_null( m_ptr, "Cannot be assigned nullptr" );
47  }
48 
49  template<
50  typename U,
51  std::enable_if_t<all_true_v<not std::is_same_v<daw::remove_cvref_t<U>,
52  daw::remove_cvref_t<T>>,
53  std::is_convertible_v<U, T>>,
54  std::nullptr_t> = nullptr>
55  constexpr not_null( not_null<U> const &other )
56  : not_null( static_cast<T>( other.get( ) ) ) {}

20 // not_null
21 //
22 // Restricts a pointer or smart pointer to only hold non-null values.
23 //
24 // Has zero size overhead over T.
25 //
26 // If T is a pointer ( i.e. T == U* ) then
27 // - allow construction from U*
28 // - disallow construction from null_ptr
29 // - disallow default construction
30 // - ensure construction from null U* fails
31 // - allow implicit conversion to U*
32 //
33 template<class T>
34 class [[deprecated( "use daw::not_null from daw_not_null.h" )]] not_null {
35 T m_ptr;
36
37 public:
38 static_assert( std::is_assignable_v<T &, std::nullptr_t>,
39 "T cannot be assigned nullptr." );
40
41 template<
42 typename U,
43 std::enable_if_t<std::is_convertible_v<U, T>, std::nullptr_t> = nullptr>
44 constexpr not_null( U ptr )
45 : m_ptr( static_cast<T>( ptr ) ) {
46
47 daw::exception::daw_throw_on_null( m_ptr, "Cannot be assigned nullptr" );
48 }
49
50 template<
51 typename U,
52 std::enable_if_t<all_true_v<not std::is_same_v<daw::remove_cvref_t<U>,
53 daw::remove_cvref_t<T>>,
54 std::is_convertible_v<U, T>>,
55 std::nullptr_t> = nullptr>
56 constexpr not_null( not_null<U> const &other )
57 : not_null( static_cast<T>( other.get( ) ) ) {}
58
59 [[nodiscard]] constexpr T get( ) const {
60 return m_ptr;
61 }
62
63 [[nodiscard]] constexpr operator T( ) const {
64 return get( );
65 }
66
67 [[nodiscard]] constexpr T operator->( ) const {
68 return get( );
69 }
70
71 // prevents compilation when someone attempts to assign a null pointer
72 // constant
73 not_null( std::nullptr_t ) = delete;
74 not_null &operator=( std::nullptr_t ) = delete;
75
76 // unwanted operators...pointers only point to single objects!
77 not_null &operator++( ) = delete;
78 not_null &operator--( ) = delete;
79 not_null operator++( int ) = delete;
80 not_null operator--( int ) = delete;
81 not_null &operator+=( std::ptrdiff_t ) = delete;
82 not_null &operator-=( std::ptrdiff_t ) = delete;
83 void operator[]( std::ptrdiff_t ) const = delete;
84 };

233 template<typename Fn, typename Projection = std::identity>
234 struct Map_t {
235 DAW_NO_UNIQUE_ADDRESS Fn m_func;
236 DAW_NO_UNIQUE_ADDRESS Projection m_projection = Projection{ };
237
238 [[nodiscard]] constexpr auto operator( )( auto &&r ) const {
239 using R = DAW_TYPEOF( r );
240 static_assert(
241 std::invocable<Projection, range_value_t<R>>,
242 "Projection must be invocable with the range_value_t<R>" );
243 using projected_t = std::invoke_result_t<Projection, range_value_t<R>>;
244
245 if constexpr( Range<R> ) {
246 static_assert( std::invocable<Fn, projected_t>,
247 "Map requires the function to be able to be called "
248 "with invoke and the range_reference_t(e.g. invoke( "
249 "MapFn, *it ) )" );
250 static_assert( traits::NoVoidResults<Fn, projected_t>,
251 "Map requires the result to not be void" );
252 return map_view<iterator_t<R>, Fn, Projection>(
253 std::begin( DAW_FWD( r ) ), std::end( DAW_FWD( r ) ), m_func,
254 m_projection );
255 } else {
256 static_assert( std::invocable<Fn, projected_t>,
257 "Map requires the function to be able to be called "
258 "with invoke and passed value" );
259 static_assert( traits::NoVoidResults<Fn, projected_t>,
260 "Map requires the result to not be void" );
261 return std::invoke( m_func,
262 std::invoke( m_projection, DAW_FWD( r ) ) );
263 }
264 }
265 };

272 template<typename Fn>
273 struct MapApply_t {
274 DAW_NO_UNIQUE_ADDRESS Fn m_func;
275
276 [[nodiscard]] constexpr auto operator( )( auto &&r ) const {
277 using R = DAW_TYPEOF( r );
278 static_assert( traits::is_applicable_v<Fn, range_reference_t<R>>,
279 "MapApply requires the function to be able to be called "
280 "with apply and the range_reference_t" );
281 auto func = m_func;
282 return map_view( std::begin( r ), std::end( r ), [=]( auto &&tp ) {
283 return std::apply( func, tp );

285 }

290 template<typename T, typename Compare>
291 struct Clamp_t {
292 T lo;
293 T hi;
294 DAW_NO_UNIQUE_ADDRESS Compare compare;
295
296 [[nodiscard]] constexpr auto operator( )( auto &&r ) const {
297 using R = DAW_TYPEOF( r );
298 auto h = hi;
299 auto l = lo;
300 auto c = compare;
301 if constexpr( Range<R> ) {
302 using value_type = range_value_t<R>;
303 static_assert( std::convertible_to<T, value_type>,
304 "Clamp requires a lo/hi values convertible to the "
305 "range value type" );
306 return map_view( std::begin( r ), std::end( r ),
307 [=]( value_type const &v ) {
308 return std::clamp( v, l, h, c );
309 } );
310 } else {
311 static_assert(
312 std::convertible_to<T, daw::remove_cvref_t<R>>,
313 "Clamp requires a lo/hi values convertible to the value type" );
314 return std::clamp( r, l, h, c );
315 }
316 }
317 };

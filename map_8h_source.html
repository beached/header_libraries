230  template<typename Fn, typename Projection = std::identity>
231  struct Map_t {
232  DAW_NO_UNIQUE_ADDRESS mutable Fn m_func;
233  DAW_NO_UNIQUE_ADDRESS Projection m_projection = Projection{ };
234 
235  [[nodiscard]] constexpr auto operator( )( auto &&r ) const {
236  using R = DAW_TYPEOF( r );
237  static_assert(
238  std::invocable<Projection, range_value_t<R>>,
239  "Projection must be invocable with the range_value_t<R>" );
240  using projected_t = std::invoke_result_t<Projection, range_value_t<R>>;
241 
242  if constexpr( Range<R> ) {
243  static_assert( std::invocable<Fn, projected_t>,
244  "Map requires the function to be able to be called "
245  "with invoke and the range_reference_t(e.g. invoke( "
246  "MapFn, *it ) )" );
247  static_assert( traits::NoVoidResults<Fn, projected_t>,
248  "Map requires the result to not be void" );
249  return map_view<iterator_t<R>, Fn, Projection>(
250  std::begin( DAW_FWD( r ) ),
251  std::end( DAW_FWD( r ) ),
252  m_func,
253  m_projection );
254  } else {
255  static_assert( std::invocable<Fn, projected_t>,
256  "Map requires the function to be able to be called "
257  "with invoke and passed value" );
258  static_assert( traits::NoVoidResults<Fn, projected_t>,
259  "Map requires the result to not be void" );
260  return std::invoke( m_func,
261  std::invoke( m_projection, DAW_FWD( r ) ) );
262  }
263  }
264  };

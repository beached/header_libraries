42 private:
43 pointer m_pointer = { };
44 std::span<T> m_range = { };
45 bool m_destroyed = false;

47 [[nodiscard]] constexpr std::size_t pointer_index( ) const {
48 if( not m_pointer ) {
49 DAW_SAFE_POINTER_ENSURE( not m_range.data( ) );
50 return 0;
51 }
52 return static_cast<std::size_t>(
53 std::distance( m_range.data( ), m_pointer ) );
54 }
55
56 [[nodiscard]] constexpr bool pointer_in_range( pointer p ) const {
57 return ( not get_base( ) and not p ) or
58 ( std::less_equal<>{ }( get_base( ), p ) and
59 std::less<>{ }( p, daw::data_end( m_range ) ) );
60 }
61
62 [[nodiscard]] constexpr bool same_range( safe_pointer const &rhs ) const {
63 if( get_base( ) == rhs.get_base( ) and
64 get_base_size( ) == rhs.get_base_size( ) ) {
65 return true;
66 }
67 if( get_base( ) and not rhs.get_base( ) and rhs.get( ) ) {
68 return pointer_in_range( rhs.get( ) );
69 }
70 return not get_base( ) and not rhs.get_base( );
71 }
72
73 [[nodiscard]] constexpr bool can_add( difference_type n ) const {
74 if( n < 0 ) {
75 return can_sub( -n );
76 }
77 return static_cast<std::size_t>( n ) <=
78 ( m_range.size( ) - pointer_index( ) );
79 }
80
81 [[nodiscard]] constexpr bool can_sub( difference_type n ) const {
82 if( n < 0 ) {
83 return can_add( -n );
84 }
85 return static_cast<std::size_t>( n ) <= pointer_index( );
86 }

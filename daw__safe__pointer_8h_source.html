46  [[nodiscard]] constexpr std::size_t pointer_index( ) const {
47  if( not m_pointer ) {
48  DAW_SAFE_POINTER_ENSURE( not m_range.data( ) );
49  return 0;
50  }
51  return static_cast<std::size_t>(
52  std::distance( m_range.data( ), m_pointer ) );
53  }
54 
55  [[nodiscard]] constexpr bool pointer_in_range( pointer p ) const {
56  return ( not get_base( ) and not p ) or
57  ( std::less_equal<>{ }( get_base( ), p ) and
58  std::less<>{ }( p, daw::data_end( m_range ) ) );
59  }
60 
61  [[nodiscard]] constexpr bool same_range( safe_pointer const &rhs ) const {
62  if( get_base( ) == rhs.get_base( ) and
63  get_base_size( ) == rhs.get_base_size( ) ) {
64  return true;
65  }
66  if( get_base( ) and not rhs.get_base( ) and rhs.get( ) ) {
67  return pointer_in_range( rhs.get( ) );
68  }
69  return not get_base( ) and not rhs.get_base( );
70  }

72  [[nodiscard]] constexpr bool can_add( difference_type n ) const {
73  if( n < 0 ) {
74  return can_sub( -n );
75  }
76  return static_cast<std::size_t>( n ) <=
77  ( m_range.size( ) - pointer_index( ) );
78  }

80  [[nodiscard]] constexpr bool can_sub( difference_type n ) const {
81  if( n < 0 ) {
82  return can_add( -n );
83  }
84  return static_cast<std::size_t>( n ) <= pointer_index( );
85  }

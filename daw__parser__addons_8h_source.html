59 constexpr void parse_int( ForwardIterator first, ForwardIterator last,
60 Result &result ) {
61 intmax_t count = std::numeric_limits<Result>::digits;
62 result = 0;
63 bool is_neg = false;
64 if( '-' == *first ) {
65 daw::exception::precondition_check<ParserOutOfRangeException>(
66 std::numeric_limits<Result>::is_signed,
67 "Negative values are unsupported with unsigned Result" );
68
69 is_neg = true;
70 ++first;
71 }
72 for( ; first != last and count > 0; ++first, --count ) {
73 result *= static_cast<Result>( 10 );
74 Result val = *first - '0';
75 result += val;
76 }
77 daw::exception::precondition_check<ParserOutOfRangeException>(
78 first == last, "Not enough room to store signed integer" );
79 if( is_neg ) {
80 result *= static_cast<Result>( -1 );
81 }

84 template<typename ForwardIterator>
85 constexpr auto parse_string_literal( ForwardIterator first,
86 ForwardIterator const last ) {
87 auto result = trim_left( first, last );
88 auto quote_char = *first;
89
90 daw::exception::precondition_check<ParserException>(
91 is_quote( quote_char ) );
92
93 auto it = result.first;
94 auto last_it = it;
95 ++it;
96 bool found = false;
97 while( it != last ) {
98 if( ( found = is_a( *it, quote_char ) ) and !is_escape( *last_it ) ) {
99 break;
100 }
101 last_it = it;
102 ++it;
103 }
104 daw::exception::precondition_check<ParserException>( found );
105
106 ++result.first; // trim quote
107 result.last = it;
108 result.found = true;
109 return result;
110 }

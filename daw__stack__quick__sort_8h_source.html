65  auto [first, last] = sort_stack.back( );
66  sort_stack.pop_back( );
67 
68  auto const dist = std::distance( first, last );
69  if( dist <= 1 ) {
70  return;
71  }
72  auto const mid = std::next( first, dist / 2 );
73  auto const pivot =
74  quick_sort_impl::median( *first, *mid, *std::prev( last ), cmp );

78

20 template<typename T, size_t N, typename Idx = uint16_t>
21 struct bounded_stack {
22 T values[N];
23 Idx idx = 0;
24
25 template<typename... Args>
26 constexpr void push_back( Args &&...args ) {
27 values[idx] = T{ DAW_FWD( args )... };
28 ++idx;
29 }

31 constexpr T &back( ) {
32 return values[idx - 1];
33 }

35 constexpr void pop_back( ) {
36 --idx;
37 }

39 constexpr bool has_more( ) const {

44 template<typename T, typename Compare>
45 constexpr T median( T const &a, T const &b, T const &c, Compare cmp ) {
46 if( cmp( a, b ) != cmp( a, c ) ) {
47 return a;
48 } else if( cmp( a, b ) != cmp( c, b ) ) {
49 return b;
50 }
51 return c;

56 template<typename RandomIterator, typename Compare = std::less<>>
57 constexpr void quick_sort( RandomIterator f, RandomIterator l,
58 Compare cmp = Compare{ } ) {
59
60 constexpr size_t tree_size = bit_count_v<std::size_t> * 2U;
61
62 quick_sort_impl::bounded_stack<std::pair<RandomIterator, RandomIterator>,
63 tree_size * 2U>
64 sort_stack;
65 sort_stack.push_back( f, l );
66
67 while( sort_stack.has_more( ) ) {
68 auto [first, last] = sort_stack.back( );
69 sort_stack.pop_back( );
70
71 auto const dist = std::distance( first, last );
72 if( dist <= 1 ) {
73 return;
74 }
75 auto const mid = std::next( first, dist / 2 );
76 auto const pivot =
77 quick_sort_impl::median( *first, *mid, *std::prev( last ), cmp );
78
79 auto const less_pivot =
80 std::partition( first, last, [=]( auto const &v ) {
81 return cmp( v, pivot );
82 } );
83
84 auto const split2 =
85 std::partition( less_pivot, last, [=]( auto const &v ) {
86 return not cmp( pivot, v );
87 } );
88
89 sort_stack.push_back( split2, last );
90 sort_stack.push_back( first, less_pivot );
91 }

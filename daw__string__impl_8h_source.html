28 ForwardIt2 s_first,
29 ForwardIt2 s_last ) {
30 for( ;; ++first ) {
31 ForwardIt1 it = first;
32 for( ForwardIt2 s_it = s_first;; ++it, ++s_it ) {

196 template<typename OStream,
197 std::enable_if_t<daw::traits::is_ostream_like_lite_v<OStream>,
198 std::nullptr_t> = nullptr>
199 inline void sv_insert_fill_chars( OStream &os, std::size_t n ) {
200 using CharT = typename OStream::char_type;
201 static_assert( traits::has_write_member_v<OStream, CharT>,
202 "OStream Must has write member" );
203
204 std::array<CharT, 8> fill_chars = { 0 };
205 fill_chars.fill( os.fill( ) );
206
207 for( ; n >= fill_chars.size( ) and os.good( ); n -= fill_chars.size( ) ) {
208 os.write( fill_chars.data( ),
209 static_cast<intmax_t>( fill_chars.size( ) ) );
210 }
211 if( n > 0 and os.good( ) ) {
212 os.write( fill_chars.data( ), static_cast<intmax_t>( n ) );
213 }
214 }

216 template<typename OStream,
217 typename CharT,
218 typename Bounds,
219 std::ptrdiff_t Ex,
220 std::enable_if_t<daw::traits::is_ostream_like_v<OStream, CharT>,
221 std::nullptr_t> = nullptr>
222 void
223 sv_insert_aligned( OStream &os,
224 daw::sv1::basic_string_view<CharT, Bounds, Ex> str ) {
225 auto const size = str.size( );
226 auto const alignment_size =
227 static_cast<std::size_t>( os.width( ) ) - size;
228 bool const align_left =
229 ( os.flags( ) & OStream::adjustfield ) == OStream::left;
230 if( !align_left ) {
231 sv_insert_fill_chars( os, alignment_size );
232 if( os.good( ) ) {
233 os.write( str.data( ), static_cast<intmax_t>( size ) );
234 }
235 } else {
236 os.write( str.data( ), static_cast<intmax_t>( size ) );
237 if( os.good( ) ) {
238 sv_insert_fill_chars( os, alignment_size );
239 }
240 }

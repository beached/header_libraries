145 template<bool use_invoke = true, typename F, typename Tuple>
146 [[nodiscard, maybe_unused]] inline constexpr decltype( auto )
147 apply( F &&f, Tuple &&t ) {
148 static_assert( is_tuple_v<Tuple>,
149 "Attempt to call apply with invalid arguments. The "
150 "arguments must be a std::tuple" );
151 if constexpr( use_invoke ) {
152 if constexpr( std::tuple_size_v<Tuple> == 0 ) {
153 return daw::invoke( DAW_FWD( f ) );
154 } else {
155 return cpp_17_details::apply_details(
156 DAW_FWD( f ), DAW_FWD( t ),
157 std::make_index_sequence<std::tuple_size_v<std::decay_t<Tuple>>>{ } );
158 }
159 } else {
160 constexpr std::size_t tp_size = std::tuple_size_v<Tuple>;
161 if constexpr( tp_size == 0 ) {
162 return DAW_FWD( f )( );
163 } else {
164 return cpp_17_details::apply_impl2(
165 DAW_FWD( f ), DAW_FWD( t ), std::make_index_sequence<tp_size>{ } );

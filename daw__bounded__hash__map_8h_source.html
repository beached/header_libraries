324 static constexpr size_t scale_hash( size_t hash, size_t range_size ) {
325 std::uint64_t const prime_a = 18446744073709551557ULL;
326 std::uint64_t const prime_b = 18446744073709551533ULL;
327 return static_cast<std::size_t>( ( hash * prime_a + prime_b ) %
328 range_size );

331 constexpr daw::optional<size_type> find_index( Key const &key ) const {
332 size_type const hash = Hash{ }( key );
333 size_type const scaled_hash = scale_hash( hash, capacity( ) );
334
335 for( size_type n = scaled_hash; n < capacity( ); ++n ) {
336 if( not m_data[n] or key_equal{ }( m_data[n].kv.key, key ) ) {
337 return n;
338 }
339 }
340 for( size_type n = 0; n < scaled_hash; ++n ) {
341 if( not m_data[n] or key_equal{ }( m_data[n].kv.key, key ) ) {
342 return n;
343 }
344 }
345 return { };
346 }

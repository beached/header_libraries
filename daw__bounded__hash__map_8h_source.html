323  static constexpr size_t scale_hash( size_t hash, size_t range_size ) {
324  std::uint64_t const prime_a = 18446744073709551557ULL;
325  std::uint64_t const prime_b = 18446744073709551533ULL;
326  return static_cast<std::size_t>( ( hash * prime_a + prime_b ) %

330  constexpr daw::optional<size_type> find_index( Key const &key ) const {
331  size_type const hash = Hash{ }( key );
332  size_type const scaled_hash = scale_hash( hash, capacity( ) );
333 
334  for( size_type n = scaled_hash; n < capacity( ); ++n ) {
335  if( not m_data[n] or key_equal{ }( m_data[n].kv.key, key ) ) {
336  return n;
337  }
338  }
339  for( size_type n = 0; n < scaled_hash; ++n ) {
340  if( not m_data[n] or key_equal{ }( m_data[n].kv.key, key ) ) {
341  return n;
342  }
343  }
344  return { };
345  }

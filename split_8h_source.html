49 template<ForwardIterator I, typename IteratorLast, typename Delimiter,
50 typename Projection>
51 struct split_view_iterator {
52 using iterator_category = std::forward_iterator_tag;
53 using value_type = range_t<I, I>;
54 using reference = value_type;
55 using const_reference = value_type;
56 using difference_type = std::ptrdiff_t;
57 using i_am_a_daw_split_view_iterator = void;
58
59 private:
60 DAW_NO_UNIQUE_ADDRESS I m_first = I{ };
61 DAW_NO_UNIQUE_ADDRESS IteratorLast m_last = IteratorLast{ };
62 DAW_NO_UNIQUE_ADDRESS Delimiter m_delimiter{ };
63 DAW_NO_UNIQUE_ADDRESS Projection m_projection{ };
64 mutable std::optional<I> m_next = std::nullopt;
65
66 constexpr void increment_iterator( ) const {
67 if( m_next or m_first == m_last ) {
68 return;
69 }
70 if constexpr( std::is_invocable_v<Delimiter, iter_reference_t<I>> ) {
71 m_next = daw::pipelines::FindIf( range_t{ m_first, m_last },
72 m_delimiter, m_projection );
73 } else if constexpr( Range<Delimiter> ) {
74 m_next = std::search( m_first, m_last, std::begin( m_delimiter ),
75 std::end( m_delimiter ) );
76 } else {
77 m_next = daw::pipelines::Find( range_t{ m_first, m_last }, m_delimiter,
78 m_projection );
79 }
80 assert( m_next );
81 }
82
83 constexpr auto get_delimiter_length( ) const {
84 return pimpl::delimiter_length_t<Delimiter>{ m_delimiter }.size( );
85 }
86
87 public:
88 split_view_iterator( ) = default;
89
90 constexpr split_view_iterator( I first, IteratorLast last, Delimiter d,
91 Projection p )
92 : m_first( std::move( first ) )
93 , m_last( std::move( last ) )
94 , m_delimiter( std::move( d ) )
95 , m_projection( std::move( p ) ) {
96 increment_iterator( );
97 }
98
99 constexpr split_view_iterator &operator++( ) {
100 assert( m_next );
101 m_first = std::move( *m_next );
102 m_next.reset( );
103 if( m_first != m_last ) {
104 std::advance( m_first, get_delimiter_length( ) );
105 }
106 return *this;
107 }
108
109 constexpr split_view_iterator operator++( int ) {
110 auto result = *this;
111 operator++( );
112 return result;
113 }
114
115 constexpr const_reference operator*( ) const {
116 increment_iterator( );
117 assert( m_next );
118 return const_reference{ m_first, *m_next };
119 }
120
121 constexpr bool operator==( split_view_iterator const &rhs ) const {
122 return m_first == rhs.m_first and m_last == rhs.m_last and
123 m_next == rhs.m_next;
124 }
125 constexpr bool operator!=( split_view_iterator const &rhs ) const = default;
126 };
